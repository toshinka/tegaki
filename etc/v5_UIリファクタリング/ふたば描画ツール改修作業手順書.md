#### 📝 preset-manager.js実装ガイド
```javascript
// 新規作成: ui/preset-manager.js
// 役割: プリセット管理専門システム

export class PresetManager {
    constructor(toolsSystem, historyManager) {
        this.toolsSystem = toolsSystem;
        this.historyManager = historyManager;
        this.activePresetId = null;
        this.presets = this.createDefaultPresets();
    }
    
    // Phase2要件: デフォルトプリセット生成
    createDefaultPresets() {
        return CONFIG.SIZE_PRESETS.map((size, index) => ({
            id: `preset_${index}`,
            size: size,
            opacity: CONFIG.DEFAULT_OPACITY, // 100%統一
            originalSize: size
        }));
    }
    
    // Phase2要件: プリセット選択（履歴記録付き）
    selectPreset(presetId) {
        const beforeState = this.captureCurrentState();
        
        const preset = this.presets.find(p => p.id === presetId);
        if (!preset) return false;
        
        this.activePresetId = presetId;
        this.applyPresetToTools(preset);
        
        // 履歴記録
        const afterState = this.captureCurrentState();
        this.historyManager?.recordPresetChange(beforeState, afterState);
        
        return true;
    }
    
    // Phase2要件: リセット機能
    resetActivePreset() {
        if (!this.activePresetId) return;
        
        const beforeState = this.captureCurrentState();
        const preset = this.presets.find(p => p.id === this.activePresetId);
        
        // デフォルト値に復元
        preset.size = preset.originalSize;
        preset.opacity = CONFIG.DEFAULT_OPACITY;
        
        this.applyPresetToTools(preset);
        
        // 履歴記録
        const afterState = this.captureCurrentState();
        this.historyManager?.recordPresetChange(beforeState, afterState);
    }
    
    // Phase2要件: 全プリセットリセット
    resetAllPresets() {
        const beforeState = this.captureCurrentState();
        
        this.presets.forEach(preset => {
            preset.size = preset.originalSize;
            preset.opacity = CONFIG.DEFAULT_OPACITY;
        });
        
        if (this.activePresetId) {
            const activePreset = this.presets.find(p => p.id === this.activePresetId);
            this.applyPresetToTools(activePreset);
        }
        
        // 履歴記録
        const afterState = this.captureCurrentState();
        this.historyManager?.recordPresetChange(beforeState, afterState);
    }
    
    // プリセット適用
    applyPresetToTools(preset) {
        this.toolsSystem.updateBrushSettings({
            size: preset.size,
            opacity: preset.opacity
        });
    }
    
    // 状態キャプチャ（履歴用）
    captureCurrentState() {
        return {
            activePresetId: this.activePresetId,
            presets: JSON.parse(JSON.stringify(this.presets))
        };
    }
}
```

### Phase 2C: パフォーマンス監視分離ガイド

#### 📝 performance-monitor.js実装ガイド
```javascript
// 新規作成: ui/performance-monitor.js
// 役割: パフォーマンス監視専門システム

export class PerformanceMonitorSystem {
    constructor() {
        this.isRunning = false;
        this.stats = {
            fps: 0,
            frameTime: 0,
            memoryUsage: 0,
            gpuUsage: 0 // 推定値
        };
        this.fpsHistory = [];
        this.updateCallbacks = [];
    }
    
    start() {
        if (this.isRunning) return;
        
        this.isRunning = true;
        this.lastTime = performance.now();
        this.frameCount = 0;
        
        // FPS計測開始
        this.measureFPS();
        
        // 統計更新定期実行
        this.statsInterval = setInterval(() => {
            this.updateStats();
        }, CONFIG.PERFORMANCE_UPDATE_INTERVAL);
    }
    
    stop() {
        this.isRunning = false;
        if (this.statsInterval) {
            clearInterval(this.statsInterval);
        }
    }
    
    // FPS計測
    measureFPS() {
        if (!this.isRunning) return;
        
        const currentTime = performance.now();
        const deltaTime = currentTime - this.lastTime;
        
        this.frameCount++;
        
        // 1秒ごとにFPS計算
        if (deltaTime >= 1000) {
            this.stats.fps = Math.round(this.frameCount * 1000 / deltaTime);
            this.stats.frameTime = Math.round(deltaTime / this.frameCount * 100) / 100;
            
            this.fpsHistory.push(this.stats.fps);
            if (this.fpsHistory.length > 60) {
                this.fpsHistory.shift();
            }
            
            this.lastTime = currentTime;
            this.frameCount = 0;
        }
        
        requestAnimationFrame(() => this.measureFPS());
    }
    
    // 統計更新
    updateStats() {
        // メモリ使用量取得
        if (performance.memory) {
            this.stats.memoryUsage = Math.round(
                performance.memory.usedJSHeapSize / 1024 / 1024 * 100
            ) / 100;
        }
        
        // GPU使用率推定（描画コール数ベース）
        this.stats.gpuUsage = this.estimateGPUUsage();
        
        // コールバック通知
        this.updateCallbacks.forEach(callback => {
            callback(this.stats);
        });
    }
    
    // GPU使用率推定
    estimateGPUUsage() {
        // FPSが低い = GPU負荷高いと推定
        const avgFps = this.fpsHistory.reduce((sum, fps) => sum + fps, 0) / 
                      Math.max(this.fpsHistory.length, 1);
        return Math.max(0, 100 - (avgFps / CONFIG.TARGET_FPS * 100));
    }
    
    // 更新コールバック登録
    addUpdateCallback(callback) {
        this.updateCallbacks.push(callback);
    }
    
    // 統計取得
    getStats() {
        return { ...this.stats };
    }
}
```

// 新規作成: ui/ui-events.js
// 役割: UIイベント処理専門システム

export class UIEventSystem {
    constructor(uiManager, historyManager, presetManager) {
        this.uiManager = uiManager;
        this.historyManager = historyManager;
        this.presetManager = presetManager;
        
        this.shortcuts = new Map();
        this.setupDefaultShortcuts();
    }
    
    init() {
        this.attachEventListeners();
    }
    
    // デフォルトショートカット設定
    setupDefaultShortcuts() {
        // Ctrl+Z: アンドゥ
        this.shortcuts.set('Control+KeyZ', () => {
            this.historyManager.undo();
        });
        
        // Ctrl+Y: リドゥ
        this.shortcuts.set('Control+KeyY', () => {
            this.historyManager.redo();
        });
        
        // R: リセット
        this.shortcuts.set('KeyR', () => {
            this.presetManager.resetActivePreset();
        });
        
        // 1-6: プリセット選択
        for (let i = 1; i <= 6; i++) {
            this.shortcuts.set(`Digit${i}`, () => {
                this.presetManager.selectPreset(`preset_${i-1}`);
            });
        }
    }
    
    // イベントリスナー設定
    attachEventListeners() {
        document.addEventListener('keydown', (e) => {
            this.handleKeyDown(e);
        });
        
        document.addEventListener('keyup', (e) => {
            this.handleKeyUp(e);
        });
    }
    
    // キーダウン処理
    handleKeyDown(event) {
        const key = this.createKeyString(event);
        const handler = this.shortcuts.get(key);
        
        if (handler) {
            event.preventDefault();
            handler();
        }
    }
    
    // キー文字列生成
    createKeyString(event) {
        const modifiers = [];
        if (event.ctrlKey) modifiers.push('Control');
        if (event.shiftKey) modifiers.push('Shift');
        if (event.altKey) modifiers.push('Alt');
        
        return [...modifiers, event.code].join('+');
    }
}
