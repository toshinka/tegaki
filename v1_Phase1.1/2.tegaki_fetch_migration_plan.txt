```javascript
export class VectorTransform {
    applyTransform(layer, transform) {
        // ✅ ベクター変形（非破壊）
        layer.transform.setFromMatrix(transform);
        
        // ✅ 履歴記録（完全復元可能）

            beforeState: layer.getOriginalState()

    validateVectorQuality(graphics) {
        return graphics.geometry.points.length > 0;
```

```javascript
export class BucketTool {
    constructor(toolManager) {
        this.toolManager = toolManager;
        this.tolerance = 32; // 色の許容範囲
    }
    
    async fill(x, y, color) {
        // Lodash活用でピクセル処理最適化
        const targetPixels = await this.getTargetPixels(x, y);
        const fillQueue = _.cloneDeep(targetPixels);
        
        // 非破壊フィル実装
        const fillGraphics = new PIXI.Graphics();
        fillGraphics.beginFill(color);
        
        // フラッドフィル アルゴリズム
        while (fillQueue.length > 0) {
            const pixel = fillQueue.pop();
            this.fillPixel(fillGraphics, pixel.x, pixel.y);
        }
        
        fillGraphics.endFill();
        return fillGraphics;
    }
}
```

```javascript
// js/animation/onion-skin.js
export class OnionSkin {
    constructor(frameManager) {
        this.frameManager = frameManager;
        this.enabled = true;
        this.beforeFrames = 2;  // 前フレーム表示数
        this.afterFrames = 2;   // 後フレーム表示数
        this.opacity = 0.3;     // オニオンスキン透明度
    }
    
    updateOnionSkin() {
        if (!this.enabled) return;
        
        const currentIndex = this.frameManager.currentFrame;
        
        // 前フレーム表示
        for (let i = 1; i <= this.beforeFrames; i++) {
            const prevIndex = currentIndex - i;
            if (prevIndex >= 0) {
                this.showFrameAsOnion(prevIndex, this.opacity * (1 - i * 0.1), 0xFF0000); // 赤系
            }
        }
        
        // 後フレーム表示
        for (let i = 1; i <= this.afterFrames; i++) {
            const nextIndex = currentIndex + i;
            if (nextIndex < this.frameManager.frames.length) {
                this.showFrameAsOnion(nextIndex, this.opacity * (1 - i * 0.1), 0x0000FF); // 青系
            }
        }
    }
}
```

```javascript
// js/animation/gif-export.js
export class GIFExporter {
    constructor(frameManager) {
        this.frameManager = frameManager;
    }
    
    async exportGIF(options = {}) {
        const {
            width = 400,
            height = 400,
            quality = 10,
            repeat = 0  // 0 = 無限ループ
        } = options;
        
        // GIF.js（CDN追加検討）を使用
        const gif = new GIF({
            workers: 2,
            quality: quality,
            width: width,
            height: height,
            repeat: repeat
        });
        
        // 各フレームをGIFに追加
        for (const frame of this.frameManager.frames) {
            const canvas = await this.frameToCanvas(frame, width, height);
            gif.addFrame(canvas, { delay: frame.duration });
        }
        
        return new Promise((resolve, reject) => {
            gif.on('finished', (blob) => {
                resolve(blob);
            });
            
            gif.on('error', reject);
            gif.render();
        });
    }
    
    async frameToCanvas(frame, width, height) {
        // フレームを高品質Canvasに変換
        const tempApp = new PIXI.Application({
            width: width,
            height: height,
            backgroundColor: 0xf0e0d6,
            preserveDrawingBuffer: true
        });
        
        // フレームレイヤーを描画
        frame.layers.forEach(layer => {
            tempApp.stage.addChild(layer.graphics);
        });
        
        // レンダリング待機
        tempApp.render();
        
        // Canvasデータ取得
        const canvas = tempApp.renderer.extract.canvas(tempApp.stage);
        
        // リソース解放
        tempApp.destroy();
        
        return canvas;
    }
}
```

```javascript
export class V8Compatibility {
    static async createApplication(options) {
        // バージョン検出
        const isV8 = PIXI.VERSION.startsWith('8');
        
        if (isV8) {
            // v8モード
            const app = new PIXI.Application();
            await app.init({
                ...options,
                preference: 'webgpu'  // WebGPU優先
            });
            return app;
        } else {
            // v7互換モード
            return new PIXI.Application(options);
        }
    }
    
    static setupRenderer(app) {
        const isV8 = PIXI.VERSION.startsWith('8');
        
        if (isV8) {
            // v8: WebGPU Renderer設定
            if (app.renderer.type === PIXI.RENDERER_TYPE.WEBGPU) {
                this.setupWebGPUOptimizations(app);
            }
        }
    }
    
    static setupWebGPUOptimizations(app) {
        // 📋 V8_FEATURE: WebGPU最適化
        app.renderer.context.device.createComputePassDescriptor({
            label: 'Drawing Optimization Pass'
        });
        
        // 120FPS設定
        app.ticker.maxFPS = 120;
    }
}
```

```javascript
        return new Promise((resolve, reject) => {
            const script = document.createElement('script');
            script.src = 'https://cdn.jsdelivr.net/npm/pixi-spine@4.0.0/dist/pixi-spine.js';
            script.onload = resolve;
            script.onerror = reject;
            document.head.appendChild(script);
```

## 🔧 実装ガイド・技術詳細

### CDN統合実装パターン
```javascript
// js/managers/library-manager.js実装例
export class LibraryManager {
    static requiredLibraries = [
        { name: 'PIXI', global: 'PIXI', required: true },
        { name: 'PIXI_UI', global: 'PIXI_UI', required: true },
        { name: 'Viewport', global: 'Viewport', required: true },
        { name: 'PIXI.filters', global: 'PIXI.filters', required: false },
        { name: 'gsap', global: 'gsap', required: false },
        { name: 'lodash', global: '_', required: false },
        { name: 'Hammer', global: 'Hammer', required: false }
    ];
    
    static async validateCDNLibraries() {
        const results = [];
        
        for (const lib of this.requiredLibraries) {
            const available = !!window[lib.global];
            results.push({
                name: lib.name,
                available,
                required: lib.required,
                version: available ? this.getLibraryVersion(lib.global) : null
            });
            
            if (lib.required && !available) {
                throw new Error(`Required library ${lib.name} not loaded`);
            }
        }
        
        return results;
    }
}
```
