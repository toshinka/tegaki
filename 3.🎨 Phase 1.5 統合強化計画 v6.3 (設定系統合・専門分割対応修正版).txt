## 🎨 Stage3 ColorPalette.js 新規実装 (chroma-js活用)

### **ColorPalette.js: Stage3カラーパレット統合**
```javascript
// Stage3: chroma-js活用カラーパレット統合
import chroma from 'chroma-js';

export class ColorPalette {
    constructor(uiController) {
        this.uiController = uiController;
        this.currentColor = chroma('#800000'); // メインカラー
        this.colorHistory = [];
        this.palettePresets = this.generatePalettePresets();
        
        this.createColorPalette();
    }
    
    // chroma-js活用パレットプリセット生成
    generatePalettePresets() {
        const baseColor = chroma('#800000');
        return {
            analogous: chroma.scale([baseColor.darken(), baseColor, baseColor.brighten()]).colors(5),
            complementary: [baseColor, baseColor.set('hsl.h', '+180')],
            triadic: chroma.scale(baseColor).mode('hsl').range([0, 120, 240]).colors(3),
            monochromatic: chroma.scale([baseColor.darken(2), baseColor.brighten(2)]).colors(7)
        };
    }
    
    // カラーパレット動的生成
    createColorPalette() {
        const paletteContainer = this.createElement('div', {
            className: 'color-palette',
            style: this.getPaletteStyles()
        });
        
        // プリセットパレット生成
        Object.entries(this.palettePresets).forEach(([name, colors]) => {
            const presetGroup = this.createPresetGroup(name, colors);
            paletteContainer.appendChild(presetGroup);
        });
        
        // カスタムカラーピッカー
        const customPicker = this.createCustomColorPicker();
        paletteContainer.appendChild(customPicker);
        
        document.body.appendChild(paletteContainer);
    }
    
    // chroma-js活用色彩計算
    updateEngineColor(chromaColor) {
        const rgb = chromaColor.rgb();
        const normalized = rgb.map(c => c / 255);
        
        // UIController経由でエンジン色彩更新
        this.uiController.updateControlValue('color', {
            rgb: normalized,
            hex: chromaColor.hex(),
            hsl: chromaColor.hsl()
        });
        
        this.addToHistory(chromaColor);
    }
}
```

## 🔄 統合制御層の設定系連携強化

### UIController.js 設定系統合強化
```javascript
// 設定系統合による統合制御層連携
export class UIController {
    constructor(oglEngine) {
        this.engine = oglEngine;
        this.settingsManager = oglEngine.settingsManager; // 設定系直接参照
        
        // Stage3対応: ColorPalette統合
        this.colorPalette = null;
        if (this.supportsChromaJS()) {
            this.initColorPalette(); // chroma-js利用可能時のみ初期化
        }
        
        this.setupDynamicUI();
        this.setupSettingsIntegration(); // 設定系統合
    }
    
    // Stage3: カラーパレット初期化
    initColorPalette() {
        try {
            import('./ui/ColorPalette.js')
                .then(module => {
                    this.colorPalette = new module.ColorPalette(this);
                })
                .catch(() => console.log('Color palette not available'));
        } catch (error) {
            console.warn('chroma-js not available:', error);
        }
    }
    
    // 設定統合による値更新拡張
    updateControlValue(controlId, value) {
        const controlActions = {
            penSize: () => this.engine.setPenSize(parseInt(value)),
            opacity: () => this.engine.setOpacity(parseInt(value)),
            pressure: () => this.engine.setPressureSensitivity(parseInt(value)),
            smoothing: () => this.engine.setSmoothing(value),
            color: (colorData) => this.engine.setColor?.(colorData) // Stage3新規
        };
        
        const action = controlActions[controlId];
        if (action) {
            action();
            // 設定系への自動保存
            this.settingsManager?.set(`drawing.${controlId}`, value);
        }
    }
}
```
