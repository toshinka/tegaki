### Phase2レイヤーマネージャー統一設計
```javascript
// Phase2統一システム活用レイヤーマネージャー設計
class LayerManager {
    constructor() {
        // 統一システム依存性確認
        this.validateUnifiedSystems();
        
        // 統一設定取得
        this.config = ConfigManager.get('layers');
        this.layers = new Map();
        this.activeLayerId = null;
        this.layerCounter = 1;
        
        // EventBus統合
        this.setupEventListeners();
    }
    
    validateUnifiedSystems() {
        const required = ['ConfigManager', 'ErrorManager', 'StateManager', 'EventBus'];
        const missing = required.filter(sys => !window[sys]);
        if (missing.length > 0) {
            throw new Error(`レイヤーシステム: 統一システム依存不足: ${missing.join(', ')}`);
        }
    }
    
    createLayer(options = {}) {
        try {
            // レイヤー上限チェック（ConfigManager統一）
            if (this.layers.size >= this.config.maxLayers) {
                ErrorManager.showError('layer-limit', 
                    `レイヤー上限に達しています（最大: ${this.config.maxLayers}）`);
                return null;
            }
            
            // レイヤー作成
            const layerId = `layer-${this.layerCounter++}`;
            const layer = new Layer(layerId, {
                name: options.name || `${this.config.layerNaming.prefix}${this.layerCounter - 1}`,
                opacity: options.opacity || this.config.defaultOpacity,
                blendMode: options.blendMode || this.config.defaultBlendMode,
                visible: options.visible !== false
            });
            
            this.layers.set(layerId, layer);
            this.activeLayerId = layerId;
            
            // EventBus統一通知
            EventBus.safeEmit(EventBus.LayerEvents.LAYER_CREATED, {
                layerId,
                layer: layer.getState(),
                totalLayers: this.layers.size
            });
            
            return layerId;
            
        } catch (error) {
            ErrorManager.showError('layer-operation', 
                `レイヤー作成失敗: ${error.message}`, 
                { options });
            return null;
        }
    }
    
    getLayerState() {
        return {
            total: this.layers.size,
            active: this.activeLayerId,
            maxLayers: this.config.maxLayers,
            layers: Array.from(this.layers.entries()).map(([id, layer]) => ({
                id,
                ...layer.getState()
            }))
        };
    }
}
```
