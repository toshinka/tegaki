```javascript
// ✅ v1.2 Three.js パーティクル活用（実装効率化）
import * as THREE from 'three';

class ThreeAirsprayTool {
    constructor() {
        this.particleSystem = new THREE.Points(
            new THREE.BufferGeometry(),
            new THREE.PointsMaterial({
                size: 2,
                transparent: true,
                blending: THREE.AdditiveBlending
            })
        );
    }
    
    // Three.js内蔵機能で10行で実現
    sprayParticles(position, count) {
        const positions = new Float32Array(count * 3);
        for(let i = 0; i < count; i++) {
            positions[i * 3] = position.x + (Math.random() - 0.5) * 20;
            positions[i * 3 + 1] = position.y + (Math.random() - 0.5) * 20;
            positions[i * 3 + 2] = 0;
        }
        this.particleSystem.geometry.setAttribute('position', 
            new THREE.BufferAttribute(positions, 3));
    }
}
```

```javascript
class ChromaColorController {
    // chroma-js内蔵機能で1行で実現
    generatePalette(baseColor) {
        return chroma.scale([baseColor, 'white']).colors(10);
    }
    
    adjustBrightness(color, factor) {
        return chroma(color).brighten(factor).hex();
    }
    
    getColorHarmony(color) {
        const c = chroma(color);
        return {
            complementary: c.set('hsl.h', '+180').hex(),
            triadic: [
                c.set('hsl.h', '+120').hex(),
                c.set('hsl.h', '+240').hex()
            ]
        };
    }
```

```javascript
class GLMatrixPenTool {
    applyTransform(points, matrix) {
        // gl-matrix最適化済み関数で高速変換
        return points.map(point => {
            const vec = vec3.fromValues(point.x, point.y, 0);
```
