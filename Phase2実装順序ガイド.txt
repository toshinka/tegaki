// src/tools/BrushTool.ts作成
// 依存関係: Phase1 PenTool.ts IDrawingTool, DrawingEngine.ts
// 参考実装: Phase1 PenTool.ts の設定管理・activate/deactivate

export class BrushTool implements IDrawingTool {
  public readonly name = 'brush';
  public readonly icon = 'ti ti-brush'; // Phase3 Tabler Icons
  public readonly category = 'drawing' as const;

  private settings = {
    size: 12, // PenToolより大きな初期値
    opacity: 0.6, // 重ね塗り表現
    texture: 'round', // round/flat/texture Phase3拡張
    pressureOpacity: true, // 筆圧→透明度
    velocitySize: true, // 速度→サイズ変化
    color: 0x800000
  };

  // Phase1 PenTool参考・差別化実装
  public onPointerMove(event: IEventData['drawing:move']): void {
    if (!this.isActive) return;
    
    // 筆独自処理・速度計算・サイズ動的変更
    const dynamicSize = this.calculateDynamicSize(
      event.pressure, 
      event.velocity
    );
    
    // DrawingEngineに動的設定送信・リアルタイム変更
    this.eventBus.emit('brush:dynamic-update', {
      size: dynamicSize,
      opacity: this.calculateDynamicOpacity(event.pressure)
    });
  }
}
```

// src/core/WebGPUDetector.ts作成
// 依存関係: Phase1 PixiApplication.ts レンダラー管理参考
// Phase3準備: WebGPU初期化・GPU性能評価・自動選択

export interface GPUCapabilities {
  supported: boolean;
  tier: 'webgpu' | 'webgl2' | 'webgl';
  features: string[];
  memoryMB: number;
  performance: 'high' | 'medium' | 'low';
}

export class WebGPUDetector {
  // Phase3準備・段階的実装・Phase1安定性継承
  public static async detectCapabilities(): Promise<GPUCapabilities> {
    // WebGPU対応検出・navigator.gpu
    if (!navigator.gpu) {
      return this.fallbackToWebGL();
    }

    // GPU Adapter要求・性能評価
    const adapter = await navigator.gpu.requestAdapter({
      powerPreference: 'high-performance'
    });

    if (!adapter) {
      return this.fallbackToWebGL();
    }

    // GPU情報取得・メモリ・機能・Phase3判定
    return this.evaluateGPUPerformance(adapter);
  }
}
```
