class PixiPopupPanel {
    constructor(pixiApp, x, y, width, height) {
        this.app = pixiApp;
        this.container = new PIXI.Container();
        this.container.x = x;
        this.container.y = y;
        
        this.createBackground(width, height);
        this.createContent();
    }
    
    createBackground(width, height) {
        const bg = new PIXI.Graphics();
        
        // ふたば色グラデーション
        const gradient = new PIXI.Texture.from(this.createGradientCanvas());
        const sprite = new PIXI.Sprite(gradient);
        sprite.width = width;
        sprite.height = height;
        
        // 角丸実装
        const mask = new PIXI.Graphics();
        mask.beginFill(0xffffff);
        mask.drawRoundedRect(0, 0, width, height, 16);
        mask.endFill();
        
        sprite.mask = mask;
        
        this.container.addChild(sprite);
        this.container.addChild(mask);
    }
    
    createGradientCanvas() {
        const canvas = document.createElement('canvas');
        canvas.width = 256;
        canvas.height = 256;
        const ctx = canvas.getContext('2d');
        
        const gradient = ctx.createLinearGradient(0, 0, 256, 256);
        gradient.addColorStop(0, 'rgba(128,0,0,0.96)');   // --futaba-maroon
        gradient.addColorStop(1, 'rgba(170,90,86,0.92)'); // --futaba-light-maroon
        
        ctx.fillStyle = gradient;
        ctx.fillRect(0, 0, 256, 256);
        
        return canvas;
    }
}

### HSV円形ピッカー（PixiJS統一座標）
```javascript
class PixiHSVWheel {
    constructor(radius = 60) {
        this.radius = radius;
        this.container = new PIXI.Container();
        this.createColorWheel();
    }
    
    createColorWheel() {
        const wheel = new PIXI.Graphics();
        
        // 色相環360度
        for (let angle = 0; angle < 360; angle++) {
            const color = this.hsvToPixiColor(angle, 100, 100);
            
            wheel.beginFill(color);
            wheel.moveTo(this.radius, this.radius);
            wheel.arc(
                this.radius, this.radius, this.radius,
                (angle - 1) * Math.PI / 180,
                angle * Math.PI / 180
            );
            wheel.endFill();
        }
        
        // 明度グラデーション（中央→外側）
        const saturationMask = new PIXI.Graphics();
        for (let r = 0; r < this.radius; r++) {
            const alpha = r / this.radius;
            saturationMask.beginFill(0x000000, 1 - alpha);
            saturationMask.drawCircle(this.radius, this.radius, r);
            saturationMask.endFill();
        }
        
        this.container.addChild(wheel);
        this.container.addChild(saturationMask);
        
        // ピッカー操作
        this.container.interactive = true;
        this.container.on('pointerdown', this.handleColorPick.bind(this));
    }
    
    hsvToPixiColor(h, s, v) {
        // HSV → RGB → PixiJS Color変換
        const c = (v / 100) * (s / 100);
        const x = c * (1 - Math.abs((h / 60) % 2 - 1));
        const m = (v / 100) - c;
        
        let r, g, b;
        if (h < 60) { r = c; g = x; b = 0; }
        else if (h < 120) { r = x; g = c; b = 0; }
        else if (h < 180) { r = 0; g = c; b = x; }
        else if (h < 240) { r = 0; g = x; b = c; }
        else if (h < 300) { r = x; g = 0; b = c; }
        else { r = c; g = 0; b = x; }
        
        r = Math.round((r + m) * 255);
        g = Math.round((g + m) * 255);
        b = Math.round((b + m) * 255);
        
        return (r << 16) | (g << 8) | b;
    }
}
```

class PixiLayerThumbnail {
    constructor(pixiApp, layerContainer) {
        this.app = pixiApp;
        this.layerContainer = layerContainer;
        this.thumbnailSize = 64;
    }
    
    async generateThumbnail() {
        // PixiJS RenderTexture でサムネイル生成
        const renderTexture = PIXI.RenderTexture.create({
            width: this.thumbnailSize,
            height: this.thumbnailSize,
            resolution: 1
        });
        
        // レイヤー内容をレンダリング
        const originalTransform = {
            x: this.layerContainer.x,
            y: this.layerContainer.y,
            scaleX: this.layerContainer.scale.x,
            scaleY: this.layerContainer.scale.y
        };
        
        // サムネイル用スケール調整
        const scale = this.thumbnailSize / Math.max(
            this.app.view.width, 
            this.app.view.height
        );
        
        this.layerContainer.x = 0;
        this.layerContainer.y = 0;
        this.layerContainer.scale.set(scale);
        
        this.app.renderer.render(this.layerContainer, { renderTexture });
        
        // 元の変形復元
        this.layerContainer.x = originalTransform.x;
        this.layerContainer.y = originalTransform.y;
        this.layerContainer.scale.set(originalTransform.scaleX, originalTransform.scaleY);
        
        // Sprite作成
        const thumbnailSprite = new PIXI.Sprite(renderTexture);
        thumbnailSprite.width = this.thumbnailSize;
        thumbnailSprite.height = this.thumbnailSize;
        
        return thumbnailSprite;
    }
}

class PixiLayerDragDrop {
    constructor(pixiApp, layerPanel) {
        this.app = pixiApp;
        this.layerPanel = layerPanel;
        this.dragData = null;
        this.dropTargets = [];
    }
    
    enableDragDrop(layerItem) {
        layerItem.interactive = true;
        layerItem.buttonMode = true;
        
        layerItem.on('pointerdown', (event) => {
            this.startDrag(event, layerItem);
        });
        
        layerItem.on('pointermove', (event) => {
            if (this.dragData) {
                this.updateDrag(event);
            }
        });
        
        layerItem.on('pointerup', () => {
            this.endDrag();
        });
    }
    
    startDrag(event, layerItem) {
        const localPos = event.data.getLocalPosition(this.layerPanel);
        
        this.dragData = {
            item: layerItem,
            offset: {
                x: localPos.x - layerItem.x,
                y: localPos.y - layerItem.y
            },
            originalPosition: { x: layerItem.x, y: layerItem.y }
        };
        
        // ドラッグプレビュー作成
        layerItem.alpha = 0.8;
        layerItem.scale.set(0.95);
        layerItem.rotation = 0.05; // 3度回転
    }
    
    updateDrag(event) {
        const localPos = event.data.getLocalPosition(this.layerPanel);
        const item = this.dragData.item;
        
        item.x = localPos.x - this.dragData.offset.x;
        item.y = localPos.y - this.dragData.offset.y;
        
        // ドロップターゲット検出
        this.updateDropTargets(item.y);
    }
    
    endDrag() {
        if (this.dragData) {
            const item = this.dragData.item;
            
            // ドラッグ状態リセット
            item.alpha = 1.0;
            item.scale.set(1.0);
            item.rotation = 0;
            
            this.dragData = null;
        }
    }
}

class PixiTimelineController {
    constructor(pixiApp) {
        this.app = pixiApp;
        this.timelineContainer = new PIXI.Container();
        this.timelineHeight = 0;
        this.isActive = false;
        
        // 初期状態（非表示）
        this.timelineContainer.y = this.app.view.height;
        this.app.stage.addChild(this.timelineContainer);
    }
    
    showTimeline() {
        this.isActive = true;
        this.timelineHeight = Math.min(350, this.app.view.height * 0.3);
        
        // PixiJS統一座標アニメーション
        const targetY = this.app.view.height - this.timelineHeight;
        
        this.animateTimeline(this.timelineContainer.y, targetY, 400);
    }
    
    hideTimeline() {
        this.isActive = false;
        const targetY = this.app.view.height;
        
        this.animateTimeline(this.timelineContainer.y, targetY, 300);
    }
    
    animateTimeline(fromY, toY, duration) {
        const startTime = Date.now();
        
        const animate = () => {
            const elapsed = Date.now() - startTime;
            const progress = Math.min(elapsed / duration, 1);
            
            // Cubic bezier easing
            const eased = this.cubicBezier(progress, 0.25, 0.46, 0.45, 0.94);
            
            this.timelineContainer.y = fromY + (toY - fromY) * eased;
            
            if (progress < 1) {
                requestAnimationFrame(animate);
            }
        };
        
        animate();
    }
    
    cubicBezier(t, x1, y1, x2, y2) {
        // Cubic bezier curve implementation
        const cx = 3 * x1;
        const bx = 3 * (x2 - x1) - cx;
        const ax = 1 - cx - bx;
        
        const cy = 3 * y1;
        const by = 3 * (y2 - y1) - cy;
        const ay = 1 - cy - by;
        
        return ((ax * t + bx) * t + cx) * t;
    }
}

class PixiFrameThumbnails {
    constructor(pixiApp) {
        this.app = pixiApp;
        this.frames = [];
        this.thumbnailContainer = new PIXI.Container();
        this.thumbnailSize = { width: 64, height: 48 };
    }
    
    createFrameThumbnail(frameData, index) {
        // PixiJS RenderTexture でフレームサムネイル生成
        const renderTexture = PIXI.RenderTexture.create({
            width: this.thumbnailSize.width,
            height: this.thumbnailSize.height,
            resolution: 1
        });
        
        // フレームデータを一時Container に構築
        const frameContainer = new PIXI.Container();
        
        frameData.layers.forEach(layerData => {
            const layerGraphics = new PIXI.Graphics();
            
            layerData.vectorStrokes.forEach(stroke => {
                this.renderStrokeToGraphics(layerGraphics, stroke);
            });
            
            frameContainer.addChild(layerGraphics);
        });
        
        // スケール調整してレンダリング
        const scale = Math.min(
            this.thumbnailSize.width / this.app.view.width,
            this.thumbnailSize.height / this.app.view.height
        );
        
        frameContainer.scale.set(scale);
        
        this.app.renderer.render(frameContainer, { renderTexture });
        
        // サムネイルSprite作成
        const thumbnailSprite = new PIXI.Sprite(renderTexture);
        thumbnailSprite.x = index * (this.thumbnailSize.width + 8);
        thumbnailSprite.y = 8;
        
        // フレーム番号テキスト
        const frameNumberText = new PIXI.Text((index + 1).toString(), {
            fontFamily: 'Arial',
            fontSize: 12,
            fill: 0xf0e0d6,
            align: 'center'
        });
        
        frameNumberText.x = thumbnailSprite.x + this.thumbnailSize.width / 2 - frameNumberText.width / 2;
        frameNumberText.y = thumbnailSprite.y + this.thumbnailSize.height + 4;
        
        this.thumbnailContainer.addChild(thumbnailSprite);
        this.thumbnailContainer.addChild(frameNumberText);
        
        // インタラクション設定
        thumbnailSprite.interactive = true;
        thumbnailSprite.buttonMode = true;
        
        thumbnailSprite.on('pointerover', () => {
            thumbnailSprite.scale.set(1.05);
        });
        
        thumbnailSprite.on('pointerout', () => {
            thumbnailSprite.scale.set(1.0);
        });
        
        return thumbnailSprite;
    }
    
    renderStrokeToGraphics(graphics, stroke) {
        graphics.lineStyle({
            width: stroke.size * 0.5, // サムネイル用縮小
            color: stroke.color,
            alpha: stroke.opacity,
            cap: PIXI.LINE_CAP.ROUND,
            join: PIXI.LINE_JOIN.ROUND
        });
        
        const points = stroke.points;
        if (points.length > 0) {
            graphics.moveTo(points[0].x, points[0].y);
            
            for (let i = 1; i < points.length - 1; i++) {
                const xc = (points[i].x + points[i + 1].x) / 2;
                const yc = (points[i].y + points[i + 1].y) / 2;
                graphics.quadraticCurveTo(points[i].x, points[i].y, xc, yc);
            }
            
            if (points.length > 1) {
                const last = points[points.length - 1];
                graphics.lineTo(last.x, last.y);
            }
        }
    }
}

class PixiShortcutController {
    constructor(pixiApp) {
        this.app = pixiApp;
        this.coordinate = new PixiCoordinateUnifier(pixiApp);
        this.setupShortcuts();
    }
    
    setupShortcuts() {
        document.addEventListener('keydown', (e) => {
            this.handleShortcut(e);
        });
    }
    
    handleShortcut(event) {
        const key = event.key.toLowerCase();
        const ctrl = event.ctrlKey;
        const shift = event.shiftKey;
        
        // ツール切り替え（PixiJS統一座標対応）
        if (!ctrl && !shift) {
            switch (key) {
                case 'b': this.switchTool('pen'); break;
                case 'e': this.switchTool('eraser'); break;
                case 'i': this.switchTool('eyedropper'); break;
                case 'g': this.switchTool('bucket'); break;
                case 't': this.switchTool('text'); break;
                case 'm': this.switchTool('select'); break;
            }
        }
        
        // 編集操作（PixiJS統一座標対応）
        if (ctrl) {
            switch (key) {
                case 'z': 
                    if (shift) {
                        this.redo();
                    } else {
                        this.undo();
                    }
                    break;
                case 'y': this.redo(); break;
                case '0': this.resetView(); break;
            }
        }
        
        // キャンバス操作
        if (key === 'h') {
            if (shift) {
                this.flipVertical();
            } else {
                this.flipHorizontal();
            }
        }
        
        if (key === 'delete') {
            this.deleteSelected();
        }
    }
    
    switchTool(toolName) {
        // EventStore経由でツール切り替え通知
        EventStore.emit('tool:switch', { tool: toolName });
    }
    
    resetView() {
        // PixiJS統一座標でのビューリセット
        EventStore.emit('canvas:resetView');
    }
}
