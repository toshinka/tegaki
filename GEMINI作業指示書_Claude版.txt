【修正2】fill-tool.js — _clearLayerGraphics が layerSprite を消す問題
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

■ 問題
_clearLayerGraphics() が layer の全子要素（layerSprite を含む）を
削除・破棄してしまう。layerSprite は renderTexture を表示する唯一の
Sprite なので、これが消えると描画内容が見えなくなる。

また fill の Graphics が renderTexture に焼き込まれないため、
他のブラシ描画と共存できない。

■ 修正方針
fill を「Graphics を addChild」方式から
「renderTexture に renderer.render() で焼き込む」方式に変更する。
_clearLayerGraphics は呼ばない。

■ _fillLayerLegacy() を以下に完全置き換える

_fillLayerLegacy(layer, color, alpha, layerManager) {
    const CONFIG = window.TEGAKI_CONFIG;
    const layerData = layer.layerData;
    const app = layerManager.app;

    if (!layerData?.renderTexture || !app?.renderer) {
        console.error('❌ FillTool: No renderTexture or renderer');
        return;
    }

    // 塗りつぶし用の Graphics を生成して renderTexture に焼き込む
    const fillGraphics = new Graphics();
    fillGraphics.rect(0, 0, CONFIG.canvas.width, CONFIG.canvas.height);
    fillGraphics.fill({ color, alpha });

    app.renderer.render({
        container: fillGraphics,
        target: layerData.renderTexture,
        clear: false  // 既存内容の上に重ね塗り
    });

    fillGraphics.destroy({ children: true });

    // 履歴用のデータ保存（pathsData は互換性のために残す）
    const pathData = {
        id: `fill_legacy_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
        type: 'fill',
        tool: 'fill',
        color, alpha,
        timestamp: Date.now(),
        settings: { color, opacity: alpha, mode: 'fill-legacy' }
    };

    if (!layerData.pathsData) layerData.pathsData = [];
    layerData.pathsData.push(pathData);

    // サムネイル更新要求
    const layerIndex = layerManager.getLayerIndex(layer);
    const layerId = layerData.id;
    if (window.eventBus || window.TegakiEventBus) {
        const eb = window.eventBus || window.TegakiEventBus;
        eb.emit('thumbnail:layer-updated', {
            layerIndex,
            layerId,
            immediate: true
        });
    }

    if (this.eventBus) {
        this.eventBus.emit('layer:filled', {
            layerId: layerData.id,
            color, alpha,
            method: 'legacy-full'
        });
    }
}

【修正3】pointer-handler.js — getCoalescedEvents でペン品質改善
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

■ 効果
getCoalescedEvents() を使うとブラウザが間引いた中間ポイントを全て取得できる。
「後伸び」「入りのディレイ感」が大幅に改善する可能性がある。

■ pointer-handler.js の pointermove ハンドラを探す
以下のような構造があるはず:
  canvas.addEventListener('pointermove', (e) => { ... })
  または
  onPointerMove(e) { ... }

■ その中で callback を呼んでいる部分を以下に変更する

【変更前（推定）】
canvas.addEventListener('pointermove', (e) => {
    // ... info 生成 ...
    callbacks.move(info, e);
});

【変更後】
canvas.addEventListener('pointermove', (e) => {
    // getCoalescedEvents() で間引かれた中間点を全て処理
    const events = e.getCoalescedEvents ? e.getCoalescedEvents() : [e];
    for (const coalescedEvent of events) {
        // coalescedEvent から info を生成して move コールバックを呼ぶ
        // 既存の info 生成ロジックを coalescedEvent に適用すること
        const info = {
            clientX: coalescedEvent.clientX,
            clientY: coalescedEvent.clientY,
            pressure: coalescedEvent.pressure,
            pointerType: coalescedEvent.pointerType,
            pointerId: coalescedEvent.pointerId,
            button: coalescedEvent.button,
            buttons: coalescedEvent.buttons,
            isPrimary: coalescedEvent.isPrimary
        };
        callbacks.move(info, coalescedEvent);
    }
});

■ 注意
既存の pointermove の info 生成ロジックをそのまま coalescedEvent に適用する。
フィールド名が違う場合は既存コードに合わせること。
pointermove 以外（down/up）は変更しない。
