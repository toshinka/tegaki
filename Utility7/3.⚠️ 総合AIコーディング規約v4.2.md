// ✅ PixiJS v8.11.0 + OffscreenCanvas Worker統合（v4.2新設）
class PixiV8OffscreenProcessor {
    constructor(pixiApp, workerCount = 4) {
        this.app = pixiApp;
        this.workerCount = workerCount;
        this.workers = [];
        this.taskQueue = [];
        this.availableWorkers = [];
    }
    
    async initialize() {
        // OffscreenCanvas Worker プール初期化
        for (let i = 0; i < this.workerCount; i++) {
            const worker = new Worker('./workers/pixiV8OffscreenWorker.js', {
                type: 'module'
            });
            
            worker.onmessage = (event) => this.handleWorkerMessage(worker, event);
            
            // Worker初期化
            const offscreenCanvas = new OffscreenCanvas(
                this.app.canvas.width,
                this.app.canvas.height
            );
            
            const transferableCanvas = offscreenCanvas.transferControlToOffscreen();
            worker.postMessage({
                type: 'initCanvas',
                canvas: transferableCanvas,
                config: {
                    width: this.app.canvas.width,
                    height: this.app.canvas.height,
                    preference: 'webgpu'
                }
            }, [transferableCanvas]);
            
            this.workers.push(worker);
        }
        
        console.log(`OffscreenCanvas Worker pool initialized: ${this.workerCount} workers`);
    }
    
    async processTransform(vectorData, transformMatrix) {
        // 並列変形処理・メインスレッド最適化
        return new Promise((resolve) => {
            const availableWorker = this.getAvailableWorker();
            if (!availableWorker) {
                this.taskQueue.push({ vectorData, transformMatrix, resolve });
                return;
            }
            
            this.assignTask(availableWorker, 'transform', {
                vectorData,
                transformMatrix
            }, resolve);
        });
    }
    
    async renderLayerOffscreen(layerId, vectorData) {
        // 並列レイヤー描画・120FPS対応
        return new Promise((resolve) => {
            const availableWorker = this.getAvailableWorker();
            if (!availableWorker) {
                this.taskQueue.push({ layerId, vectorData, resolve, type: 'render' });
                return;
            }
            
            this.assignTask(availableWorker, 'renderLayer', {
                layerId,
                vectorData,
                timestamp: performance.now()
            }, resolve);
        });
    }
    
    getAvailableWorker() {
        return this.availableWorkers.pop() || null;
    }
    
    assignTask(worker, taskType, data, resolve) {
        worker.postMessage({
            type: taskType,
            data,
            taskId: this.generateTaskId()
        });
        
        worker.resolve = resolve;
    }
    
    handleWorkerMessage(worker, event) {
        const { type, data } = event.data;
        
        switch (type) {
            case 'canvasReady':
                this.availableWorkers.push(worker);
                break;
            case 'taskComplete':
                if (worker.resolve) {
                    worker.resolve(data);
                    worker.resolve = null;
                }
                this.availableWorkers.push(worker);
                this.processNextTask();
                break;
        }
    }
    
    processNextTask() {
        if (this.taskQueue.length > 0 && this.availableWorkers.length > 0) {
            const task = this.taskQueue.shift();
            const worker = this.availableWorkers.pop();
            
            if (task.type === 'render') {
                this.assignTask(worker, 'renderLayer', {
                    layerId: task.layerId,
                    vectorData: task.vectorData
                }, task.resolve);
            } else {
                this.assignTask(worker, 'transform', {
                    vectorData: task.vectorData,
                    transformMatrix: task.transformMatrix
                }, task.resolve);
            }
        }
    }
    
    generateTaskId() {
        return `task_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    }
}

    }
    
    async startRecording() {
        if (!this.videoEncoder || this.isRecording) return false;
        
        this.isRecording = true;
        this.chunks = [];
        this.frameCount = 0;
        this.startTime = performance.now();
        
        // 120FPS キャプチャループ開始
        this.captureLoop();
        
        console.log(`Recording started: ${this.targetFPS}FPS`);
        return true;
    }
    
    async stopRecording() {
        if (!this.isRecording) return null;
        
        this.isRecording = false;
        
        // エンコーダーフラッシュ・完了待機
        await this.videoEncoder.flush();
        
        // MP4ファイル生成
        const blob = new Blob(this.chunks, { type: 'video/mp4' });
        
        console.log(`Recording completed: ${this.frameCount} frames, ${blob.size} bytes`);
        return blob;
    }
    
    captureLoop() {
        if (!this.isRecording) return;
        
        const currentTime = performance.now();
        const elapsedTime = currentTime - this.startTime;
        const expectedFrameCount = Math.floor(elapsedTime * this.targetFPS / 1000);
        
        if (this.frameCount < expectedFrameCount) {
            this.captureFrame();
            this.frameCount++;
        }
        
        // 次フレーム処理をスケジュール
        requestAnimationFrame(() => this.captureLoop());
    }
    
    captureFrame() {
        // PixiJS v8.11.0 RenderTexture キャプチャ
        const renderTexture = this.app.renderer.generateTexture(this.app.stage);
        const canvas = this.app.renderer.extract.canvas(renderTexture);
        
        // VideoFrame作成・WebCodecs送信
        const timestamp = this.frameCount * (1000000 / this.targetFPS); // microseconds
        const videoFrame = new VideoFrame(canvas, {
            timestamp,
            duration: 1000000 / this.targetFPS
        });
        
        // エンコード処理
        if (this.videoEncoder.state === 'configured') {
            const keyFrame = this.frameCount % (this.targetFPS * 2) === 0; // 2秒間隔
            this.videoEncoder.encode(videoFrame, { keyFrame });
        }
        
        videoFrame.close();
    }
    
    handleEncodedChunk(chunk, metadata) {
        // エンコード済みチャンク保存
        const chunkData = new Uint8Array(chunk.byteLength);
        chunk.copyTo(chunkData);
        this.chunks.push(chunkData);
        
        // プログレス通知
        if (this.onProgress) {
            const duration = this.frameCount / this.targetFPS;
            this.onProgress(this.frameCount, duration, chunk.byteLength);
        }
    }
    
    // リアルタイムストリーミング出力
    async setupLiveStreaming(streamConfig) {
        if (!this.videoEncoder) return false;
        
        // WebRTC・配信用設定
        this.streamConfig = {
            ...streamConfig,
            realtime: true,
            lowLatency: true,
            targetBitrate: streamConfig.bitrate || 4000000
        };
        
        // ストリーミング用エンコーダー再設定
        const config = {
            codec: 'avc1.42001f',
            width: this.app.canvas.width,
            height: this.app.canvas.height,
            bitrate: this.streamConfig.targetBitrate,
            framerate: this.targetFPS,
            latencyMode: 'realtime',
            avc: { format: 'annexb' }
        };
        
        await this.videoEncoder.configure(config);
        console.log('Live streaming configured:', config);
        return true;
    }
}

class PixiV8PerformanceMonitor {
    constructor(targetFPS = 120) {
        this.targetFPS = targetFPS;
        this.targetFrameTime = 1000 / targetFPS; // ms
        this.isMonitoring = false;
        this.metrics = {
            fps: 0,
            frameTime: 0,
            memoryUsage: 0,
            gpuMemory: 0,
            inputLatency: 0,
            renderTime: 0
        };
        this.observers = [];
    }
    
    start() {
        if (this.isMonitoring) return;
        
        this.isMonitoring = true;
        this.setupPerformanceObservers();
        this.startFPSMonitoring();
        this.startMemoryMonitoring();
        
        console.log(`Performance monitoring started: ${this.targetFPS}FPS target`);
    }
    
    stop() {
        this.isMonitoring = false;
        this.observers.forEach(observer => observer.disconnect());
        this.observers = [];
        
        console.log('Performance monitoring stopped');
    }
    
    setupPerformanceObservers() {
        if (typeof PerformanceObserver === 'undefined') {
            console.warn('PerformanceObserver not supported');
            return;
        }
        
        // フレーム描画監視
        const frameObserver = new PerformanceObserver((list) => {
            const entries = list.getEntries();
            entries.forEach(entry => {
                if (entry.entryType === 'measure' && entry.name.includes('frame')) {
                    this.metrics.renderTime = entry.duration;
                    this.checkPerformanceThresholds();
                }
            });
        });
        
        frameObserver.observe({ entryTypes: ['measure'] });
        this.observers.push(frameObserver);
        
        // 入力遅延監視
        const eventObserver = new PerformanceObserver((list) => {
            const entries = list.getEntries();
            entries.forEach(entry => {
                if (entry.entryType === 'event') {
                    this.metrics.inputLatency = entry.processingEnd - entry.startTime;
                    
                    // 1ms以下目標チェック
                    if (this.metrics.inputLatency > 1) {
                        this.optimizeInputProcessing();
                    }
                }
            });
        });
        
        try {
            eventObserver.observe({ entryTypes: ['event'] });
            this.observers.push(eventObserver);
        } catch (e) {
            console.warn('Event timing observer not supported');
        }
        
        // ナビゲーション・リソース監視
        const navigationObserver = new PerformanceObserver((list) => {
            const entries = list.getEntries();
            entries.forEach(entry => {
                if (entry.entryType === 'navigation') {
                    console.log('Page load performance:', entry.loadEventEnd - entry.fetchStart);
                }
            });
        });
        
        navigationObserver.observe({ entryTypes: ['navigation'] });
        this.observers.push(navigationObserver);
    }
    
    startFPSMonitoring() {
        let frameCount = 0;
        let lastTime = performance.now();
        let lastFrameTime = lastTime;
        
        const measureFPS = (currentTime) => {
            if (!this.isMonitoring) return;
            
            frameCount++;
            const deltaTime = currentTime - lastFrameTime;
            lastFrameTime = currentTime;
            
            // フレーム時間計算
            this.metrics.frameTime = deltaTime;
            
            // FPS計算（1秒間隔）
            if (currentTime - lastTime >= 1000) {
                this.metrics.fps = Math.round(frameCount * 1000 / (currentTime - lastTime));
                frameCount = 0;
                lastTime = currentTime;
                
                // 性能分析・最適化提案
                this.analyzePerformance();
            }
            
            // フレーム測定マーク
            performance.mark('frame-start');
            
            requestAnimationFrame(measureFPS);
        };
        
        requestAnimationFrame(measureFPS);
    }
    
    startMemoryMonitoring() {
        if (!performance.memory) {
            console.warn('Memory monitoring not supported');
            return;
        }
        
        const monitorMemory = () => {
            if (!this.isMonitoring) return;
            
            // メモリ使用量監視
            this.metrics.memoryUsage = performance.memory.usedJSHeapSize / 1024 / 1024; // MB
            
            // GPU メモリ推定（WebGPU情報から）
            if (navigator.gpu) {
                this.estimateGPUMemoryUsage();
            }
            
            // メモリ警告チェック
            if (this.metrics.memoryUsage > 1024) { // 1GB以上
                this.optimizeMemoryUsage();
            }
            
            setTimeout(monitorMemory, 1000); // 1秒間隔
        };
        
        monitorMemory();
    }
    
    analyzePerformance() {
        const issues = [];
        const optimizations = [];
        
        // FPS性能分析
        if (this.metrics.fps < this.targetFPS * 0.9) {
            issues.push(`FPS低下: ${this.metrics.fps}/${this.targetFPS}FPS`);
            
            if (this.metrics.frameTime > this.targetFrameTime * 1.5) {
                optimizations.push('フレーム処理最適化が必要');
            }
            
            if (this.metrics.memoryUsage > 512) {
                optimizations.push('メモリ最適化が必要');
            }
        }
        
        // 入力遅延分析
        if (this.metrics.inputLatency > 1) {
            issues.push(`入力遅延: ${this.metrics.inputLatency.toFixed(2)}ms`);
            optimizations.push('入力処理最適化が必要');
        }
        
        // 描画時間分析
        if (this.metrics.renderTime > this.targetFrameTime * 0.8) {
            issues.push(`描画時間過多: ${this.metrics.renderTime.toFixed(2)}ms`);
            optimizations.push('GPU最適化が必要');
        }
        
        // 性能レポート
        if (issues.length > 0) {
            console.warn('Performance Issues:', issues);
            console.log('Suggested Optimizations:', optimizations);
            
            // 自動最適化実行
            this.autoOptimize(optimizations);
        }
        
        // 性能コールバック
        if (this.onPerformanceUpdate) {
            this.onPerformanceUpdate(this.metrics, issues, optimizations);
        }
    }
    
    checkPerformanceThresholds() {
        // Procreateクラス性能閾値チェック
        const thresholds = {
            fps: this.targetFPS * 0.95,      // 95%以上維持
            frameTime: this.targetFrameTime,  // 目標フレーム時間以下
            inputLatency: 1,                  // 1ms以下
            memoryUsage: 1024                 // 1GB以下
        };
        
        Object.entries(thresholds).forEach(([metric, threshold]) => {
            if (this.metrics[metric] > threshold) {
                this.handlePerformanceThresholdExceeded(metric, this.metrics[metric], threshold);
            }
        });
    }
    
    handlePerformanceThresholdExceeded(metric, current, threshold) {
        console.warn(`Performance threshold exceeded: ${metric} = ${current} > ${threshold}`);
        
        // 緊急最適化
        switch (metric) {
            case 'fps':
                this.reduceFPSTarget();
                break;
            case 'inputLatency':
                this.optimizeInputProcessing();
                break;
            case 'memoryUsage':
                this.forceGarbageCollection();
                break;
            case 'frameTime':
                this.optimizeRenderPipeline();
                break;
        }
    }
    
    autoOptimize(optimizations) {
        optimizations.forEach(optimization => {
            switch (optimization) {
                case 'フレーム処理最適化が必要':
                    this.optimizeFrameProcessing();
                    break;
                case 'メモリ最適化が必要':
                    this.optimizeMemoryUsage();
                    break;
                case '入力処理最適化が必要':
                    this.optimizeInputProcessing();
                    break;
                case 'GPU最適化が必要':
                    this.optimizeGPUUsage();
                    break;
            }
        });
    }
    
    optimizeFrameProcessing() {
        // フレーム処理最適化
        console.log('Optimizing frame processing...');
        
        // 描画品質調整
        if (this.targetFPS > 60) {
            this.targetFPS = 60; // 60FPSに縮退
            console.log('Reduced target FPS to 60 for stability');
        }
    }
    
    optimizeInputProcessing() {
        // 入力処理最適化
        console.log('Optimizing input processing...');
        
        // 入力頻度調整
        if (this.inputFrequency > 60) {
            this.inputFrequency = 60; // 60Hzに制限
        }
    }
    
    optimizeMemoryUsage() {
        // メモリ使用最適化
        console.log('Optimizing memory usage...');
        
        // 手動ガベージコレクション
        if (window.gc) {
            window.gc();
        }
        
        // キャッシュクリア
        this.clearUnusedCaches();
    }
    
    optimizeGPUUsage() {
        // GPU使用最適化
        console.log('Optimizing GPU usage...');
        
        // シェーダー最適化
        this.optimizeShaders();
        
        // テクスチャメモリ管理
        this.optimizeTextureMemory();
    }
    
    estimateGPUMemoryUsage() {
        // GPU メモリ使用量推定
        try {
            if (this.app && this.app.renderer && this.app.renderer.texture) {
                const textureManager = this.app.renderer.texture;
                const textureCount = textureManager.managedTextures.length;
                
                // テクスチャメモリ推定
                let estimatedGPUMemory = 0;
                textureManager.managedTextures.forEach(texture => {
                    if (texture.valid) {
                        const bytes = texture.width * texture.height * 4; // RGBA
                        estimatedGPUMemory += bytes;
                    }
                });
                
                this.metrics.gpuMemory = estimatedGPUMemory / 1024 / 1024; // MB
            }
        } catch (error) {
            console.warn('GPU memory estimation failed:', error);
        }
    }
    
    clearUnusedCaches() {
        // 未使用キャッシュクリア
        if (this.app && this.app.renderer) {
            this.app.renderer.texture.destroyUnusedTextures();
            this.app.renderer.geometry.destroyUnusedGeometry();
        }
    }
    
    getPerformanceReport() {
        return {
            timestamp: Date.now(),
            metrics: { ...this.metrics },
            status: this.getPerformanceStatus(),
            recommendations: this.getRecommendations()
        };
    }
    
    getPerformanceStatus() {
        if (this.metrics.fps >= this.targetFPS * 0.95 && 
            this.metrics.inputLatency <= 1 && 
            this.metrics.memoryUsage <= 1024) {
            return 'excellent';
        } else if (this.metrics.fps >= this.targetFPS * 0.8) {
            return 'good';
        } else if (this.metrics.fps >= this.targetFPS * 0.6) {
            return 'fair';
        } else {
            return 'poor';
        }
    }
    
    getRecommendations() {
        const recommendations = [];
        
        if (this.metrics.fps < this.targetFPS * 0.9) {
            recommendations.push('FPS向上のため描画最適化を推奨');
        }
        
        if (this.metrics.inputLatency > 1) {
            recommendations.push('入力遅延改善のため処理最適化を推奨');
        }
        
        if (this.metrics.memoryUsage > 512) {
            recommendations.push('メモリ使用量削減を推奨');
        }
        
        return recommendations;
    }
}
