javascript// Tier検出・自動最適化
const detectPerformanceTier = () => {
  const canvas = app.renderer.gl?.canvas;
  const webgpuSupported = app.renderer.type === 'webgpu';
  const memory = navigator.deviceMemory || 4;
  
  if (webgpuSupported && memory >= 8) {
    return 'tier1'; // WebGPU・120FPS目標
  } else if (app.renderer.type === 'webgl2') {
    return 'tier2'; // WebGL2・60FPS目標  
  } else {
    return 'tier3'; // 基本機能のみ
  }
};
