class PixiV8WebGPUComputeProcessor {
    constructor(webgpuDevice) {
        this.device = webgpuDevice;
        this.computePipelines = new Map();
        this.setupShaderModules();
    }
    
    async setupShaderModules() {
        // Airbrush compute shader
        const airbrushShader = await fetch('/src/shaders/airbrush.wgsl')
            .then(response => response.text());
        
        this.shaderModules = {
            airbrush: this.device.createShaderModule({
                code: airbrushShader
            }),
            blur: this.device.createShaderModule({
                code: await fetch('/src/shaders/blur.wgsl').then(r => r.text())
            }),
            performance: this.device.createShaderModule({
                code: await fetch('/src/shaders/performance.wgsl').then(r => r.text())
            })
        };
        
        this.createComputePipelines();
    }
    
    createComputePipelines() {
        // High-performance airbrush pipeline
        this.computePipelines.set('airbrush', this.device.createComputePipeline({
            layout: 'auto',
            compute: {
                module: this.shaderModules.airbrush,
                entryPoint: 'main'
            }
        }));
        
        // Real-time blur pipeline
        this.computePipelines.set('blur', this.device.createComputePipeline({
            layout: 'auto',
            compute: {
                module: this.shaderModules.blur,
                entryPoint: 'main'
            }
        }));
    }
    
    async processAirbrushGPU(brushData, targetTexture) {
        const pipeline = this.computePipelines.get('airbrush');
        const commandEncoder = this.device.createCommandEncoder();
        const computePass = commandEncoder.beginComputePass();
        
        computePass.setPipeline(pipeline);
        computePass.setBindGroup(0, this.createBindGroup(brushData, targetTexture));
        computePass.dispatchWorkgroups(
            Math.ceil(targetTexture.width / 8),
            Math.ceil(targetTexture.height / 8)
        );
        
        computePass.end();
        this.device.queue.submit([commandEncoder.finish()]);
        
        return targetTexture;
    }
}

class PixiV8WebCodecsExporter {
    constructor(pixiApp) {
        this.app = pixiApp;
        this.videoEncoder = null;
        this.setupWebCodecs();
    }
    
    async setupWebCodecs() {
        if (typeof VideoEncoder === 'undefined') {
            console.warn('WebCodecs not supported');
            return;
        }
        
        // High-quality encoder setup for Procreate-class output
        this.videoEncoder = new VideoEncoder({
            output: (chunk, metadata) => this.handleEncodedChunk(chunk, metadata),
            error: (error) => console.error('WebCodecs Encoder error:', error)
        });
        
        // 120FPS export capability
        await this.videoEncoder.configure({
            codec: 'avc1.42001f', // H.264 High Profile
            width: 1920,
            height: 1080,
            bitrate: 8000000, // 8Mbps for high quality
            framerate: 120, // 120FPS export
            avc: { format: 'avc' }
        });
        
        console.log('WebCodecs 120FPS export ready');
    }
    
    async exportAnimation120FPS(frameData, duration) {
        const totalFrames = Math.floor(duration * 120); // 120FPS
        const frameInterval = 1000 / 120; // 8.33ms per frame
        
        for (let frame = 0; frame < totalFrames; frame++) {
            const timestamp = frame * frameInterval * 1000; // microseconds
            
            // Render frame with PixiJS v8.11.0
            const renderTexture = this.app.renderer.generateTexture(this.app.stage);
            const canvas = this.app.renderer.extract.canvas(renderTexture);
            
            // Create VideoFrame for WebCodecs
            const videoFrame = new VideoFrame(canvas, {
                timestamp,
                duration: frameInterval * 1000
            });
            
            // Encode frame
            if (this.videoEncoder.state === 'configured') {
                this.videoEncoder.encode(videoFrame, { keyFrame: frame % 30 === 0 });
            }
            
            videoFrame.close();
            
            // Progress callback
            if (frame % 30 === 0) { // Every 30 frames
                this.onExportProgress?.(frame / totalFrames);
            }
        }
        
        await this.videoEncoder.flush();
        this.onExportComplete?.();
    }
    
    handleEncodedChunk(chunk, metadata) {
        // Handle encoded video chunk
        const chunkData = new Uint8Array(chunk.byteLength);
        chunk.copyTo(chunkData);
        
        this.onChunkReady?.(chunkData, metadata);
    }
}

class PixiV8ChromeAPIManager {
    constructor() {
        this.capabilities = null;
        this.currentTier = null;
        this.performanceMode = 'auto';
    }
    
    async detectAndConfigure() {
        // Chrome API機能検出
        this.capabilities = await this.detectCapabilities();
        
        // 最適Tier自動選択
        this.currentTier = this.selectOptimalTier(this.capabilities);
        
        // 設定適用
        await this.applyTierConfiguration(this.currentTier);
        
        console.log(`Chrome API Tier: ${this.currentTier.name}`);
        console.log('Capabilities:', this.capabilities);
        
        return this.currentTier;
    }
    
    async detectCapabilities() {
        const capabilities = {};
        
        // WebGPU検出・詳細情報取得
        capabilities.webgpu = await this.detectWebGPU();
        
        // OffscreenCanvas検出
        capabilities.offscreenCanvas = typeof OffscreenCanvas !== 'undefined';
        
        // WebCodecs検出・対応フォーマット確認
        capabilities.webCodecs = await this.detectWebCodecs();
        
        // PerformanceObserver検出
        capabilities.performanceObserver = typeof PerformanceObserver !== 'undefined';
        
        // Scheduling API検出
        capabilities.schedulingAPI = !!navigator.scheduling?.postTask;
        
        // 高リフレッシュレート検出
        capabilities.highRefreshRate = await this.detectHighRefreshRate();
        
        // GPU メモリ・性能推定
        capabilities.gpuMemory = await this.estimateGPUMemory();
        
        return capabilities;
    }
    
    async detectWebGPU() {
        if (!navigator.gpu) return { supported: false };
        
        try {
            const adapter = await navigator.gpu.requestAdapter({
                powerPreference: 'high-performance'
            });
            
            if (!adapter) return { supported: false };
            
            const device = await adapter.requestDevice({
                requiredFeatures: ['texture-compression-bc']
            });
            
            const limits = adapter.limits;
            
            return {
                supported: true,
                adapter: adapter.info,
                limits: {
                    maxTextureDimension2D: limits.maxTextureDimension2D,
                    maxBufferSize: limits.maxBufferSize,
                    maxComputeWorkgroupSizeX: limits.maxComputeWorkgroupSizeX
                },
                features: [...adapter.features],
                device: !!device
            };
        } catch (error) {
            console.warn('WebGPU detection failed:', error);
            return { supported: false, error: error.message };
        }
    }
    
    async detectWebCodecs() {
        if (typeof VideoEncoder === 'undefined') {
            return { supported: false };
        }
        
        try {
            // 120FPS H.264対応確認
            const config = {
                codec: 'avc1.42001f',
                width: 1920,
                height: 1080,
                bitrate: 8000000,
                framerate: 120
            };
            
            const isSupported = await VideoEncoder.isConfigSupported(config);
            
            return {
                supported: isSupported.supported,
                config: isSupported.config,
                formats: ['h264', 'vp8', 'vp9'], // 対応確認済み
                maxFramerate: isSupported.supported ? 120 : 60
            };
        } catch (error) {
            return { supported: false, error: error.message };
        }
    }
    
    async detectHighRefreshRate() {
        // 120Hz ディスプレイ・ペン入力対応確認
        try {
            if (screen.orientation) {
                // 推定リフレッシュレート取得
                const refreshRate = screen.orientation.angle === 0 ? 
                    await this.measureRefreshRate() : 60;
                
                return {
                    supported: refreshRate >= 120,
                    rate: refreshRate,
                    penSupport: 'onpointerdown' in window
                };
            }
            
            return { supported: false, rate: 60 };
        } catch {
            return { supported: false, rate: 60 };
        }
    }
    
    async estimateGPUMemory() {
        // GPU メモリ推定・4K キャンバス対応確認
        try {
            if (navigator.gpu) {
                const adapter = await navigator.gpu.requestAdapter();
                const limits = adapter?.limits;
                
                // メモリ推定（最大テクスチャサイズから算出）
                const maxTextureSize = limits?.maxTextureDimension2D || 2048;
                const estimatedMemory = Math.floor(maxTextureSize / 1024) * 256; // MB推定
                
                return {
                    estimated: estimatedMemory,
                    maxTextureSize,
                    supports4K: maxTextureSize >= 4096
                };
            }
            
            return { estimated: 512, maxTextureSize: 2048, supports4K: false };
        } catch {
            return { estimated: 256, maxTextureSize: 1024, supports4K: false };
        }
    }
    
    selectOptimalTier(capabilities) {
        // 検出結果に基づく最適Tier自動選択
        if (capabilities.webgpu?.supported && 
            capabilities.offscreenCanvas && 
            capabilities.webCodecs?.supported && 
            capabilities.highRefreshRate?.supported &&
            capabilities.gpuMemory?.supports4K) {
            return { ...CHROME_API_FALLBACK_STRATEGY.tier1, name: 'Tier1-Maximum' };
        }
        
        if (capabilities.offscreenCanvas && 
            capabilities.performanceObserver &&
            capabilities.gpuMemory?.estimated >= 512) {
            return { ...CHROME_API_FALLBACK_STRATEGY.tier2, name: 'Tier2-High' };
        }
        
        if (capabilities.performanceObserver) {
            return { ...CHROME_API_FALLBACK_STRATEGY.tier3, name: 'Tier3-Standard' };
        }
        
        return { ...CHROME_API_FALLBACK_STRATEGY.tier4, name: 'Tier4-Basic' };
    }
    
    async applyTierConfiguration(tier) {
        // 選択されたTierの設定適用
        window.RUNTIME_CONFIG = {
            ...window.DESIGN_CONFIG,
            performance: tier.performance,
            apis: tier.apis,
            features: tier.features,
            tier: tier.name
        };
        
        // 動的機能有効化
        await this.enableTierFeatures(tier);
        
        console.log(`Applied configuration for ${tier.name}`);
    }
    
    async enableTierFeatures(tier) {
        // Tier別機能有効化
        if (tier.performance.computeShaders) {
            await this.enableWebGPUCompute();
        }
        
        if (tier.performance.workers > 0) {
            await this.enableOffscreenWorkers(tier.performance.workers);
        }
        
        if (tier.features.includes('RealTimeExport')) {
            await this.enableWebCodecsExport();
        }
    }
}
