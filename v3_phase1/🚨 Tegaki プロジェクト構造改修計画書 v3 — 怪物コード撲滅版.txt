## 🔥 撲滅方針：「動くゴミ」から「理解可能な構造」へ

### ⚔️ 作戦1：Bootstrap完全シンプル化
```javascript
// 【現在】158行の複雑なBootstrap 
// 【修正後】15行のシンプルBootstrap

(function() {
    const scripts = [
        // 基盤3つだけで開始
        'js/utils/error-manager.js',
        'js/utils/config-manager.js', 
        'js/utils/event-bus.js',
        // 残りは即座読み込み
        'js/main.js'
    ];
    
    scripts.reduce((promise, script) => 
        promise.then(() => loadScript(script)), 
        Promise.resolve()
    ).then(() => new window.Tegaki.TegakiApplication());
    
    function loadScript(src) {
        return new Promise((resolve, reject) => {
            const script = document.createElement('script');
            script.src = src;
            script.onload = resolve;
            script.onerror = reject; // 握りつぶし禁止
            document.head.appendChild(script);
        });
    }
})();
```

### ⚔️ 作戦3：Manager責務明確化
```javascript
// 各Managerは「エラー処理なし」「単機能」「例外throw」
class CanvasManager {
    constructor(pixiApp) {
        this.app = pixiApp; // 受け取るだけ
        this.layers = new Map();
    }
    
    createLayer(name) {
        if (this.layers.has(name)) throw new Error(`Layer ${name} exists`);
        
        const layer = new PIXI.Container();
        this.layers.set(name, layer);
        this.app.stage.addChild(layer);
        return layer; // 例外throwかreturnのみ
    }
    // エラー処理は一切なし。例外は上位に委ね
}
```
