// 設定適用関数
export function applyLineQualitySettings(oglCore, settings) {
    if (!oglCore || !settings) return;
    
    // 基本描画設定適用
    if (settings.lineQuality) {
        oglCore.strokeConfig.adaptiveDistance = settings.lineQuality.adaptiveDistance;
        oglCore.strokeConfig.smoothingFactor = settings.lineQuality.velocitySmoothing * 0.2;
    }
    
    // タブレット設定適用
    if (settings.tablet) {
        Object.assign(oglCore.penConfig, {
            distanceScaling: settings.tablet.distanceScaling,
            pressureDeadZone: settings.tablet.deadZone,
            pointerContinuity: settings.tablet.pointerContinuity
        });
    }
    
    // 高度設定適用
    if (settings.advanced) {
        switch (settings.advanced.performanceMode) {
            case 'quality':
                oglCore.strokeConfig.minDistance = 0.5;
                oglCore.strokeConfig.segmentDensity = 2.0;
                break;
            case 'performance':
                oglCore.strokeConfig.minDistance = 1.2;
                oglCore.strokeConfig.segmentDensity = 1.0;
                break;
            default: // balanced
                oglCore.strokeConfig.minDistance = 0.8;
                oglCore.strokeConfig.segmentDensity = 1.5;
        }
    }
}

// 問題検出・自動修正システム
export class LineQualityDiagnostics {
    constructor(oglCore, mathEnhancer) {
        this.core = oglCore;
        this.math = mathEnhancer;
        this.issueHistory = [];
    }
    
    // 🔧 修正: リアルタイム品質監視
    analyzeStrokeQuality(stroke) {
        if (!stroke?.points || stroke.points.length < 3) return null;
        
        const quality = this.math.evaluateStrokeQuality(stroke.points);
        const issues = this.detectIssues(stroke, quality);
        
        if (issues.length > 0) {
            this.issueHistory.push({
                timestamp: Date.now(),
                strokeId: stroke.id || Date.now(),
                issues: issues,
                quality: quality.score
            });
            
            // 自動修正提案
            const recommendations = this.generateRecommendations(issues);
            return { quality, issues, recommendations };
        }
        
        return { quality, issues: [], recommendations: [] };
    }
    
    // 問題検出
    detectIssues(stroke, quality) {
        const issues = [];
        
        // モール・毛虫化検出
        if (quality.densityVariation > 2.0) {
            issues.push({
                type: 'segmentation',
                severity: 'high',
                description: 'モール・毛虫状の線が検出されました'
            });
        }
        
        // 描画距離問題検出
        const avgDistance = this.calculateAverageDistance(stroke.points);
        if (avgDistance < 0.5 && stroke.pointerType === 'pen') {
            issues.push({
                type: 'short_distance',
                severity: 'medium',
                description: '液タブペンでの描画距離が短すぎます'
            });
        }
        
        // 過度な曲率検出
        if (quality.avgCurvature > 0.4) {
            issues.push({
                type: 'excessive_curvature',
                severity: 'medium',
                description: '過度な曲率が検出されました'
            });
        }
        
        return issues;
    }
