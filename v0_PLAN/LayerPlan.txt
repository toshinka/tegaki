==============================
Futaba Drawing Tool
Layer Implementation Plan (独立版)
==============================

■目的
- スリム化済みのDOMパン/キャンバス移動環境にレイヤー機能を統合
- 削除ボタンなしで横フリックによるレイヤー削除
- サイドバーアイコンでレイヤーパネルON/OFF
- DOMパンやレイヤー変形との整合性保持
- 将来的な拡張（無限キャンバス、World座標系パン）を考慮

■シンボル辞典
シンボル　　　　　　責任範囲　　　　　　　　　主要メソッド
LayerManager　　　　レイヤー管理　　　　　　 addLayer(), setActiveLayer(), removeLayer(), transformLayer(), moveLayer()
LayerPanel　　　　　サイドバーUI制御　　　　　 setupIcon(), setupPanelSwipe(), createLayerIcon()

■理想的なフロー
1. レイヤー初期化
   - LayerManager 初期化
   - 初期レイヤー生成、stage に追加
   - LayerPanel 初期化

2. レイヤー選択・操作
   - サイドバーアイコン押下で LayerManager.setActiveLayer()
   - transformLayer() / moveLayer() で変形・移動
   - 横フリックで LayerManager.removeLayer()

3. DOMパン・描画との共存
   - coreEngine.isDomPan などフラグで互換性維持
   - transformLayer や moveLayer は DOM 位置・PIXI座標を意識

■フローチャート（簡易）
[LayerManager 初期化] → [新規レイヤー追加] → [LayerPanel 初期化] → [アイコン押下でアクティブ切替]
→ [横フリックで削除] → [変形・移動]

■UI設計
- サイドバーに SVG アイコン（CDN取得）
- アイコン押下でレイヤーパネル表示ON/OFF
- レイヤーパネル内で横フリック操作による削除判定
- 削除ボタンは設置せず、UIを簡潔に

■拡張ポイント
- 無限キャンバス対応
- World座標系パンとのハイブリッド
- 回転・拡縮・反転・移動などの transformLayer に統一
- DOMパン時でも PIXI.Container 座標系と同期可能

■作業AI向けポイント
- 命名を統一（LayerManager / LayerPanel / transformLayer / moveLayer）
- フローが明確
- メソッド単位で責任範囲を分離
- 将来的拡張を考慮した柔軟設計

/* ------------------------------
   LayerManager: レイヤー管理
------------------------------- */
class LayerManager {
    constructor(coreEngine) {
        this.core = coreEngine;
        this.layers = [];  // PIXI.Container配列
        this.activeLayer = null;
    }

    addLayer(layer) {
        this.layers.push(layer);
        if (!this.activeLayer) this.activeLayer = layer;
    }

    setActiveLayer(layer) {
        this.activeLayer = layer;
        this.updateLayerPanelUI();
    }

    removeLayer(layer) {
        const idx = this.layers.indexOf(layer);
        if (idx !== -1) {
            // DOM上はフリック操作で削除
            this.layers.splice(idx, 1);
            layer.parent.removeChild(layer);
        }
        if (this.activeLayer === layer) this.activeLayer = this.layers[0] || null;
        this.updateLayerPanelUI();
    }

    /* レイヤーパネル表示更新 */
    updateLayerPanelUI() {
        // ここでサイドバーのアイコンとパネル状態を同期
        this.layers.forEach(layer => {
            const panelEl = document.getElementById(`layer-panel-${layer.id}`);
            if (panelEl) panelEl.classList.toggle('active', layer === this.activeLayer);
        });
    }

    /* レイヤー変形共通メソッド */
    transformLayer(layer, options) {
        if (!layer) return;
        if (options.scale !== undefined) layer.scale.set(options.scale);
        if (options.rotation !== undefined) layer.rotation = options.rotation;
        if (options.flipX) layer.scale.x *= -1;
        if (options.flipY) layer.scale.y *= -1;
    }

    moveLayer(layer, dx, dy) {
        if (!layer) return;
        layer.x += dx;
        layer.y += dy;
    }
}

/* ------------------------------
   LayerPanel: サイドバーUI制御
------------------------------- */
class LayerPanel {
    constructor(layerManager) {
        this.layerManager = layerManager;
        this.panelEl = document.getElementById('layer-panel');
        this.iconEl = document.getElementById('layer-icon');
        this.setupIcon();
        this.setupPanelSwipe();
    }

    /* サイドバーアイコン押下でON/OFF切替 */
    setupIcon() {
        this.iconEl.addEventListener('click', () => {
            this.panelEl.classList.toggle('visible');
        });
    }

    /* 横フリックで削除操作 */
    setupPanelSwipe() {
        let startX = 0;
        this.panelEl.addEventListener('touchstart', e => {
            startX = e.touches[0].clientX;
        });
        this.panelEl.addEventListener('touchend', e => {
            const endX = e.changedTouches[0].clientX;
            const dx = endX - startX;
            if (Math.abs(dx) > 50) { // 横フリック判定
                const layerId = this.panelEl.dataset.activeLayerId;
                const layer = this.layerManager.layers.find(l => l.id === layerId);
                if (layer) this.layerManager.removeLayer(layer);
            }
        });
    }

    /* レイヤーパネル生成 (SVGアイコン取得はCDN) */
    createLayerIcon(layer) {
        const icon = document.createElement('img');
        icon.src = `https://cdn.example.com/icons/layer.svg`;
        icon.id = `layer-icon-${layer.id}`;
        icon.classList.add('layer-icon');
        icon.addEventListener('click', () => {
            this.layerManager.setActiveLayer(layer);
        });
        document.getElementById('sidebar').appendChild(icon);
    }
}

/* ------------------------------
   使用フロー例
------------------------------- */
// 1. LayerManager 初期化
const layerManager = new LayerManager(coreEngine);

// 2. 新規レイヤー作成
const newLayer = new PIXI.Container();
newLayer.id = `layer-${Date.now()}`;
layerManager.addLayer(newLayer);
coreEngine.stage.addChild(newLayer);

// 3. LayerPanel 初期化
const layerPanel = new LayerPanel(layerManager);
layerPanel.createLayerIcon(newLayer);
