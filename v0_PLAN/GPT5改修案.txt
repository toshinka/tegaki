原因は JavaScript の構文エラーとモジュール未定義 に起因しています。順を追って整理します。

1. エラー解析
(1) layer-manager.js:78 Uncaught SyntaxError: Unexpected end of input

原因：ファイル末尾に } や ) が閉じられていない、もしくは IIFE / 名前空間ラッパーが閉じられていない。

対策：

ファイル末尾に })(); または })(window); が抜けていないか確認

クラス定義の {} が閉じているか確認

VSCode 等で括弧の対応をチェック

(2) ui-manager.js:849 Uncaught SyntaxError: Unexpected token '}'

原因：

不要な閉じ括弧 } がある

または関数 / クラス / IIFE の終了位置のズレ

対策：

直前の function / class / (function(){ ... })() ブロックの括弧バランスを確認

行番号 849 前後をチェック

編集時にコメントアウトや改行で誤って } を入れた可能性あり

(3) main.html:186 [Main] Required module FutabaLayerManager not found

原因：

main.html 側で FutabaLayerManager が参照されているが、layer-manager.js 内でグローバルに登録されていない か、名前空間が違う

置き換え IIFE / 名前空間パターンを使っている場合、例えば window.MyApp.LayerModule と定義しているのに FutabaLayerManager を参照している可能性

対策：

モジュールをグローバル変数として正しく登録

window.FutabaLayerManager = class { ... }


または

window.MyApp = window.MyApp || {};
MyApp.FutabaLayerManager = class { ... }


main.html 側でも同じ名前で参照

this.layers = new MyApp.FutabaLayerManager(this);

(4) [Main] Initialization failed Error: Required libraries not found

原因：

上記のモジュール未定義が原因で初期化処理が失敗

対策：

syntax error を修正 → モジュール登録を確認 → main.html の初期化順序を確認

2. 推奨対応手順

括弧のバランスを確認

各 JS ファイル末尾の IIFE / クラスの閉じ括弧をチェック

layer-manager.js 末尾：})(); または })(window);

main.html で参照する名前と JS 側で登録している名前を完全一致させる

window.MyApp = window.MyApp || {};

(function(global){
  class FutabaLayerManager {
      constructor(main){
          this.main = main;
          this.layers = [];
      }

      register(main){
          this.main = main;
      }

      createLayer(name){
          const layer = {id:this.layers.length, name, transform:null};
          this.layers.push(layer);
          return layer;
      }

      moveLayer(fromIndex, toIndex){
          const payload = {fromIndex, toIndex};
          this.main.notify({type:'layers.moveRequest', payload});
      }

      moveConfirm(payload){
          const layer = this.layers.splice(payload.fromIndex,1)[0];
          this.layers.splice(payload.toIndex,0,layer);
          console.log('[Layer] moved:', payload);
      }
  }

  global.MyApp.FutabaLayerManager = FutabaLayerManager;
})(window);

window.MyApp = window.MyApp || {};

(function(global){
  class HistoryService {
      constructor(){ this.stack=[]; this.index=-1; }

      record(action){
          this.stack = this.stack.slice(0,this.index+1);
          this.stack.push(action);
          this.index++;
          console.log('[History] recorded:', action);
      }

      undo(){
          if(this.index<0) return;
          const action = this.stack[this.index--];
          console.log('[History] undo:', action);
      }

      redo(){
          if(this.index>=this.stack.length-1) return;
          const action = this.stack[++this.index];
          console.log('[History] redo:', action);
      }
  }

  global.MyApp.HistoryService = HistoryService;
})(window);
