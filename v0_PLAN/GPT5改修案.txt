🔎 原因の切り分け

main.html のイベントループ

mousemove / pointermove で呼ばれる座標更新が
worldToScreen / screenToWorld を経由していない。

そのため「ストロークはワールド座標」「カメラ移動はコンテナ座標」という分離が保たれていない。

MainController → EnginePosition 呼び出し

カメラ移動を layer.position でやっていた時代の呼び出しが残っている。

それが cameraContainer.position に統合されていない。

レイヤー生成

createLayer が stage 直下に追加されている可能性あり。

本来は cameraContainer.addChild(layer) でなければカメラ移動に追従しない。

✅ 改修案（main.html 側にも修正必須）

A. 初期化

await EnginePosition.init(document.getElementById("drawingCanvas"));
EnginePosition.createLayer("Layer 1");


createLayer を呼ぶときに、必ず cameraContainer に入れるようにする。

B. カメラ移動

// 例: ドラッグでカメラ移動
canvas.addEventListener("pointermove", (e)=>{
  if(draggingCamera){
    EnginePosition.setCamera(
      cameraStart.x - (e.clientX - dragStart.x),
      cameraStart.y - (e.clientY - dragStart.y)
    );
  }
});


setCamera(x,y) を必ず呼ぶ。

レイヤーや Graphics を直接 position 移動しない。

C. 座標変換

// スクリーン→ワールド
function toWorld(x,y){
  return {
    x: x + EnginePosition.getCamera().x,
    y: y + EnginePosition.getCamera().y
  };
}


ペン入力時にこの変換を通すことで、stroke が「ワールド座標」として保持される。

📝 まとめ

engine-position.js 側は「カメラコンテナ方式」にすでに改修済み。

でも main.html が古い呼び出しをしているので反映されていない。

main.html を修正して、EnginePosition.setCamera() と screen→world を必ず通すようにすれば、カメラ移動で描画内容がちゃんと追従するようになります。


※以下は修正サンプルです。参考に。

<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <title>ThreeSatellite Rev4 Sample</title>
  <script src="https://cdn.jsdelivr.net/npm/pixi.js@8.x/dist/pixi.min.js"></script>
  <script src="engine-position.js"></script>
  <script src="layer-tool-ui.js"></script>
  <script src="error-service.js"></script>
  <style>
    body { margin: 0; overflow: hidden; }
    #toolbar { position: fixed; top: 5px; left: 5px; background: #eee; padding: 5px; }
  </style>
</head>
<body>
  <div id="toolbar">
    <button id="moveCamera">Move</button>
    <button id="drawPen">Pen</button>
    <button id="eraser">Eraser</button>
    <button id="addLayer">+ Layer</button>
  </div>
  <canvas id="drawingCanvas"></canvas>

  <script>
    // ==============================
    // MainController 相当
    // ==============================
    let mode = "pen"; // pen | eraser | move
    let drawing = false;
    let lastPos = null;
    let cameraStart = { x: 0, y: 0 };
    let dragStart = { x: 0, y: 0 };

    async function init() {
      await EnginePosition.init(document.getElementById("drawingCanvas"));
      EnginePosition.createLayer("Layer 1");
      bindUI();
      bindInput();
    }

    function bindUI() {
      document.getElementById("moveCamera").onclick = () => { mode = "move"; };
      document.getElementById("drawPen").onclick = () => { mode = "pen"; };
      document.getElementById("eraser").onclick = () => { mode = "eraser"; };
      document.getElementById("addLayer").onclick = () => {
        EnginePosition.createLayer("Layer " + Date.now());
      };
    }

    function bindInput() {
      const canvas = document.getElementById("drawingCanvas");

      canvas.addEventListener("pointerdown", (e) => {
        if (mode === "move") {
          // カメラ移動開始
          const cam = EnginePosition.getCamera();
          cameraStart = { x: cam.x, y: cam.y };
          dragStart = { x: e.clientX, y: e.clientY };
          drawing = true;
        } else {
          // ペン/消しゴム
          const world = toWorld(e.clientX, e.clientY);
          lastPos = world;
          drawing = true;
        }
      });

      canvas.addEventListener("pointermove", (e) => {
        if (!drawing) return;

        if (mode === "move") {
          // カメラ移動
          const dx = e.clientX - dragStart.x;
          const dy = e.clientY - dragStart.y;
          EnginePosition.setCamera(cameraStart.x - dx, cameraStart.y - dy);
        } else {
          // ペン/消しゴム
          const world = toWorld(e.clientX, e.clientY);
          if (lastPos) {
            EnginePosition.drawLine(
              EnginePosition.getActiveLayer(),
              lastPos,
              world,
              mode
            );
          }
          lastPos = world;
        }
      });

      canvas.addEventListener("pointerup", () => {
        drawing = false;
        lastPos = null;
      });
    }

    // スクリーン座標 → ワールド座標変換
    function toWorld(x, y) {
      return EnginePosition.screenToWorld(x, y);
    }

    init();
  </script>
</body>
</html>

📌 ポイント

カメラ移動

EnginePosition.setCamera(x, y) を必ず呼ぶ

layer.position を直接動かさない

座標変換

toWorld(e.clientX, e.clientY) を必ず通す

ストロークはワールド座標で記録される

レイヤー生成

EnginePosition.createLayer() は cameraContainer にぶら下がるので、カメラ移動で一緒に動く
