/**
 * 🆕 COORDINATE_FEATURE: CoordinateManager統合初期化
 */
async initializeCoordinateManagerIntegration() {
    console.log('🔄 ToolManager座標統合初期化開始...');
    
    try {
        // CoordinateManager依存性確認
        if (!window.CoordinateManager) {
            throw new Error('CoordinateManager が必要です。座標統合を完了してください。');
        }
        
        // CoordinateManagerインスタンス生成
        this.coordinateManager = new window.CoordinateManager();
        
        // 座標統合設定確認
        const coordinateConfig = ConfigManager.getCoordinateConfig();
        this.coordinateIntegration = {
            enabled: coordinateConfig.integration?.managerCentralization || false,
            duplicateElimination: coordinateConfig.integration?.duplicateElimination || false,
            performance: coordinateConfig.performance || {}
        };
        
        if (!this.coordinateIntegration.enabled) {
            console.warn('⚠️ 座標統合が無効です。ConfigManagerで coordinate.integration.managerCentralization を true に設定してください。');
        }
        
        console.log('✅ ToolManager座標統合初期化完了');
        console.log('🔄 統合設定:', this.coordinateIntegration);
        
    } catch (error) {
        console.error('❌ ToolManager座標統合初期化失敗:', error);
        
        // CoordinateManagerなしでも動作継続
        this.coordinateManager = null;
        this.coordinateIntegration = {
            enabled: false,
            duplicateElimination: false,
            performance: {},
            error: error.message
        };
    }
}

/**
 * 🔄 座標統合状態取得
 */
getCoordinateIntegrationState() {
    return {
        coordinateManagerAvailable: !!this.coordinateManager,
        integrationEnabled: this.coordinateIntegration?.enabled || false,
        duplicateElimination: this.coordinateIntegration?.duplicateElimination || false,
        coordinateManagerState: this.coordinateManager ? 
            this.coordinateManager.getCoordinateState() : null,
        phase2Ready: !!(this.coordinateManager && 
                        this.coordinateIntegration?.enabled &&
                        this.coordinateIntegration?.duplicateElimination)
    };
}

/**
 * 🔄 座標統合診断実行
 */
runToolCoordinateIntegrationDiagnosis() {
    console.group('🔍 ToolManager座標統合診断');
    
    const state = this.getCoordinateIntegrationState();
    
    // 統合機能テスト
    const integrationTests = {
        coordinateManagerAvailable: !!this.coordinateManager,
        coordinateIntegrationEnabled: this.coordinateIntegration?.enabled || false,
        initializationMethodExists: typeof this.initializeCoordinateManagerIntegration === 'function',
        toolCoordinateProcessing: true // 今後ツール処理での座標統合実装時に詳細化
    };
    
    // 診断結果
    const diagnosis = {
        state,
        integrationTests,
        compliance: {
            coordinateUnified: integrationTests.coordinateManagerAvailable && integrationTests.coordinateIntegrationEnabled,
            duplicateEliminated: this.coordinateIntegration?.duplicateElimination || false,
            phase2Ready: state.phase2Ready
        }
    };
    
    console.log('📊 ToolManager座標統合診断結果:', diagnosis);
    
    // 推奨事項
    const recommendations = [];
    
    if (!integrationTests.coordinateManagerAvailable) {
        recommendations.push('CoordinateManagerの初期化が必要');
    }
    
    if (!integrationTests.coordinateIntegrationEnabled) {
        recommendations.push('座標統合設定の有効化が必要 (coordinate.integration.managerCentralization)');
    }
    
    if (!diagnosis.compliance.duplicateEliminated) {
        recommendations.push('重複排除設定の有効化を推奨 (coordinate.integration.duplicateElimination)');
    }
    
    if (recommendations.length > 0) {
        console.warn('⚠️ ToolManager推奨事項:', recommendations);
    } else {
        console.log('✅ ToolManager座標統合診断: 全ての要件を満たしています');
    }
    
    console.groupEnd();
    
    return diagnosis;

/**
 * 🔄 座標統合境界チェック（CoordinateManager使用版）
 */
checkBoundaryWithCoordinateIntegration(event) {
    if (!this.coordinateManager) {
        return this.checkBoundaryFallback(event);
    }
    
    try {
        // キャンバス要素の矩形取得
        const canvasRect = this.canvasElement?.getBoundingClientRect();
        if (!canvasRect) {
            throw new Error('キャンバス要素が見つかりません');
        }
        
        // CoordinateManager経由で統一座標取得
        const coordinates = this.coordinateManager.extractPointerCoordinates(
            event, 
            canvasRect, 
            null // PixiJSアプリは不要（境界チェックのみ）
        );
        
        // 座標妥当性確認
        if (!this.coordinateManager.validateCoordinateIntegrity(coordinates.canvas)) {
            throw new Error('無効な座標が検出されました');
        }
        
        // 境界内かどうかの判定
        const isInBounds = this.coordinateManager.isPointInBounds(
            coordinates.canvas,
            { x: 0, y: 0, width: canvasRect.width, height: canvasRect.height }
        );
        
        // 境界状態更新
        this.updateBoundaryStatus(coordinates.canvas, isInBounds);
        
        return {
            coordinates,
            inBounds: isInBounds,
            integratedProcessing: true
        };
        
    } catch (error) {
        if (window.ErrorManager) {
            window.ErrorManager.showError('boundary-coordinate-check', 
                `座標統合境界チェックエラー: ${error.message}`, 
                { event: event?.type }
            );
        }
        return this.checkBoundaryFallback(event);
    }
}

/**
 * 🔄 境界チェックフォールバック（緊急時用）
 */
checkBoundaryFallback(event) {
    console.warn('🔧 境界チェックフォールバック処理実行');
    
    try {
        const rect = this.canvasElement?.getBoundingClientRect();
        if (!rect) {
            return { inBounds: false, coordinates: null, fallbackProcessing: true };
        }
        
        const clientX = event.clientX || 0;
        const clientY = event.clientY || 0;
        
        // 基本的な境界判定
        const relativeX = clientX - rect.left;
        const relativeY = clientY - rect.top;
        const inBounds = relativeX >= 0 && relativeY >= 0 && 
                        relativeX <= rect.width && relativeY <= rect.height;
        
        return {
            coordinates: { x: relativeX, y: relativeY },
            inBounds,
            fallbackProcessing: true
        };
        
    } catch (error) {
        console.error('❌ フォールバック境界チェック失敗:', error);
        return { inBounds: false, coordinates: null, fallbackProcessing: true, error: error.message };
    }
}

/**
 * 📐 Phase2準備: レイヤー座標変換基盤（強化版）
 */
transformCoordinatesForLayer(coords, layerTransform = null) {
    if (!this.validateCoordinateIntegrity(coords)) {
        throw new Error('Invalid coordinates for layer transformation');
    }
    
    if (layerTransform) {
        console.log('🚀 Phase2準備: レイヤー座標変換実行');
        
        let transformedCoords = { ...coords };
        
        // 平行移動
        if (layerTransform.offsetX !== undefined) {
            transformedCoords.x += layerTransform.offsetX;
        }
        if (layerTransform.offsetY !== undefined) {
            transformedCoords.y += layerTransform.offsetY;
        }
        
        // スケール変換
        if (layerTransform.scaleX !== undefined) {
            transformedCoords.x *= layerTransform.scaleX;
        }
        if (layerTransform.scaleY !== undefined) {
            transformedCoords.y *= layerTransform.scaleY;
        }
        
        // 精度適用
        transformedCoords.x = this.applyPrecision(transformedCoords.x);
        transformedCoords.y = this.applyPrecision(transformedCoords.y);
        
        return transformedCoords;
    }
    
    return coords;
}
