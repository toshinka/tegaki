```javascript
// tools/pen-tool.js強化版
class PenTool {
    setupPointerEvents(canvas) {
        // Pointer Events API統合
        canvas.addEventListener('pointerdown', this.onPointerStart.bind(this));
        canvas.addEventListener('pointermove', this.onPointerMove.bind(this));
        canvas.addEventListener('pointerup', this.onPointerEnd.bind(this));
        
        // 筆圧対応
        canvas.addEventListener('pointermove', (e) => {
            if (e.pressure > 0) {
                this.adjustStrokeByPressure(e.pressure);
            }
        });
    }
    
    adjustStrokeByPressure(pressure) {
        // 筆圧: 0.0～1.0 → 線幅: 最小50%～最大150%
        const baseWidth = this.settings.strokeWidth;
        const pressureWidth = baseWidth * (0.5 + pressure);
        this.currentStroke.lineStyle.width = pressureWidth;
    }
}
```

```javascript
// managers/canvas-manager.js拡張
class CanvasManager {
    constructor(app) {
        this.app = app;
        this.viewport = {
            x: 0, y: 0,
            scaleX: 1, scaleY: 1,
            rotation: 0
        };
        this.panMode = false;
        this.setupCanvasInteraction();
    }
    
    setupCanvasInteraction() {
        const canvas = this.app.view;
        
        // Space+ドラッグでパン
        canvas.addEventListener('pointerdown', (e) => {
            if (this.panMode) {
                this.startPan(e);
            }
        });
        
        canvas.addEventListener('pointermove', (e) => {
            if (this.isPanning) {
                this.updatePan(e);
            }
        });
        
        canvas.addEventListener('pointerup', () => {
            this.endPan();
        });
    }
    
    panCanvas(deltaX, deltaY) {
        this.viewport.x += deltaX;
        this.viewport.y += deltaY;
        this.updateCanvasTransform();
    }
    
    resetView() {
        this.viewport = { x: 0, y: 0, scaleX: 1, scaleY: 1, rotation: 0 };
        this.updateCanvasTransform();
        console.log('🔄 ビューリセット完了');
    }
    
    updateCanvasTransform() {
        if (this.app.stage) {
            this.app.stage.position.set(this.viewport.x, this.viewport.y);
            this.app.stage.scale.set(this.viewport.scaleX, this.viewport.scaleY);
            this.app.stage.rotation = this.viewport.rotation;
        }
    }
}
```

```javascript
class CanvasResizePopup {

                <div class="preset-label">${preset.name}</div>

            container.appendChild(button);
        });

    getPresetIcon(preset) {
        // プリセット用の小型アイコン生成
        const aspectRatio = preset.width / preset.height;
        let iconSvg;
        
        if (aspectRatio > 2) {
            // 横長
            iconSvg = '<rect x="2" y="7" width="20" height="10" stroke="currentColor" fill="none" stroke-width="1"/>';
        } else if (aspectRatio < 0.8) {
            // 縦長
            iconSvg = '<rect x="7" y="2" width="10" height="20" stroke="currentColor" fill="none" stroke-width="1"/>';
        } else {
            // 正方形
            iconSvg = '<rect x="4" y="4" width="16" height="16" stroke="currentColor" fill="none" stroke-width="1"/>';
        }
        
        return `<svg width="24" height="24" viewBox="0 0 24 24">${iconSvg}</svg>`;
    }
    
    applyPreset(preset) {
        // キャンバスサイズのみ変更（中央寄せなし）
        this.appCore.resizeCanvas(preset.width, preset.height);
        console.log(`📐 プリセット適用: ${preset.name} (${preset.width}×${preset.height})`);
    }
}
```

```javascript
// js/app-core.js改修
class AppCore {
    createPixiApplication(options) {
        // 高DPI無効化（コメントアウトで無効）
        const pixiOptions = {
            width: options.width,
            height: options.height,
            backgroundColor: 0xf0e0d6,
            antialias: true,
            // resolution: window.devicePixelRatio || 1,  // コメントアウト
            resolution: 1, // 固定値で精度保証
            autoDensity: false // 自動密度調整も無効化
        };
        
        console.log(`📱 高DPI無効化: resolution固定1.0 (実際のDPR: ${window.devicePixelRatio || 1})`);
        
        return new PIXI.Application(pixiOptions);
    }
    
    resizeCanvas(width, height) {
        if (this.app) {
            // 正確なサイズ設定
            this.app.renderer.resize(width, height);
            this.app.view.style.width = `${width}px`;
            this.app.view.style.height = `${height}px`;
            
            console.log(`📐 キャンバスリサイズ: ${width}×${height} (精度保証)`);
            
            // サイズ検証
            this.validateCanvasSize(width, height);
        }
    }
    
    validateCanvasSize(expectedWidth, expectedHeight) {
        const canvas = this.app.view;
        const actualWidth = canvas.width;
        const actualHeight = canvas.height;
        
        if (actualWidth !== expectedWidth || actualHeight !== expectedHeight) {
            console.warn(`⚠️ サイズ不一致: 期待${expectedWidth}×${expectedHeight} / 実際${actualWidth}×${actualHeight}`);
        } else {
            console.log(`✅ サイズ精度保証確認: ${actualWidth}×${actualHeight}`);
        }
    }
}
```

```javascript
// ui/pen-popup.js新規実装
class PenPopup {
    constructor(toolManager) {
        this.toolManager = toolManager;
        this.maxPenSize = 500;
        this.presets = this.loadPresets();
        this.setupPopupUI();
    }
    
    setupPopupUI() {
        const popup = document.getElementById('pen-popup');
        if (!popup) return;
        
        popup.innerHTML = `
            <div class="pen-popup-content">
                <div class="pen-preview-area">
                    <div class="pen-preview-outer-circle">
                        <div class="pen-preview-circle" id="pen-preview"></div>
                    </div>
                </div>
                
                <div class="pen-controls">
                    <label>サイズ: <span id="pen-size-value">10</span></label>
                    <input type="range" id="pen-size-slider" min="1" max="${this.maxPenSize}" value="10">
                    
                    <label>不透明度: <span id="pen-opacity-value">100</span>%</label>  
                    <input type="range" id="pen-opacity-slider" min="0" max="100" value="100">
                </div>
                
                <div class="pen-presets">
                    <div class="preset-row" id="preset-buttons"></div>
                    <button class="reset-presets-btn" id="reset-presets">全リセット</button>
                </div>
            </div>
        `;
        
        this.setupEventListeners();
        this.createPresetButtons();
        this.updatePreview();
    }
    
    setupEventListeners() {
        const sizeSlider = document.getElementById('pen-size-slider');
        const opacitySlider = document.getElementById('pen-opacity-slider');
        const resetBtn = document.getElementById('reset-presets');
        
        // リアルタイムプレビュー更新
        sizeSlider.addEventListener('input', () => {
            this.updatePreview();
            this.updateToolSettings();
        });
        
        opacitySlider.addEventListener('input', () => {
            this.updatePreview();
            this.updateToolSettings();
        });
        
        // リセット機能
        resetBtn.addEventListener('click', () => {
            this.resetAllPresets();
        });
    }
    
    updatePreview() {
        const preview = document.getElementById('pen-preview');
        const sizeSlider = document.getElementById('pen-size-slider');
        const opacitySlider = document.getElementById('pen-opacity-slider');
        const sizeValue = document.getElementById('pen-size-value');
        const opacityValue = document.getElementById('pen-opacity-value');
        
        if (!preview || !sizeSlider || !opacitySlider) return;
        
        const size = parseInt(sizeSlider.value);
        const opacity = parseInt(opacitySlider.value);
        
        // 外枠◯内に収まるサイズ計算（最大60px）
        const maxPreviewSize = 60;
        const previewSize = Math.min(maxPreviewSize, (size / this.maxPenSize) * maxPreviewSize);
        
        // プレビュー更新
        preview.style.width = `${previewSize}px`;
        preview.style.height = `${previewSize}px`;
        preview.style.opacity = opacity / 100;
        preview.style.backgroundColor = this.toolManager.getCurrentColor();
        
        // 値表示更新
        sizeValue.textContent = size;
        opacityValue.textContent = opacity;
        
        console.log(`🖋️ ペンプレビュー更新: サイズ${size} 不透明度${opacity}% 表示${previewSize}px`);
    }
    
    createPresetButtons() {
        const container = document.getElementById('preset-buttons');
        if (!container) return;
        
        container.innerHTML = '';
        
        this.presets.forEach((preset, index) => {
            const button = document.createElement('button');
            button.className = 'preset-button';
            button.onclick = () => this.selectPreset(index);
            
            // 四角枠のみ（塗りつぶしなし）
            button.innerHTML = `
                <div class="preset-frame">
                    <div class="preset-number">${index + 1}</div>
                    <div class="preset-info">
                        <div>サイズ: ${preset.size}</div>
                        <div>不透明度: ${preset.opacity}%</div>
                    </div>
                </div>
            `;
            
            container.appendChild(button);
        });
    }
    
    selectPreset(index) {
        if (index >= 0 && index < this.presets.length) {
            const preset = this.presets[index];
            
            // UI更新
            document.getElementById('pen-size-slider').value = preset.size;
            document.getElementById('pen-opacity-slider').value = preset.opacity;
            
            // プレビュー更新
            this.updatePreview();
            this.updateToolSettings();
            
            console.log(`🎯 プリセット選択: ${index + 1} (サイズ${preset.size} 不透明度${preset.opacity}%)`);
        }
    }
    
    saveCurrentAsPreset(index) {
        const size = parseInt(document.getElementById('pen-size-slider').value);
        const opacity = parseInt(document.getElementById('pen-opacity-slider').value);
        
        this.presets[index] = { size, opacity };
        this.savePresets();
        this.createPresetButtons(); // UI更新
        
        console.log(`💾 プリセット保存: ${index + 1} (サイズ${size} 不透明度${opacity}%)`);
    }
    
    resetAllPresets() {
        this.presets = this.getDefaultPresets();
        this.savePresets();
        this.createPresetButtons();
        console.log('🔄 全プリセットリセット完了');
    }
    
    loadPresets() {
        try {
            const saved = localStorage.getItem('pen-presets');
            return saved ? JSON.parse(saved) : this.getDefaultPresets();
        } catch (error) {
            console.warn('⚠️ プリセット読み込みエラー:', error);
            return this.getDefaultPresets();
        }
    }
    
    savePresets() {
        try {
            localStorage.setItem('pen-presets', JSON.stringify(this.presets));
        } catch (error) {
            console.warn('⚠️ プリセット保存エラー:', error);
        }
    }
    
    getDefaultPresets() {
        return [
            { size: 5, opacity: 100 },   // 細ペン
            { size: 10, opacity: 100 },  // 標準ペン  
            { size: 20, opacity: 100 },  // 太ペン
            { size: 50, opacity: 50 },   // ぼかしペン
            { size: 100, opacity: 25 }   // エアブラシ
        ];
    }
}
```

#### 改修内容

- **線のクリッピング**: キャンバス境界で正確に切り取り
- **マージン設定**: 境界ギリギリでの描画しやすさ向上

#### 技術実装
```javascript
// tools/pen-tool.js拡張
class PenTool {
    onPointerMove(e) {
        if (!this.isDrawing) return;
        
        // マウス座標取得（境界外も許可）
        const rect = this.app.view.getBoundingClientRect();
        const rawX = e.clientX - rect.left;
        const rawY = e.clientY - rect.top;
        
        // キャンバス境界取得
        const canvasBounds = {
            left: 0,
            top: 0, 
            right: this.app.view.width,
            bottom: this.app.view.height
        };
        
        // 描画座標計算（境界外から内側への線を許可）
        const drawPoint = this.calculateDrawPoint(rawX, rawY, canvasBounds);
        
        if (drawPoint) {
            this.addPointToCurrentStroke(drawPoint.x, drawPoint.y, e.pressure || 0.5);
        }
    }
    
    calculateDrawPoint(x, y, bounds) {
        // 前回の点と現在の点で線分を作成
        const prevPoint = this.getLastStrokePoint();
        
        if (!prevPoint) {
            // 最初の点：キャンバス内であれば使用
            if (this.isInsideCanvas(x, y, bounds)) {
                return { x, y };
            }
            return null;
        }
        
        // 線分とキャンバス境界の交点を計算
        const intersections = this.calculateCanvasIntersections(
            prevPoint.x, prevPoint.y, x, y, bounds
        );
        
        if (intersections.length > 0) {
            // 最適な交点を選択（通常は最初の交点）
            return intersections[0];
        }
        
        // キャンバス内であればそのまま使用
        if (this.isInsideCanvas(x, y, bounds)) {
            return { x, y };
        }
        
        return null;
    }
    
    calculateCanvasIntersections(x1, y1, x2, y2, bounds) {
        const intersections = [];
        
        // 上辺との交点
        const topIntersect = this.lineIntersection(
            x1, y1, x2, y2,
            bounds.left, bounds.top, bounds.right, bounds.top
        );
        if (topIntersect && this.isValidIntersection(topIntersect, bounds)) {
            intersections.push(topIntersect);
        }
        
        // 下辺との交点
        const bottomIntersect = this.lineIntersection(
            x1, y1, x2, y2,
            bounds.left, bounds.bottom, bounds.right, bounds.bottom
        );
        if (bottomIntersect && this.isValidIntersection(bottomIntersect, bounds)) {
            intersections.push(bottomIntersect);
        }
        
        // 左辺との交点
        const leftIntersect = this.lineIntersection(
            x1, y1, x2, y2,
            bounds.left, bounds.top, bounds.left, bounds.bottom
        );
        if (leftIntersect && this.isValidIntersection(leftIntersect, bounds)) {
            intersections.push(leftIntersect);
        }
        
        // 右辺との交点
        const rightIntersect = this.lineIntersection(
            x1, y1, x2, y2,
            bounds.right, bounds.top, bounds.right, bounds.bottom
        );
        if (rightIntersect && this.isValidIntersection(rightIntersect, bounds)) {
            intersections.push(rightIntersect);
        }
        
        return intersections;
    }
    
    lineIntersection(x1, y1, x2, y2, x3, y3, x4, y4) {
        // 線分の交点計算（数学的計算）
        const denom = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
        
        if (Math.abs(denom) < 0.0001) {
            return null; // 平行線
        }
        
        const t = ((x1 - x3) * (y3 - y4) - (y1 - y3) * (x3 - x4)) / denom;
        const u = -((x1 - x2) * (y1 - y3) - (y1 - y2) * (x1 - x3)) / denom;
        
        // 線分内に交点があるかチェック
        if (t >= 0 && t <= 1 && u >= 0 && u <= 1) {
            return {
                x: x1 + t * (x2 - x1),
                y: y1 + t * (y2 - y1)
            };
        }
        
        return null;
    }
    
    isInsideCanvas(x, y, bounds) {
        return x >= bounds.left && x <= bounds.right &&
               y >= bounds.top && y <= bounds.bottom;
    }
    
    isValidIntersection(point, bounds) {
        // 交点がキャンバス境界上にあるかチェック
        return point.x >= bounds.left && point.x <= bounds.right &&
               point.y >= bounds.top && point.y <= bounds.bottom;
    }
}
```

- **単一責任原則**: 各クラスは単一機能のみ担当
- **開放閉鎖原則**: 機能拡張時は既存コード変更なし
- **リスコフ置換原則**: サブクラスは基底クラスの代替可能
- **インターフェース分離**: 不要な依存関係を避ける
- **依存性逆転**: 抽象に依存・具象に依存しない

### Pure JavaScript維持
- **ESM/TypeScript禁止**: `import`/`export`文使用禁止
- **Pure JavaScript**: 完全なPure JavaScript維持  
- **グローバル登録**: `window`オブジェクトでのクラス公開
- **Script読み込み**: `<script>`タグでの順次読み込み
