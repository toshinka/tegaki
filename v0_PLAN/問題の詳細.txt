        /**

         * Phase1.1ss3rev2 - エントリーポイント

         * ⚡ 重要修正: tweedle.js依存関係解決
         * 📋 V8_MIGRATION: Application.init()対応予定
         */
        
        console.log('🎨 ふたば☆ちゃんねる風ベクターお絵描きツール Phase1.1ss3rev2 起動開始...');
        
        // DOMContentLoaded後に初期化実行
        document.addEventListener('DOMContentLoaded', async () => {
            try {
                console.log('📦 依存関係チェック開始...');
                
                // 必須ライブラリ確認
                if (!window.PIXI) {
                    throw new Error('PixiJS が読み込まれていません');
                }
                console.log(`✅ PixiJS v${PIXI.VERSION} 検出`);
                
                // tweedle.js確認（重要修正）
                if (window.tweedle_js || window.TWEEDLE) {
                    console.log('✅ tweedle.js 検出 - @pixi/ui完全対応');
                } else {
                    console.warn('⚠️ tweedle.js 未検出 - 基本機能で継続');
                }
                
                // PixiJS拡張機能初期化
                if (window.PixiExtensions) {
                    console.log('🔧 PixiJS拡張機能初期化開始...');
                    await window.PixiExtensions.initialize();
                    
                    const stats = window.PixiExtensions.getStats();
                    console.log(`📊 統合成功: ${stats.coverage} (${stats.loaded}/${stats.total})`);
                } else {
                    console.warn('⚠️ PixiJS拡張機能が見つかりません');
                }
                
                // メインアプリケーション初期化
                if (window.VectorDrawingApp) {
                    console.log('🚀 メインアプリケーション初期化開始...');
                    await window.VectorDrawingApp.initialize();
                    console.log('✅ アプリケーション初期化完了');
                } else {
                    console.error('❌ メインアプリケーションが見つかりません');
                }
                
                console.log('🎉 ふたば☆ちゃんねる風ベクターお絵描きツール 起動完了！');
                
            } catch (error) {
                console.error('❌ 初期化エラー:', error);
                
                // エラーハンドリング - ユーザーに通知
                const errorDiv = document.createElement('div');
                errorDiv.className = 'error-message';
                errorDiv.innerHTML = `
                    <h3>⚠️ 初期化エラー</h3>
                    <p>${error.message}</p>
                    <p>ページをリロードしてください。問題が継続する場合は開発者にお知らせください。</p>
                    <button onclick="location.reload()">リロード</button>
                `;
                document.body.appendChild(errorDiv);
            }
        });
        
        // v8移行準備コメント
        /*
         * 📋 V8_MIGRATION: PixiJS v8移行時の変更予定
         * 
         * 現在 (v7):
         * - const app = new PIXI.Application(config);
         * 
         * v8予定:
         * - const app = new PIXI.Application();
         * - await app.init(config);
         * 
         * 互換性レイヤーでv8移行をスムーズに実行予定
         */
    </script>
