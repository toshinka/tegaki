### Phase2: 🔧 統合改修対応（構文修正後）

#### Step2-1: ConfigManager統合対応
```javascript
// settings-manager.js とConfigManagerの競合解決

// Option A: 段階的移行（推奨）
class SettingsManager {
    constructor() {
        // ConfigManagerが存在する場合は委譲
        this.useConfigManager = !!window.configManager;
        if (!this.useConfigManager) {
            this.settings = {};
        }
    }
    
    get(key) {
        if (this.useConfigManager) {
            return window.configManager.get(key);
        }
        return this.settings[key];
    }
}

// Option B: 完全統合（スリム化準備）
if (window.configManager) {
    // ConfigManagerが存在する場合はaliasのみ提供
    window.settingsManager = {
        get: (key) => window.configManager.get(key),
        set: (key, value) => window.configManager.set(key, value)
    };
} else {
    // フォールバック実装
    window.settingsManager = new SettingsManager();
}
```

### 3. AppCore getAppState実装

```javascript
class AppCore {
    constructor() {
        this.state = {
            initialized: false,
            currentTool: 'pen',
            canvas: {
                width: 800,
                height: 600
            },
            brush: {
                size: 5,
                color: '#000000'
            }
        };
    }
    
    // API-Gateway対応のgetAppState実装
    getAppState() {
        return {
            // 基本状態
            initialized: this.state.initialized,
            currentTool: this.state.currentTool,
            
            // キャンバス情報
            canvas: {
                width: this.state.canvas.width,
                height: this.state.canvas.height,
                pixiApp: !!this.pixiApp
            },
            
            // ツール情報
            brush: this.state.brush,
            
            // 統合管理システム状態
            managers: {
                config: !!window.configManager,
                error: !!window.errorManager,
                initialization: !!window.initializationManager
            },
            
            // 統計情報
            stats: {
                totalDrawTime: this.getTotalDrawTime(),
                commandCount: this.getCommandCount()
            }
        };
    }
    
    getTotalDrawTime() {
        return this.state.totalDrawTime || 0;
    }
    
    getCommandCount() {
        return this.state.commandCount || 0;
    }
}
```
