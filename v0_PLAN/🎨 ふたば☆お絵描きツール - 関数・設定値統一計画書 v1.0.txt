### 2. 循環依存の排除

#### 📍 現状の問題
```javascript
// 潜在的循環依存
AppCore → DrawingToolSystem → AppCore (参照)
UIController → DrawingToolSystem → UIController (イベント)
```

#### ✅ 解決後の構造
```javascript
// イベント駆動アーキテクチャによる循環依存排除
class EventBus {
    static events = new Map();
    
    static emit(eventType, data) {
        const listeners = this.events.get(eventType) || [];
        listeners.forEach(callback => callback(data));
    }
    
    static on(eventType, callback) {
        if (!this.events.has(eventType)) {
            this.events.set(eventType, []);
        }
        this.events.get(eventType).push(callback);
    }
    
    static off(eventType, callback) {
        const listeners = this.events.get(eventType) || [];
        this.events.set(eventType, listeners.filter(cb => cb !== callback));
    }
}

// 使用例：循環依存の排除
class DrawingToolSystem {
    setBrushSize(size) {
        this.brushSize = size;
        // 直接UIControllerを呼ばずにイベント発行
        EventBus.emit('brushSizeChanged', { size });
    }
}

class UIController {
    init() {
        // イベント監視でUI更新
        EventBus.on('brushSizeChanged', ({ size }) => {
            this.updateSizePresets();
            this.updateSliderValue('pen-size-slider', size);
        });
    }
}
```
