class CanvasManager {
  constructor() {
    this.drawContainer = null;
    this.isV8Ready = false;
  }

    // PIXI.Application完全初期化まで待機
    await this.app.init();

  }
  
  getDrawContainer() {
    if (!this.isV8Ready || !this.drawContainer) {
