```javascript
// managers/boundary-manager.js（新規作成）
/**
 * 🎯 キャンバス境界管理システム
 * 🎯 AI_WORK_SCOPE: 境界越えイベント処理・境界判定・境界描画制御
 * 🎯 DEPENDENCIES: ConfigManager, ErrorManager, EventBus, coordinate-manager.js
 * 🎯 UNIFIED: ConfigManager(境界設定), ErrorManager(境界エラー), EventBus(境界イベント)
 * 🎯 ISOLATION_TEST: ✅ 単体テスト可能
 */

class BoundaryManager {
    constructor() {
        this.validateUnifiedSystems();
        this.config = ConfigManager.get('canvas.boundary');
        this.coordinateManager = null; // 後で依存注入
        
        // 境界追跡状態
        this.isTrackingOutside = false;
        this.outsidePointer = null;
        this.boundaryMargin = this.config.margin || 20;
        this.expandedHitArea = null;
        
        console.log('🎯 BoundaryManager 初期化完了（統一システム統合版）');
    }
    
    validateUnifiedSystems() {
        const required = ['ConfigManager', 'ErrorManager', 'EventBus'];
        const missing = required.filter(sys => !window[sys]);
        if (missing.length > 0) {
            throw new Error(`BoundaryManager: 統一システム依存不足: ${missing.join(', ')}`);
        }
    }
    
    /**
     * 🎯 境界管理システム初期化
     */
    initialize(canvasElement, coordinateManager) {
        try {
            this.coordinateManager = coordinateManager;
            this.canvasElement = canvasElement;
            
            // 境界追跡システム設定
            this.initializeBoundaryTracking();
            
            // 拡張ヒットエリア作成
            this.createExpandedHitArea();
            
            console.log('✅ BoundaryManager 初期化完了');
            
            EventBus.safeEmit('boundary.manager.initialized', {
                margin: this.boundaryMargin,
                config: this.config
            });
            
        } catch (error) {
            ErrorManager.showError('boundary-init', 
                `境界管理初期化失敗: ${error.message}`, 
                { config: this.config }
            );
            throw error;
        }
    }
    
    /**
     * 🎯 グローバル境界追跡システム
     */
    initializeBoundaryTracking() {
        // document全体でPointerイベント監視
        document.addEventListener('pointerdown', this.handleGlobalPointerDown.bind(this));
        document.addEventListener('pointermove', this.handleGlobalPointerMove.bind(this));
        document.addEventListener('pointerup', this.handleGlobalPointerUp.bind(this));
        
        console.log('✅ グローバル境界追跡システム開始');
    }
    
    /**
     * 🎯 拡張ヒットエリア作成
     */
    createExpandedHitArea() {
        if (!this.canvasElement) return null;
        
        const rect = this.canvasElement.getBoundingClientRect();
        this.expandedHitArea = {
            left: rect.left - this.boundaryMargin,
            top: rect.top - this.boundaryMargin,
            right: rect.right + this.boundaryMargin,
            bottom: rect.bottom + this.boundaryMargin
        };
        
        return this.expandedHitArea;
    }
    
    /**
     * 🎯 グローバルポインターダウン処理
     */
    handleGlobalPointerDown(event) {
        try {
            const canvasRect = this.canvasElement.getBoundingClientRect();
            const isInsideCanvas = this.isPointInCanvas(event.clientX, event.clientY, canvasRect);
            
            if (!isInsideCanvas) {
                // キャンバス外ポインター検出
                this.isTrackingOutside = true;
                this.outsidePointer = {
                    id: event.pointerId,
                    startX: event.clientX,
                    startY: event.clientY,
                    timestamp: Date.now()
                };
                
                EventBus.safeEmit('boundary.outside.start', {
                    pointer: this.outsidePointer,
                    canvasRect
                });
                
                console.log(`🎯 境界外ポインター検出: (${event.clientX}, ${event.clientY})`);
            }
            
        } catch (error) {
            ErrorManager.showError('boundary-tracking', 
                `境界追跡エラー: ${error.message}`, 
                { event: event.type, pointerId: event.pointerId }
            );
        }
    }
    
    /**
     * 🎯 グローバルポインター移動処理
     */
    handleGlobalPointerMove(event) {
        if (!this.isTrackingOutside || event.pointerId !== this.outsidePointer?.id) {
            return;
        }
        
        try {
            const canvasRect = this.canvasElement.getBoundingClientRect();
            const isInsideCanvas = this.isPointInCanvas(event.clientX, event.clientY, canvasRect);
            
            if (isInsideCanvas) {
                // 境界越えイン検出
                this.handleBoundaryCrossIn(event, canvasRect);
            }
            
        } catch (error) {
            ErrorManager.showError('boundary-cross', 
                `境界越え処理エラー: ${error.message}`, 
                { event: event.type, pointerId: event.pointerId }
            );
        }
    }
    
    /**
     * 🎯 境界越えイン処理（核心機能）
     */
    handleBoundaryCrossIn(event, canvasRect) {
        if (!this.coordinateManager) {
            console.warn('⚠️ CoordinateManager が利用できません');
            return;
        }
        
        // キャンバス座標変換
        const canvasCoords = this.coordinateManager.screenToCanvas(
            event.clientX, event.clientY, canvasRect
        );
        
        // 境界越え描画開始イベント発火
        EventBus.safeEmit('boundary.cross.in', {
            position: canvasCoords,
            pressure: event.pressure || 0.5,
            pointerId: event.pointerId,
            originalEvent: event,
            fromOutside: true
        });
        
        console.log(`🎯 境界越えイン: (${canvasCoords.x.toFixed(1)}, ${canvasCoords.y.toFixed(1)})`);
        
        // 追跡状態リセット
        this.isTrackingOutside = false;
        this.outsidePointer = null;
    }
    
    /**
     * 🎯 グローバルポインターアップ処理
     */
    handleGlobalPointerUp(event) {
        if (this.isTrackingOutside && event.pointerId === this.outsidePointer?.id) {
            // 境界外で終了
            this.isTrackingOutside = false;
            this.outsidePointer = null;
            
            EventBus.safeEmit('boundary.outside.end', {
                pointerId: event.pointerId
            });
        }
    }
    
    /**
     * 🎯 点がキャンバス内かチェック
     */
    isPointInCanvas(clientX, clientY, canvasRect) {
        return clientX >= canvasRect.left && 
               clientX <= canvasRect.right && 
               clientY >= canvasRect.top && 
               clientY <= canvasRect.bottom;
    }
    
    /**
     * 🎯 境界状態取得（診断・デバッグ用）
     */
    getBoundaryState() {
        return {
            isTrackingOutside: this.isTrackingOutside,
            outsidePointer: this.outsidePointer,
            boundaryMargin: this.boundaryMargin,
            expandedHitArea: this.expandedHitArea,
            config: this.config
        };
    }
    
    /**
     * 🎯 境界管理システム破棄
     */
    destroy() {
        document.removeEventListener('pointerdown', this.handleGlobalPointerDown.bind(this));
        document.removeEventListener('pointermove', this.handleGlobalPointerMove.bind(this));
        document.removeEventListener('pointerup', this.handleGlobalPointerUp.bind(this));
        
        this.isTrackingOutside = false;
        this.outsidePointer = null;
        
        console.log('🎯 BoundaryManager 破棄完了');
    }
}

// グローバル登録
if (typeof window !== 'undefined') {
    window.BoundaryManager = BoundaryManager;
    console.log('🎯 BoundaryManager グローバル登録完了');
}
```

```javascript
// js/utils/coordinate-manager.js（新規作成）
/**
 * 📐 座標変換管理システム
 * 🎯 AI_WORK_SCOPE: スクリーン座標・キャンバス座標・PixiJS座標変換
 * 🎯 DEPENDENCIES: ConfigManager, ErrorManager
 * 🎯 UNIFIED: ConfigManager(キャンバス設定), ErrorManager(座標エラー)
 * 🎯 ISOLATION_TEST: ✅ 単体テスト可能・数学的確定処理
 */

class CoordinateManager {
    constructor() {
        this.validateUnifiedSystems();
        this.canvasConfig = ConfigManager.getCanvasConfig();
        
        // 座標変換基準
        this.canvasWidth = this.canvasConfig.width;
        this.canvasHeight = this.canvasConfig.height;
        
        console.log('📐 CoordinateManager 初期化完了（統一システム統合版）');
    }
    
    validateUnifiedSystems() {
        const required = ['ConfigManager', 'ErrorManager'];
        const missing = required.filter(sys => !window[sys]);
        if (missing.length > 0) {
            throw new Error(`CoordinateManager: 統一システム依存不足: ${missing.join(', ')}`);
        }
    }
    
    /**
     * 📐 スクリーン座標 → キャンバス座標変換
     */
    screenToCanvas(screenX, screenY, canvasRect) {
        try {
            if (!canvasRect) {
                throw new Error('canvasRect が必要です');
            }
            
            // スケール比率計算
            const scaleX = this.canvasWidth / canvasRect.width;
            const scaleY = this.canvasHeight / canvasRect.height;
            
            // 変換実行
            const canvasX = (screenX - canvasRect.left) * scaleX;
            const canvasY = (screenY - canvasRect.top) * scaleY;
            
            // 境界クランプ
            const clampedX = Math.max(0, Math.min(this.canvasWidth, canvasX));
            const clampedY = Math.max(0, Math.min(this.canvasHeight, canvasY));
            
            return { x: clampedX, y: clampedY };
            
        } catch (error) {
            ErrorManager.showError('coordinate-convert', 
                `座標変換エラー: ${error.message}`, 
                { screenX, screenY, canvasRect }
            );
            return { x: 0, y: 0 };
        }
    }
    
    /**
     * 📐 キャンバス座標 → PixiJS座標変換
     */
    canvasToPixi(canvasX, canvasY, pixiApp) {
        try {
            if (!pixiApp?.stage) {
                throw new Error('有効なPixiJSアプリが必要です');
            }
            
            const globalPoint = new PIXI.Point(canvasX, canvasY);
            const localPoint = pixiApp.stage.toLocal(globalPoint);
            
            return { x: localPoint.x, y: localPoint.y };
            
        } catch (error) {
            ErrorManager.showError('coordinate-pixi', 
                `PixiJS座標変換エラー: ${error.message}`, 
                { canvasX, canvasY }
            );
            return { x: canvasX, y: canvasY }; // フォールバック
        }
    }
    
    /**
     * 📐 スクリーン座標 → PixiJS座標変換（直接変換）
     */
    screenToPixi(screenX, screenY, canvasRect, pixiApp) {
        const canvasCoords = this.screenToCanvas(screenX, screenY, canvasRect);
        return this.canvasToPixi(canvasCoords.x, canvasCoords.y, pixiApp);
    }
    
    /**
     * 📐 PointerEvent → 座標情報抽出（統一処理）
     */
    extractPointerCoordinates(event, canvasRect, pixiApp = null) {
        try {
            const originalEvent = event.data?.originalEvent || event.originalEvent || event;
            
            const screenX = originalEvent.clientX || originalEvent.pageX || 0;
            const screenY = originalEvent.clientY || originalEvent.pageY || 0;
            
            const canvasCoords = this.screenToCanvas(screenX, screenY, canvasRect);
            
            const result = {
                screen: { x: screenX, y: screenY },
                canvas: canvasCoords,
                pressure: event.pressure || originalEvent.pressure || 0.5
            };
            
            if (pixiApp) {
                result.pixi = this.canvasToPixi(canvasCoords.x, canvasCoords.y, pixiApp);
            }
            
            return result;
            
        } catch (error) {
            ErrorManager.showError('coordinate-extract', 
                `座標抽出エラー: ${error.message}`, 
                { event: event.type }
            );
            return {
                screen: { x: 0, y: 0 },
                canvas: { x: 0, y: 0 },
                pressure: 0.5
            };
        }
    }
    
    /**
     * 📐 距離計算
     */
    calculateDistance(point1, point2) {
        const dx = point2.x - point1.x;
        const dy = point2.y - point1.y;
        return Math.sqrt(dx * dx + dy * dy);
    }
    
    /**
     * 📐 点が範囲内かチェック
     */
    isPointInBounds(point, bounds) {
        return point.x >= bounds.left && 
               point.x <= bounds.right && 
               point.y >= bounds.top && 
               point.y <= bounds.bottom;
    }
    
    /**
     * 📐 座標系設定更新（キャンバスリサイズ時）
     */
    updateCanvasSize(width, height) {
        this.canvasWidth = width;
        this.canvasHeight = height;
        
        console.log(`📐 座標系更新: ${width}x${height}`);
    }
    
    /**
     * 📐 座標管理状態取得（診断用）
     */
    getCoordinateState() {
        return {
            canvasWidth: this.canvasWidth,
            canvasHeight: this.canvasHeight,
            config: this.canvasConfig
        };
    }
}

// グローバル登録
if (typeof window !== 'undefined') {
    window.CoordinateManager = CoordinateManager;
    console.log('📐 CoordinateManager グローバル登録完了');
}
```

```javascript
// js/app-core.js（リファクタリング版・500行以下）
/**
 * 🎨 AppCoreシステム（分割最適化版）
 * 🎯 AI_WORK_SCOPE: PixiJSアプリケーション基盤・システム統合・初期化制御
 * 🎯 DEPENDENCIES: 統一システム4種、BoundaryManager、CoordinateManager
 * 🎯 SPLIT_RESULT: 650行 → 350行（46%削減）
 */

class AppCore {
    constructor() {
        this.validateUnifiedSystems();
        this.initializeConfig();
        
        // 基本プロパティ
        this.app = null;
        this.drawingContainer = null;
        this.uiContainer = null;
        
        // 分離された管理システム
        this.boundaryManager = null;
        this.coordinateManager = null;
        this.toolSystem = null;
        this.uiController = null;
        
        // 状態管理
        this.isInitializing = false;
        this.initializationComplete = false;
        
        console.log('🎨 AppCore インスタンス作成完了（分割最適化版）');
    }
    
    /**
     * 統一システム依存性確認
     */
    validateUnifiedSystems() {
        const required = ['ConfigManager', 'ErrorManager', 'StateManager', 'EventBus'];
        const missing = required.filter(sys => !window[sys]);
        if (missing.length > 0) {
            throw new Error(`AppCore: 統一システム依存性エラー: ${missing.join(', ')}`);
        }
    }
    
    /**
     * 設定初期化
     */
    initializeConfig() {
        const canvasConfig = ConfigManager.getCanvasConfig();
        this.canvasWidth = canvasConfig.width;
        this.canvasHeight = canvasConfig.height;
        this.backgroundColor = canvasConfig.backgroundColor;
    }
    
    /**
     * アプリケーション初期化（分割最適化版）
     */
    async initialize() {
        try {
            console.log('🚀 AppCore 初期化開始（分割最適化版）...');
            this.isInitializing = true;
            
            // 基盤システム初期化
            await this.initializeBasicSystems();
            
            // 管理システム初期化
            await this.initializeManagers();
            
            // アプリケーション初期化
            await this.initializeApplication();
            
            // 完了処理
            this.completeInitialization();
            
        } catch (error) {
            await this.handleInitializationError(error);
        }
    }
    
    /**
     * 基盤システム初期化
     */
    async initializeBasicSystems() {
        // DOM確認
        await this.verifyDOMElements();
        
        // 座標管理システム初期化
        this.coordinateManager = new CoordinateManager();
        
        // PixiJSアプリケーション初期化
        await this.initializePixiApp();
        
        // コンテナ初期化
        this.initializeContainers();
        
        console.log('✅ 基盤システム初期化完了');
    }
    
    /**
     * 管理システム初期化
     */
    async initializeManagers() {
        // 境界管理システム初期化
        this.boundaryManager = new BoundaryManager();
        this.boundaryManager.initialize(this.app.view, this.coordinateManager);
        
        // ツールシステム初期化
        this.toolSystem = new DrawingToolSystem(this);
        
        console.log('✅ 管理システム初期化完了');
    }
    
    /**
     * アプリケーション初期化
     */
    async initializeApplication() {
        // UI制御初期化
        await this.initializeUI();
        
        // イベントリスナー設定
        this.setupEventListeners();
        
        // 初期化検証
        this.verifyInitialization();
        
        console.log('✅ アプリケーション初期化完了');
    }
    
    /**
     * DOM要素確認
     */
    async verifyDOMElements() {
        const canvasElement = document.getElementById('drawing-canvas');
        if (!canvasElement) {
            throw new Error('drawing-canvas 要素が見つかりません');
        }
        
        // キャンバス要素クリア
        while (canvasElement.firstChild) {
            canvasElement.removeChild(canvasElement.firstChild);
        }
    }
    
    /**
     * PixiJSアプリケーション初期化
     */
    async initializePixiApp() {
        const canvasConfig = ConfigManager.getCanvasConfig();
        const pixiConfig = ConfigManager.getPixiConfig();
        
        this.app = new PIXI.Application({
            width: canvasConfig.width,
            height: canvasConfig.height,
            backgroundColor: canvasConfig.backgroundColor,
            antialias: pixiConfig.antialias,
            resolution: pixiConfig.resolution
        });
        
        const canvasElement = document.getElementById('drawing-canvas');
        canvasElement.appendChild(this.app.view);
        
        this.app.view.style.cursor = pixiConfig.cursor;
    }
    
    /**
     * コンテナ初期化
     */
    initializeContainers() {
        this.drawingContainer = new PIXI.Container();
        this.drawingContainer.name = 'drawing-layer';
        this.app.stage.addChild(this.drawingContainer);
        
        this.uiContainer = new PIXI.Container();
        this.uiContainer.name = 'ui-layer';
        this.app.stage.addChild(this.uiContainer);
        
        // インタラクティブ設定
        const canvasConfig = ConfigManager.getCanvasConfig();
        this.app.stage.interactive = true;
        this.app.stage.hitArea = new PIXI.Rectangle(0, 0, canvasConfig.width, canvasConfig.height);
    }
    
    /**
     * UI制御初期化
     */
    async initializeUI() {
        try {
            this.uiController = new UIController(this.toolSystem);
            this.uiController.initialize();
        } catch (error) {
            ErrorManager.showError('warning', `UI制御初期化失敗: ${error.message}`);
            this.uiController = new MinimalUIController(this.toolSystem);
        }
    }
    
    /**
     * イベントリスナー設定（分割最適化版）
     */
    setupEventListeners() {
        // PixiJS描画イベント
        this.app.stage.on('pointerdown', this.handlePointerDown.bind(this));
        this.app.stage.on('pointermove', this.handlePointerMove.bind(this));
        this.app.stage.on('pointerup', this.handlePointerUp.bind(this));
        
        // 境界越えイベント（BoundaryManager統合）
        EventBus.on('boundary.cross.in', this.handleBoundaryCrossIn.bind(this));
        
        // ウィンドウイベント
        window.addEventListener('resize', this.handleResize.bind(this));
        
        // ツールイベント
        EventBus.on('tool.changed', this.handleToolChanged.bind(this));
        
        console.log('✅ イベントリスナー設定完了（分割最適化版）');
    }
    
    /**
     * 境界越え描画開始処理（新規追加）
     */
    handleBoundaryCrossIn(data) {
        if (!this.toolSystem) return;
        
        try {
            // ツールシステムに境界越え描画開始を通知
            this.toolSystem.startBoundaryDrawing(data.position.x, data.position.y, data.pressure);
            
            console.log(`🎯 境界越え描画開始: (${data.position.x.toFixed(1)}, ${data.position.y.toFixed(1)})`);
            
        } catch (error) {
            ErrorManager.showError('boundary-drawing', 
                `境界越え描画エラー: ${error.message}`, 
                data
            );
        }
    }
    
    /**
     * ポインターダウンハンドラー（座標管理統合版）
     */
    handlePointerDown(event) {
        if (!this.toolSystem || !this.coordinateManager) return;
        
        const coords = this.coordinateManager.extractPointerCoordinates(
            event, 
            this.app.view.getBoundingClientRect(), 
            this.app
        );
        
        this.toolSystem.startDrawing(coords.canvas.x, coords.canvas.y, coords.pressure);
        
        EventBus.safeEmit('drawing.started', {
            position: coords.canvas,
            pressure: coords.pressure,
            tool: this.toolSystem.currentTool
        });
    }
    
    /**
     * ポインター移動ハンドラー（座標管理統合版）
     */
    handlePointerMove(event) {
        if (!this.coordinateManager) return;
        
        const coords = this.coordinateManager.extractPointerCoordinates(
            event, 
            this.app.view.getBoundingClientRect()
        );
        
        // 座標表示更新
        const coordinatesElement = document.getElementById('coordinates');
        if (coordinatesElement) {
            coordinatesElement.textContent = `x: ${Math.round(coords.canvas.x)}, y: ${Math.round(coords.canvas.y)}`;
        }
        
        if (this.toolSystem) {
            this.toolSystem.continueDrawing(coords.canvas.x, coords.canvas.y);
        }
    }
    
    /**
     * ポインターアップハンドラー
     */
    handlePointerUp(event) {
        if (!this.toolSystem) return;
        
        this.toolSystem.stopDrawing();
        
        EventBus.safeEmit('drawing.ended');
    }
    
    /**
     * リサイズハンドラー
     */
    handleResize() {
        if (!this.app) return;
        console.log('🔄 ウィンドウリサイズ検出');
        EventBus.safeEmit('window.resized');
    }
    
    /**
     * ツール変更ハンドラー
     */
    handleToolChanged(data) {
        console.log(`🔧 ツール変更: ${data.previousTool} → ${data.tool}`);
    }
    
    /**
     * 初期化完了処理
     */
    completeInitialization() {
        this.isInitializing = false;
        this.initializationComplete = true;
        
        EventBus.safeEmit('appCore.initialized', {
            success: true,
            components: this.getInitializationStats()
        });
        
        console.log('✅ AppCore 初期化完了（分割最適化版）');
    }
    
    /**
     * 初期化エラー処理
     */
    async handleInitializationError(error) {
        console.error('💀 AppCore初期化エラー:', error);
        
        ErrorManager.showError('error', error.message, {
            additionalInfo: 'AppCore初期化失敗',
            showReload: true
        });
        
        this.initializationFailed = true;
        await this.initializeFallbackMode(error);
    }
    
    /**
     * 初期化検証
     */
    verifyInitialization() {
        const verification = {
            pixiApp: !!this.app,
            drawingContainer: !!this.drawingContainer,
            toolSystem: !!this.toolSystem,
            boundaryManager: !!this.boundaryManager,
            coordinateManager: !!this.coordinateManager,
            uiController: !!this.uiController
        };
        
        const passCount = Object.values(verification).filter(Boolean).length;
        const totalCount = Object.keys(verification).length;
        
        console.log(`✅ 初期化検証: ${passCount}/${totalCount} (${(passCount/totalCount*100).toFixed(1)}%)`);
        
        if (passCount < totalCount) {
            const failed = Object.entries(verification)
                .filter(([key, value]) => !value)
                .map(([key, value]) => key);
            
            ErrorManager.showError('warning', `初期化未完了: ${failed.join(', ')}`);
        }
    }
    
    /**
     * 初期化統計取得
     */
    getInitializationStats() {
        return {
            pixiApp: !!this.app,
            drawingContainer: !!this.drawingContainer,
            toolSystem: !!this.toolSystem,
            boundaryManager: !!this.boundaryManager,
            coordinateManager: !!this.coordinateManager,
            uiController: !!this.uiController,
            initializationComplete: this.initializationComplete
        };
    }
    
    /**
     * フォールバックモード初期化
     */
    async initializeFallbackMode(error) {
        console.log('🛡️ フォールバックモード初期化中...');
        
        try {
            if (!this.app) {
                const fallbackConfig = {
                    width: ConfigManager.get('canvas.width') || 400,
                    height: ConfigManager.get('canvas.height') || 400,
                    backgroundColor: ConfigManager.get('canvas.backgroundColor') || 0xf0e0d6
                };
                
                this.app = new PIXI.Application(fallbackConfig);
                document.getElementById('drawing-canvas').appendChild(this.app.view);
            }
            
            if (!this.drawingContainer) {
                this.initializeContainers();
            }
            
            this.toolSystem = new SimpleFallbackToolSystem(this);
            
            ErrorManager.showError('recovery', '基本描画機能は利用可能です');
            
        } catch (fallbackError) {
            ErrorManager.showCriticalError(error.message, {
                additionalInfo: fallbackError.message
            });
        }
    }
    
    /**
     * キャンバスリサイズ（座標管理統合版）
     */
    resize(newWidth, newHeight, centerContent = false) {
        if (!this.app) return;
        
        const oldWidth = this.canvasWidth;
        const oldHeight = this.canvasHeight;
        
        // ConfigManager経由での妥当性確認
        const canvasConfig = ConfigManager.getCanvasConfig();
        const validWidth = Math.max(canvasConfig.minWidth, Math.min(canvasConfig.maxWidth, newWidth));
        const validHeight = Math.max(canvasConfig.minHeight, Math.min(canvasConfig.maxHeight, newHeight));
        
        this.canvasWidth = validWidth;
        this.canvasHeight = validHeight;
        
        // 座標管理システム更新
        if (this.coordinateManager) {
            this.coordinateManager.updateCanvasSize(validWidth, validHeight);
        }
        
        // アプリケーションリサイズ
        this.app.renderer.resize(validWidth, validHeight);
        this.app.stage.hitArea = new PIXI.Rectangle(0, 0, validWidth, validHeight);
        
        EventBus.safeEmit('canvas.resized', {
            width: validWidth,
            height: validHeight,
            previousWidth: oldWidth,
            previousHeight: oldHeight
        });
        
        console.log(`📐 キャンバスリサイズ: ${validWidth}x${validHeight}`);
    }
    
    /**
     * システム破棄
     */
    destroy() {
        if (this.boundaryManager) {
            this.boundaryManager.destroy();
        }
        
        if (this.app) {
            this.app.destroy(true);
        }
        
        console.log('🎨 AppCore 破棄完了');
    }
}
```

```javascript
// tools/pen-tool.js（境界対応拡張版）
/**
 * 🖊️ ペンツール（境界対応統合版）
 * 🎯 BOUNDARY_SUPPORT: 境界越え描画開始・継続描画統合
 * 🎯 DEPENDENCIES: ConfigManager, ErrorManager, EventBus, BoundaryManager
 */

class PenTool {
    constructor(toolManager) {
        this.toolManager = toolManager;
        this.name = 'pen';
        this.isActive = false;
        this.isDrawing = false;
        this.currentStroke = null;
        
        // ConfigManager統合設定
        this.config = ConfigManager.getDrawingConfig('pen');
        this.boundaryConfig = ConfigManager.get('canvas.boundary');
        
        // 境界対応プロパティ
        this.supportsBoundaryDrawing = true;
        this.fromBoundary = false;
        
        console.log('🖊️ PenTool 初期化完了（境界対応統合版）');
    }
    
    /**
     * 🎯 境界越え描画開始処理（新規実装）
     */
    handleBoundaryCrossIn(x, y, pointerEvent) {
        if (!this.isActive) return;
        
        try {
            console.log(`🎯 ペンツール境界越え描画開始: (${x.toFixed(1)}, ${y.toFixed(1)})`);
            
            const pressure = pointerEvent.pressure || 0.5;
            
            // 境界越え専用描画開始
            this.startBoundaryDrawing(x, y, pressure);
            
            // EventBus通知
            EventBus.safeEmit('pen.boundary.drawing.started', {
                position: { x, y },
                pressure,
                fromOutside: true,
                strokeId: this.currentStroke?.id
            });
            
        } catch (error) {
            ErrorManager.showError('boundary-drawing', 
                `ペン境界越え描画エラー: ${error.message}`, 
                { tool: this.name, position: { x, y } }
            );
        }
    }
    
    /**
     * 🎯 境界越え描画開始（通常startDrawingをバイパス）
     */
    startBoundaryDrawing(x, y, pressure) {
        // 境界チェックをバイパスして直接描画開始
        this.isDrawing = true;
        this.fromBoundary = true;
        
        this.currentStroke = {
            id: this.generateStrokeId(),
            tool: this.name,
            points: [{ x, y, pressure, timestamp: Date.now() }],
            fromBoundary: true,
            graphics: new PIXI.Graphics()
        };
        
        // 描画グラフィクス初期化
        this.initializeStrokeGraphics();
        
        // 最初のポイント描画
        this.addStrokePoint(x, y, pressure);
        
        // ツールマネージャーのdrawingContainer追加
        if (this.toolManager?.appCore?.drawingContainer) {
            this.toolManager.appCore.drawingContainer.addChild(this.currentStroke.graphics);
        }
        
        console.log(`🖊️ 境界越え描画セッション開始: ${this.currentStroke.id}`);
    }
    
    /**
     * 通常描画開始（既存処理・境界判定あり）
     */
    startDrawing(x, y, pressure = 0.5) {
        if (this.isDrawing) return;
        
        // キャンバス境界内チェック（通常描画時のみ）
        if (!this.isPointInCanvas(x, y)) {
            console.log('⚠️ キャンバス境界外描画開始は無視（通常モード）');
            return;
        }
        
        this.isDrawing = true;
        this.fromBoundary = false;
        
        this.currentStroke = {
            id: this.generateStrokeId(),
            tool: this.name,
            points: [{ x, y, pressure, timestamp: Date.now() }],
            fromBoundary: false,
            graphics: new PIXI.Graphics()
        };
        
        this.initializeStrokeGraphics();
        this.addStrokePoint(x, y, pressure);
        
        if (this.toolManager?.appCore?.drawingContainer) {
            this.toolManager.appCore.drawingContainer.addChild(this.currentStroke.graphics);
        }
        
        console.log(`🖊️ 通常描画開始: (${x.toFixed(1)}, ${y.toFixed(1)})`);
    }
    
    /**
     * 描画継続（境界・通常共通処理）
     */
    continueDrawing(x, y, pressure = 0.5) {
        if (!this.isDrawing || !this.currentStroke) return;
        
        const lastPoint = this.getLastPoint();
        if (!lastPoint) return;
        
        // 最小距離チェック
        const distance = this.calculateDistance(lastPoint, { x, y });
        if (distance < this.config.minDistance) return;
        
        // ポイント追加
        this.addStrokePoint(x, y, pressure);
        
        // 線描画
        this.drawLineBetweenPoints(lastPoint, { x, y, pressure });
    }
    
    /**
     * 描画終了
     */
    stopDrawing() {
        if (!this.isDrawing || !this.currentStroke) return;
        
        // ストローク完了
        this.currentStroke.isComplete = true;
        this.currentStroke.endTime = Date.now();
        
        // EventBus通知
        EventBus.safeEmit('pen.drawing.completed', {
            strokeId: this.currentStroke.id,
            pointCount: this.currentStroke.points.length,
            fromBoundary: this.fromBoundary,
            duration: this.currentStroke.endTime - this.currentStroke.points[0].timestamp
        });
        
        console.log(`🖊️ 描画終了: ${this.currentStroke.points.length}ポイント ${this.fromBoundary ? '(境界越え)' : '(通常)'}`);
        
        // 状態リセット
        this.isDrawing = false;
        this.fromBoundary = false;
        this.currentStroke = null;
    }
    
    /**
     * ストロークグラフィクス初期化
     */
    initializeStrokeGraphics() {
        if (!this.currentStroke) return;
        
        const graphics = this.currentStroke.graphics;
        graphics.clear();
        
        // ペン設定適用
        graphics.lineStyle(this.config.defaultSize, this.config.defaultColor, this.config.defaultOpacity);
    }
    
    /**
     * ストロークポイント追加
     */
    addStrokePoint(x, y, pressure) {
        if (!this.currentStroke) return;
        
        this.currentStroke.points.push({
            x, y, pressure,
            timestamp: Date.now()
        });
    }
    
    /**
     * ポイント間線描画
     */
    drawLineBetweenPoints(startPoint, endPoint) {
        if (!this.currentStroke?.graphics) return;
        
        const graphics = this.currentStroke.graphics;
        
        // スムージング適用
        if (this.config.smoothingEnabled) {
            this.drawSmoothLine(graphics, startPoint, endPoint);
        } else {
            graphics.moveTo(startPoint.x, startPoint.y);
            graphics.lineTo(endPoint.x, endPoint.y);
        }
    }
    
    /**
     * スムース線描画
     */
    drawSmoothLine(graphics, start, end) {
        const midX = (start.x + end.x) / 2;
        const midY = (start.y + end.y) / 2;
        
        graphics.moveTo(start.x, start.y);
        graphics.quadraticCurveTo(start.x, start.y, midX, midY);
        graphics.quadraticCurveTo(midX, midY, end.x, end.y);
    }
    
    /**
     * ユーティリティメソッド
     */
    generateStrokeId() {
        return `pen-stroke-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
    }
    
    getLastPoint() {
        return this.currentStroke?.points[this.currentStroke.points.length - 1] || null;
    }
    
    calculateDistance(point1, point2) {
        const dx = point2.x - point1.x;
        const dy = point2.y - point1.y;
        return Math.sqrt(dx * dx + dy * dy);
    }
    
    isPointInCanvas(x, y) {
        const canvasConfig = ConfigManager.getCanvasConfig();
        return x >= 0 && x <= canvasConfig.width && y >= 0 && y <= canvasConfig.height;
    }
    
    /**
     * ツール状態取得（診断用）
     */
    getToolState() {
        return {
            name: this.name,
            isActive: this.isActive,
            isDrawing: this.isDrawing,
            supportsBoundaryDrawing: this.supportsBoundaryDrawing,
            currentStroke: this.currentStroke ? {
                id: this.currentStroke.id,
                pointCount: this.currentStroke.points.length,
                fromBoundary: this.currentStroke.fromBoundary
            } : null,
            config: this.config
        };
    }
}
```

```javascript
    });

    if (!this.coordinateManager) return;

        // PixiJS座標系変換
        const pixiCoords = this.coordinateManager.canvasToPixi(
            data.position.x, 
            data.position.y, 
            this.app
        );

                pixiCoords.y,

                    pressure: data.pressure,
                    pointerId: data.pointerId,
                    originalEvent: data.originalEvent
                }
            );

    } catch (error) {
        ErrorManager.showError('pixi-boundary', 
            `PixiJS境界越え処理エラー: ${error.message}`,
```

### パフォーマンス指標・監視システム
```javascript
// js/utils/boundary-performance-monitor.js（新規作成）
/**
 * 📊 境界描画パフォーマンス監視
 * 🎯 PERFORMANCE_TARGETS: 境界検出5ms以内・描画開始16ms以内・60FPS維持
 */

class BoundaryPerformanceMonitor {
    constructor() {
        this.metrics = {
            boundaryDetection: { times: [], average: 0, max: 0, min: Infinity },
            crossInProcessing: { times: [], average: 0, max: 0, min: Infinity },
            drawingStart: { times: [], average: 0, max: 0, min: Infinity },
            coordinateConversion: { times: [], average: 0, max: 0, min: Infinity },
            memoryUsage: { samples: [], average: 0, peak: 0 }
        };
        
        this.targets = {
            boundaryDetection: 5.0,      // 5ms以内
            crossInProcessing: 8.0,      // 8ms以内
            drawingStart: 16.0,          // 16ms以内（60FPS維持）
            coordinateConversion: 2.0,   // 2ms以内
            memoryUsage: 10 * 1024 * 1024 // 10MB以内
        };
        
        this.isMonitoring = false;
        this.startTime = null;
    }
    
    /**
     * 監視開始
     */
    startMonitoring() {
        if (this.isMonitoring) return;
        
        this.isMonitoring = true;
        
        // EventBusリスナー設定
        EventBus.on('boundary.outside.start', () => {
            this.startTimer('boundaryDetection');
        });
        
        EventBus.on('boundary.cross.in', () => {
            this.endTimer('boundaryDetection');
            this.startTimer('crossInProcessing');
        });
        
        EventBus.on('pen.boundary.drawing.started', () => {
            this.endTimer('crossInProcessing');
            this.endTimer('drawingStart');
        });
        
        // メモリ監視
        this.startMemoryMonitoring();
        
        console.log('📊 境界描画パフォーマンス監視開始');
    }
    
    /**
     * タイマー開始
     */
    startTimer(metric) {
        this.startTimes = this.startTimes || {};
        this.startTimes[metric] = performance.now();
    }
    
    /**
     * タイマー終了・記録
     */
    endTimer(metric) {
        if (!this.startTimes || !this.startTimes[metric]) return;
        
        const duration = performance.now() - this.startTimes[metric];
        this.recordMetric(metric, duration);
        
        delete this.startTimes[metric];
    }
    
    /**
     * メトリクス記録
     */
    recordMetric(metric, value) {
        const metricData = this.metrics[metric];
        if (!metricData) return;
        
        metricData.times.push(value);
        metricData.max = Math.max(metricData.max, value);
        metricData.min = Math.min(metricData.min, value);
        
        // 最新100件の平均を計算
        const recentTimes = metricData.times.slice(-100);
        metricData.average = recentTimes.reduce((sum, time) => sum + time, 0) / recentTimes.length;
        
        // 目標値チェック
        const target = this.targets[metric];
        if (value > target) {
            console.warn(`⚠️ パフォーマンス目標超過: ${metric} ${value.toFixed(2)}ms (目標: ${target}ms)`);
        }
    }
    
    /**
     * メモリ監視開始
     */
    startMemoryMonitoring() {
        const sampleMemory = () => {
            if (!this.isMonitoring) return;
            
            if (performance.memory) {
                const used = performance.memory.usedJSHeapSize;
                this.metrics.memoryUsage.samples.push(used);
                this.metrics.memoryUsage.peak = Math.max(this.metrics.memoryUsage.peak, used);
                
                // 最新20件の平均
                const recentSamples = this.metrics.memoryUsage.samples.slice(-20);
                this.metrics.memoryUsage.average = recentSamples.reduce((sum, sample) => sum + sample, 0) / recentSamples.length;
            }
            
            setTimeout(sampleMemory, 1000); // 1秒間隔
        };
        
        sampleMemory();
    }
    
    /**
     * パフォーマンスレポート取得
     */
    getPerformanceReport() {
        const report = {
            timestamp: new Date().toISOString(),
            targets: this.targets,
            metrics: {},
            compliance: {},
            overall: { compliant: true, issues: [] }
        };
        
        // 各メトリクスの報告
        Object.keys(this.metrics).forEach(key => {
            const metric = this.metrics[key];
            const target = this.targets[key];
            
            if (key === 'memoryUsage') {
                report.metrics[key] = {
                    average: (metric.average / 1024 / 1024).toFixed(2) + 'MB',
                    peak: (metric.peak / 1024 / 1024).toFixed(2) + 'MB',
                    samples: metric.samples.length
                };
                
                report.compliance[key] = metric.average <= target;
                if (metric.average > target) {
                    report.overall.compliant = false;
                    report.overall.issues.push(`メモリ使用量超過: ${(metric.average / 1024 / 1024).toFixed(2)}MB`);
                }
            } else {
                report.metrics[key] = {
                    average: metric.average.toFixed(2) + 'ms',
                    min: metric.min.toFixed(2) + 'ms',
                    max: metric.max.toFixed(2) + 'ms',
                    samples: metric.times.length
                };
                
                report.compliance[key] = metric.average <= target;
                if (metric.average > target) {
                    report.overall.compliant = false;
                    report.overall.issues.push(`${key}処理時間超過: ${metric.average.toFixed(2)}ms`);
                }
            }
        });
        
        return report;
    }
    
    /**
     * 監視停止
     */
    stopMonitoring() {
        this.isMonitoring = false;
        console.log('📊 境界描画パフォーマンス監視停止');
    }
}

// グローバル登録
if (typeof window !== 'undefined') {
    window.BoundaryPerformanceMonitor = BoundaryPerformanceMonitor;
    
    // 診断関数
    window.getBoundaryPerformance = function() {
        if (!window.boundaryPerformanceMonitor) {
            window.boundaryPerformanceMonitor = new BoundaryPerformanceMonitor();
            window.boundaryPerformanceMonitor.startMonitoring();
            return "パフォーマンス監視を開始しました。しばらく境界描画をテストしてから再度実行してください。";
        }
        
        return window.boundaryPerformanceMonitor.getPerformanceReport();
    };
    
    console.log('📊 BoundaryPerformanceMonitor グローバル登録完了');
}
```
