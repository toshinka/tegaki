## 🛠️ 具体的改修内容

### TegakiApplication 拡張API
```javascript
getV8FeatureStatus() {
    return {
        pixiVersion: window.PIXI?.VERSION || 'not loaded',
        webgpuEnabled: this.pixiApp?.renderer?.type === 'webgpu',
        rendererType: this.pixiApp?.renderer?.type || 'unknown',
        containerSupport: !!window.PIXI?.Container,
        graphicsSupport: !!window.PIXI?.Graphics,
        systemReady: this.isReady
    };
}

getDebugInfo() {
    const v8Status = this.getV8FeatureStatus();
    return {
        ...v8Status,
        initializationSteps: this.initSteps || [],
        currentTool: this.toolManager?.getCurrentTool?.()?.constructor?.name,
        canvasSize: this.canvasManager?.getCurrentSize?.(),
        managerCount: Object.keys(this.managers || {}).length
    };
}
```

### index.html デバッグ修正
```javascript
window.debugTegakiV8 = function() {
    const app = window.Tegaki?.TegakiApplicationInstance;
    if (!app) {
        console.warn('⚠️ TegakiApplication未初期化');
        return null;
    }
    
    // 安全なAPI呼び出し
    const debugInfo = app.getDebugInfo?.() || {};
    const v8Status = app.getV8FeatureStatus?.() || {};
    
    console.log('🚀 v8デバッグ情報:', debugInfo);
    console.log('🎯 v8機能状況:', v8Status);
    
    return { debugInfo, v8Status };
};
```
