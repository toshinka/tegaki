```javascript
/**
 * 統一座標変換システム - 座標ズレ完全解決版
 */
class CoordinateManager {
    constructor() {
        this.canvasManager = null;
        this.dpr = window.devicePixelRatio || 1;
        this.transformMatrix = new PIXI.Matrix();
    }
    
    // ✅ 統一座標変換メソッド
    clientToWorld(clientX, clientY) {
        // Step 1: DOM座標 → Canvas座標
        const canvasPoint = this.clientToCanvas(clientX, clientY);
        
        // Step 2: Canvas座標 → World座標 (Container変形対応)
        const worldPoint = this.canvasToWorld(canvasPoint.x, canvasPoint.y);
        
        return worldPoint;
    }
    
    // DPR・Container変形・CSS変形すべてに対応
    clientToCanvas(clientX, clientY) {
        const canvas = this.canvasManager.getCanvasElement();
        const rect = canvas.getBoundingClientRect();
        
        // CSS表示サイズベースの正規化座標
        const normalizedX = (clientX - rect.left) / rect.width;
        const normalizedY = (clientY - rect.top) / rect.height;
        
        // Canvas論理サイズに変換
        const canvasX = normalizedX * canvas.width / this.dpr;
        const canvasY = normalizedY * canvas.height / this.dpr;
        
        return { x: canvasX, y: canvasY };
    }
    
    canvasToWorld(canvasX, canvasY) {
        const drawContainer = this.canvasManager.getDrawContainer();
        
        // Container変形を考慮した座標変換
        const localPoint = drawContainer.toLocal({ x: canvasX, y: canvasY });
        
        return { x: localPoint.x, y: localPoint.y };
    }
}
```

```javascript
/**

class CanvasManager {
    // ✅ 精密座標変換メソッド
    getCanvasRect() {
        return this.pixiApp.canvas.getBoundingClientRect();
    }
    
    getCanvasLogicalSize() {
        return {
            width: this.pixiApp.canvas.width / this.effectiveDPR,
            height: this.pixiApp.canvas.height / this.effectiveDPR
        };
    }
    
    // 座標変換係数取得
    getTransformCoefficients() {
        const rect = this.getCanvasRect();
        const logical = this.getCanvasLogicalSize();
        
        return {
            scaleX: logical.width / rect.width,
            scaleY: logical.height / rect.height,
            offsetX: rect.left,
            offsetY: rect.top
        };
    }
}
```

### 🎨 **Phase 2: Graphics管理システム再構築**

#### **2.1 Graphics階層分離の完全実装**
```javascript
/**
 * Graphics管理 - 描画消失問題完全解決版
 */
class CanvasManager {
    initializeGraphicsManagement() {
        // 階層構造作成
        this.graphicsLayers = {
            permanent: new PIXI.Container(),    // 確定描画
            temporary: new PIXI.Container(),    // 一時描画
            overlay: new PIXI.Container()       // UI重ね合わせ
        };
        
        // zIndex設定
        this.graphicsLayers.permanent.zIndex = 100;
        this.graphicsLayers.temporary.zIndex = 200;
        this.graphicsLayers.overlay.zIndex = 300;
        
        // DrawContainerに追加
        Object.values(this.graphicsLayers).forEach(layer => {
            this.v8DrawingContainer.addChild(layer);
        });
        
        // 一時Graphics作成
        this.temporaryGraphics = new PIXI.Graphics();
        this.graphicsLayers.temporary.addChild(this.temporaryGraphics);
    }
    
    // ✅ ストローク専用Graphics作成
    createStrokeGraphics(strokeId) {
        const graphics = new PIXI.Graphics();
        graphics.name = strokeId;
        graphics.zIndex = Date.now(); // 描画順序保証
        
        // 永続レイヤーに追加
        this.graphicsLayers.permanent.addChild(graphics);
        this.permanentGraphics.set(strokeId, graphics);
        
        return graphics;
    }
    
    // ✅ 一時Graphics専用クリア (全体に影響なし)
    clearTemporaryGraphics() {
        if (this.temporaryGraphics) {
            this.temporaryGraphics.clear();
        }
    }
}
```

#### **2.2 PenTool 非破壊描画システム**
```javascript
/**
 * PenTool 非破壊描画 - Graphics分離完全版
 */
class PenTool {
    startStroke(point) {
        // ❌ 削除: this.graphics.clear() // 全Graphics消去
        
        // ✅ 新規Graphics作成のみ
        this.currentStrokeGraphics = this.canvasManager.createStrokeGraphics();
        this.canvasManager.clearTemporaryGraphics(); // 一時のみクリア
        
        // 描画開始
        this.drawStrokePoint(point, true); // isFirst: true
    }
    
    addStrokePoint(point) {
        this.drawStrokePoint(point, false);
    }
    
    drawStrokePoint(point, isFirst = false) {
        // 一時Graphics: リアルタイム表示
        if (isFirst) {
            this.temporaryGraphics.moveTo(point.x, point.y);
        } else {
            this.temporaryGraphics.lineTo(point.x, point.y);
        }
        this.temporaryGraphics.stroke({
            width: this.strokeWidth,
            color: this.strokeColor,
            alpha: this.strokeOpacity
        });
        
        // 永続Graphics: 確定保存用
        if (isFirst) {
            this.currentStrokeGraphics.moveTo(point.x, point.y);
        } else {
            this.currentStrokeGraphics.lineTo(point.x, point.y);
        }
        this.currentStrokeGraphics.stroke({
            width: this.strokeWidth,
            color: this.strokeColor,
            alpha: this.strokeOpacity
        });
    }
    
    endStroke(point) {
        // 最終描画
        if (point) {
            this.addStrokePoint(point);
        }
        
        // 一時Graphics クリア (永続Graphicsは保持)
        this.canvasManager.clearTemporaryGraphics();
        
        // TPF保存
        this.saveToRecord();
        
        // 状態リセット
        this.resetStrokeState();
    }
}
```

```javascript
/**
 * AppCore 初期化統制 - Manager順序保証版
 */
class AppCore {
    async initializeV8Managers() {
        // Step 1: 基盤Manager初期化
        await this.initializeCoreManagers();
        
        // Step 2: 座標・描画Manager初期化
        await this.initializeDrawingManagers();
        
        // Step 3: Tool・UI Manager初期化
        await this.initializeToolManagers();
        
        // Step 4: 統合検証
        await this.verifySystemIntegration();
    }
    
    async initializeCoreManagers() {
        // 1. CanvasManager (最優先)
        this.canvasManager = new CanvasManager();
        await this.canvasManager.initializeV8Application(this.pixiApp);
        this.verifyManagerReady(this.canvasManager, 'CanvasManager');
        
        // 2. ErrorManager, EventBus (既存)
        this.registerExistingManagers();
    }
    
    async initializeDrawingManagers() {
        // 3. CoordinateManager
        this.coordinateManager = new CoordinateManager();
        await this.coordinateManager.setCanvasManager(this.canvasManager);
        this.verifyManagerReady(this.coordinateManager, 'CoordinateManager');
        
        // 4. RecordManager
        this.recordManager = new RecordManager();
        await this.recordManager.initialize();
        this.verifyManagerReady(this.recordManager, 'RecordManager');
    }
    
    async initializeToolManagers() {
        // 5. ToolManager (全依存Manager準備完了後)
        this.toolManager = new ToolManager();
        await this.toolManager.setManagers({
            canvas: this.canvasManager,
            coordinate: this.coordinateManager,
            record: this.recordManager,
            eventbus: this.eventBus,
            error: this.errorManager
        });
        
        // 6. Tool作成
        await this.toolManager.initializeV8Tools();
        this.verifyManagerReady(this.toolManager, 'ToolManager');
    }
    
    // ✅ Manager準備検証の強化
    verifyManagerReady(manager, managerName) {
        if (!manager.isReady()) {
            const debugInfo = manager.getDebugInfo ? manager.getDebugInfo() : {};
            throw new Error(`${managerName} initialization failed: ${JSON.stringify(debugInfo)}`);
        }
    }
}
```
