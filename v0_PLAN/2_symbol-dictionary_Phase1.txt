> **対象Phase**: Phase1.1ss5rev2（統一システム統合版）

## 📋 概要

このシンボル辞典は、統一システム（ConfigManager・ErrorManager・StateManager・EventBus）を含む、ふたば☆お絵描きツールの全クラス・関数・設定値を網羅した技術リファレンスです。統一システムによる重複排除・循環依存解決・AI協働開発の効率化を支援します。

```javascript
class EventBus {
    // 📊 主要API
    static emit(eventType, data)         // イベント発行
    static on(eventType, callback)       // イベントリスナー登録
    static off(eventType, callback)      // イベントリスナー削除
    static offAll(eventType)             // 特定タイプ全削除
    static clear()                       // 全イベントクリア
    static once(eventType, callback)     // 一度のみリスナー
    static getStats()                    // イベント統計取得
    static getListeners(eventType)       // リスナー一覧取得
    static debug()                       // デバッグ情報表示
    
    // 📡 標準イベントタイプ（EventBus.Events）
    BRUSH_SIZE_CHANGED: 'brushSizeChanged'         // ブラシサイズ変更
    BRUSH_COLOR_CHANGED: 'brushColorChanged'       // ブラシ色変更
    OPACITY_CHANGED: 'opacityChanged'              // 不透明度変更
    PRESSURE_CHANGED: 'pressureChanged'            // 筆圧変更
    SMOOTHING_CHANGED: 'smoothingChanged'          // 線補正変更
    TOOL_CHANGED: 'toolChanged'                    // ツール変更
    DRAWING_STARTED: 'drawingStarted'              // 描画開始
    DRAWING_CONTINUED: 'drawingContinued'          // 描画継続
    DRAWING_ENDED: 'drawingEnded'                  // 描画終了
    PATH_CREATED: 'pathCreated'                    // パス作成
    POPUP_OPENED: 'popupOpened'                    // ポップアップ表示
    POPUP_CLOSED: 'popupClosed'                    // ポップアップ非表示
    SLIDER_CHANGED: 'sliderChanged'                // スライダー変更
    PRESET_SELECTED: 'presetSelected'              // プリセット選択
    CANVAS_RESIZED: 'canvasResized'                // キャンバスリサイズ
    CANVAS_CLEARED: 'canvasCleared'                // キャンバスクリア
    INITIALIZATION_COMPLETED: 'initializationCompleted'  // 初期化完了
    ERROR_OCCURRED: 'errorOccurred'                // エラー発生
    PERFORMANCE_WARNING: 'performanceWarning'     // パフォーマンス警告
    
    // 📊 統計情報
    stats.emitted                        // 発行回数
    stats.listeners                      // リスナー数
    stats.types                          // イベントタイプ集合
}

// グローバルショートカット
```

```javascript
// 統一エラー処理 + EventBus連携
class DrawingToolSystem {
    startDrawing(x, y) {
        try {
            // 前提条件確認
            if (!this.appCore || !this.appCore.app) {
                throw new Error('AppCore未初期化');
            }
            
            // 描画処理
            this.currentPath = this.createPenPath(x, y);
            this.isDrawing = true;
            
            // 成功イベント発行
            EventBus.emit(EventBus.Events.DRAWING_STARTED, {
                tool: this.currentTool,
                position: { x, y },
                pathId: this.currentPath.id
            });
            
        } catch (error) {
            // 統一エラー処理
            ErrorManager.showError('error', error, {
                additionalInfo: `描画開始失敗: ツール=${this.currentTool}, 座標=(${x}, ${y})`
            });
            
            // エラーイベント発行
            EventBus.emit(EventBus.Events.ERROR_OCCURRED, {
                type: 'drawing_start_error',
                error,
                context: { tool: this.currentTool, position: { x, y } }
            });
            
            // 状態リセット
            this.isDrawing = false;
            this.currentPath = null;
        }
    }
}
```

```javascript
function resizeCanvasUnified(width, height, centerContent = false) {
    try {
        // 妥当性確認（ConfigManager）
        if (!ConfigManager.validate('canvas.width', width) || 
            !ConfigManager.validate('canvas.height', height)) {
            ErrorManager.showError('warning', `無効なサイズ: ${width}×${height}`);
            return;
        }

        ConfigManager.set('canvas.height', height);
        
        // リサイズ実行
        appCore.resize(width, height, centerContent);
        
        // イベント発行（EventBus）
        EventBus.emit(EventBus.Events.CANVAS_RESIZED, {
            width, height, centerContent,
            timestamp: Date.now()
        });
        
    } catch (error) {
        ErrorManager.showError('error', error, {
            additionalInfo: `リサイズ失敗: ${width}×${height}`
        });
    }
}
```
