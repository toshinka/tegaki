#### **STEP4: AppCore座標変換重複除去**
**対象ファイル**: `js/app-core.js`
**修正行数**: 349行目付近の`handlePixiBoundaryCross`メソッド

**修正前**:
```javascript
handlePixiBoundaryCross(data) {
    // ❌ 不要な座標変換（data.positionは既にキャンバス座標）
    const pixiCoords = this.coordinateManager.canvasToPixi(
        data.position.x, data.position.y, this.app
    );
}
```

**修正後**:
```javascript
handlePixiBoundaryCross(data) {
    if (!this.coordinateManager) return;
    
    try {
        // ✅ BoundaryManagerから来る座標は既にキャンバス座標なのでそのまま使用
        const canvasCoords = data.position; // 座標変換不要
        
        // アクティブツールに直接渡す
        const currentTool = this.toolManager?.getCurrentTool();
        const currentToolInstance = this.toolManager?.registeredTools.get(currentTool);
        
        if (currentToolInstance?.handleBoundaryCrossIn) {
            currentToolInstance.handleBoundaryCrossIn(
                canvasCoords.x, canvasCoords.y, data
            );
        }
        
        console.log(`🎯 境界越え処理完了: (${canvasCoords.x.toFixed(1)}, ${canvasCoords.y.toFixed(1)})`);
        
    } catch (error) {
        if (window.ErrorManager && typeof window.ErrorManager.showError === 'function') {
            window.ErrorManager.showError('error', 
                `境界越え処理エラー: ${error.message}`, 
                { additionalInfo: '境界越え処理', data }
            );
        }
    }
}
```

**修正後**:
```javascript
registerBasicTools() {
    try {
        // 実際のツールクラスを検索・登録
        if (window.PenTool) {
            const penTool = new window.PenTool(this);
            if (typeof penTool.initialize === 'function') {
                penTool.initialize();
            }
            this.registeredTools.set('pen', penTool);
            console.log('✅ PenTool登録完了');
        }
        
        if (window.EraserTool) {
            const eraserTool = new window.EraserTool(this);
            if (typeof eraserTool.initialize === 'function') {
                eraserTool.initialize();
            }
            this.registeredTools.set('eraser', eraserTool);
            console.log('✅ EraserTool登録完了');
        }
        
        // フォールバック：基本ツール名のみ登録
        const basicTools = this.configManager?.getAvailableTools() || ['pen', 'eraser'];
        basicTools.forEach(toolName => {
            if (!this.registeredTools.has(toolName)) {
                this.registeredTools.set(toolName, {
                    name: toolName,
                    type: 'basic',
                    settings: this.globalSettings?.[toolName] || {},
                    startDrawing: (x, y, pressure) => {
                        console.log(`🔧 ${toolName} 基本描画: (${x.toFixed(1)}, ${y.toFixed(1)})`);
                        this.handleDrawingFallback('start', x, y, pressure);
                    },
                    continueDrawing: (x, y, pressure) => {
                        this.handleDrawingFallback('continue', x, y, pressure);
                    },
                    stopDrawing: () => {
                        this.handleDrawingFallback('stop');
                    }
                });
            }
        });
        
    } catch (error) {
        console.warn('⚠️ 基本ツール登録で問題発生:', error.message);
    }
}
```

### **Phase 4: 境界越えメソッド実装（10分）**

#### **STEP6: PenTool境界越えメソッド確認・修正**
**対象ファイル**: `tools/pen-tool.js`
**新規メソッド**: `handleBoundaryCrossIn`

```javascript
/**
 * 境界越え描画開始（新規追加）
 * @param {number} x - キャンバス座標X
 * @param {number} y - キャンバス座標Y  
 * @param {Object} data - 境界越えデータ
 */
handleBoundaryCrossIn(x, y, data) {
    try {
        console.log(`🖊️ Pen: 境界越え描画開始 (${x.toFixed(1)}, ${y.toFixed(1)})`);
        
        // 既存の描画開始メソッドを呼び出し
        this.startDrawing(x, y, data.pressure || 0.5);
        
        // EventBus通知
        if (this.eventBus) {
            this.eventBus.emit('pen:boundary:cross:in', {
                position: { x, y },
                pressure: data.pressure,
                tool: this.name
            });
        }
        
    } catch (error) {
        this.safeError(`境界越え描画エラー: ${error.message}`, 'warning');
    }
}
```

/**
 * 境界越え消去開始（新規追加）
