## 5. Phase2レイヤーシステム設計

### 5.1 レイヤーアーキテクチャ（先取り）
```javascript
// レイヤー管理統一システム統合
class LayerManager {
    constructor() {
        this.config = ConfigManager.getLayerConfig();
        this.layers = new Map();
        this.activeLayer = null;
    }
    
    createLayer(options) {
        try {
            const layer = new Layer(options, this.config);
            this.layers.set(layer.id, layer);
            
            EventBus.safeEmit('layer.created', { 
                layerId: layer.id, 
                options 
            });
            
            return layer;
        } catch (error) {
            ErrorManager.showError('error', `レイヤー作成失敗: ${error.message}`);
        }
    }
    
    getLayerState() {
        return {
            total: this.layers.size,
            active: this.activeLayer?.id,
            visible: Array.from(this.layers.values())
                        .filter(layer => layer.visible)
                        .length
        };
    }
}
```

### 10.2 レイヤーUI統一システム統合
```javascript
// レイヤーUI統一システム活用設計
class LayerUI {
    constructor() {
        this.config = ConfigManager.getLayerConfig();
        this.setupEventListeners();
    }
    
    setupEventListeners() {
        EventBus.on(EventBus.LayerEvents.LAYER_CREATED, (data) => {
            this.addLayerToUI(data.layerId, data.options);
        });
        
        EventBus.on(EventBus.LayerEvents.LAYER_SELECTED, (data) => {
            this.updateActiveLayerUI(data.layerId);
        });
    }
    
    addLayerToUI(layerId, options) {
        try {
            // レイヤーUI要素作成・追加
            const layerElement = this.createLayerElement(layerId, options);
            this.layerContainer.appendChild(layerElement);
            
            // 統一システム連携
            EventBus.safeEmit('layerUI.added', { layerId, element: layerElement });
            
        } catch (error) {
            ErrorManager.showError('layer-render', 
                `レイヤーUI作成失敗: ${error.message}`, 
                { layerId, options }
            );
        }
    }
}
```
