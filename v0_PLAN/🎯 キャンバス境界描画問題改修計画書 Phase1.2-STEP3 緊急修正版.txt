### AppCore縮小版設計
```javascript
// app-core.js (縮小版 - 200行以下)
class AppCore {
    constructor() {
        this.app = null;
        this.width = 800;
        this.height = 600;
        this.isInitialized = false;
        
        // 境界システムは外部委譲
        this.boundaryManager = null;
        this.coordinateManager = null;
    }
    
    async initialize(options = {}) {
        // 基本PixiJS初期化のみ
        await this.createPixiApplication(options);
        this.setupBasicConfiguration();
        this.initializeDrawingStage();
        
        // 外部システム統合
        this.integrateBoundarySystem();
        this.integrateCoordinateSystem();
        
        this.isInitialized = true;
        return true;
    }
    
    integrateBoundarySystem() {
        if (window.BoundaryManager) {
            this.boundaryManager = new window.BoundaryManager(this);
            this.boundaryManager.initialize();
        }
    }
    
    integrateCoordinateSystem() {
        if (window.CoordinateManager) {
            this.coordinateManager = new window.CoordinateManager(this);
            this.coordinateManager.initialize();
        }
    }
    
    // PixiJS管理機能のみ
    // 境界・座標機能は削除
}
```

### BoundaryManager設計
```javascript
// boundary-manager.js (新規)
class BoundaryManager {
    constructor(appCore) {
        this.appCore = appCore;
        this.config = window.ConfigManager.get('canvas.boundary');
        this.isTrackingOutside = false;
        this.boundaryMargin = 20;
    }
    
    initialize() {
        this.setupGlobalTracking();
        this.setupPixiJSIntegration();
        this.setupEventBusIntegration();
        
        window.EventBus.safeEmit('boundary.manager.initialized', {
            timestamp: Date.now()
        });
    }
    
    setupGlobalTracking() {
        document.addEventListener('pointerdown', this.handleGlobalPointerDown.bind(this));
        document.addEventListener('pointermove', this.handleGlobalPointerMove.bind(this));
        document.addEventListener('pointerup', this.handleGlobalPointerUp.bind(this));
    }
    
    handleBoundaryCrossIn(event, canvasRect) {
        // 境界越え処理実装
        const canvasX = event.clientX - canvasRect.left;
        const canvasY = event.clientY - canvasRect.top;
        
        window.EventBus.safeEmit('boundary.cross.in', {
            position: { x: canvasX, y: canvasY },
            pressure: event.pressure || 0.5,
            pointerId: event.pointerId
        });
    }
}
```

### CoordinateManager設計
```javascript
// coordinate-manager.js (新規)
class CoordinateManager {
    constructor(appCore) {
        this.appCore = appCore;
        this.config = window.ConfigManager.get('canvas.coordinates');
        this.transformCache = new Map();
    }
    
    initialize() {
        this.setupTransformationSystem();
        this.setupCaching();
        
        window.EventBus.safeEmit('coordinate.manager.initialized', {
            timestamp: Date.now()
        });
    }
    
    canvasToPixi(x, y) {
        const cacheKey = `${x}-${y}`;
        if (this.transformCache.has(cacheKey)) {
            return this.transformCache.get(cacheKey);
        }
        
        const result = { x, y }; // 基本変換
        this.transformCache.set(cacheKey, result);
        return result;
    }
    
    globalToCanvas(globalX, globalY) {
        const canvas = this.appCore.getCanvas();
        if (!canvas) return { x: globalX, y: globalY };
        
        const rect = canvas.getBoundingClientRect();
        return {
            x: globalX - rect.left,
            y: globalY - rect.top
        };
    }
}
```
