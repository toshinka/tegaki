/**
 * 🎨 UIManager - UI系統括制御
 * 責務: 
 * - ツールボタン制御（@pixi/ui Button活用）
 * - ポップアップ管理（@pixi/ui Popup活用）
 * - スライダー制御（@pixi/ui Slider活用）  
 * - ステータスバー更新（@pixi/ui Text/Label活用）
 * - プリセット表示管理
 * 
 * 移行元: ui-manager.js の全機能（@pixi/uiで再実装し、車輪の再発明を抑制）
 * 依存: app-core.js (AppCore instance), @pixi/ui (CDN経由)
 */

class UIManager {
    constructor(appCore) {
        this.appCore = appCore;
        // @pixi/ui コンポーネントの初期化
        this.uiContainer = new PIXI.Container(); // UIルートコンテナ
    }
    
    async init() {
        // @pixi/ui を活用したUI構築
        this.setupToolButtons(); // Buttonコンポーネント使用
        this.setupSliders();     // Sliderコンポーネント使用
        this.setupPopups();      // Popupコンポーネント使用
        // 既存機能の移植（@pixi/uiで置き換え）
    }
    
    setupToolButtons() {
        // 例: @pixi/ui Button でツールボタン作成
        const penButton = new PIXI.UI.Button({
            text: 'Pen',
            onClick: () => this.appCore.managers.tool.setTool('pen')
        });
        this.uiContainer.addChild(penButton);
    }
    
    // 同様に他のUI要素を@pixi/uiで実装
}

After（軽量・200行以下）javascript

class AppCore {
    constructor() {
        this.managers = {};
    }
    
    async init() {
        await this.initializePixiJS();
        await this.initializeManagers(); // ← Manager統括のみ
        this.setupGlobalEvents();
    }
    
    async initializeManagers() {
        this.managers.ui = new UIManager(this); // @pixi/ui 活用でUI軽量化
        this.managers.tool = new ToolManager(this);
        this.managers.canvas = new CanvasManager(this);
        this.managers.memory = new MemoryManager(this);
        this.managers.settings = new SettingsManager(this);
        
        // 全Manager初期化
        await Promise.all(
            Object.values(this.managers).map(m => m.init())
        );
    }
    
    // 最小限の共通機能のみ残存
    getLocalPointerPosition(event) { ... }
    emit(eventName, data) { ... }
    getStats() { ... }
}
