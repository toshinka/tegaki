```javascript
class DataManager {
    static async saveDrawing(appCore) {
        const drawingData = {
            paths: appCore.paths,
            canvasSize: {
                width: appCore.canvasWidth,
                height: appCore.canvasHeight
            },
            timestamp: Date.now()
        };
        
        // ローカルストレージ版（fetch準備）
        localStorage.setItem('drawing_latest', JSON.stringify(drawingData));
        console.log('✅ 描画データ保存完了');
    }
    
    static async loadDrawing(appCore) {
        const saved = localStorage.getItem('drawing_latest');
        if (saved) {
            const drawingData = JSON.parse(saved);
            // appCore への復元処理
            return drawingData;
        }
        return null;
    }
}
```
