```javascript
// js/utils/icon-manager.js 修正版
/**
 * 🎯 AI_WORK_SCOPE: @tabler/icons統合・フォールバック・最適化
 * 🎯 DEPENDENCIES: なし（独立ユーティリティ）
 * 🎯 NODE_MODULES: @tabler/icons-react
 * 🎯 車輪の再発明回避: @tabler活用・SVG最適化・キャッシュシステム
 */

class IconManager {
    constructor() {
        this.tablerIcons = null;
        this.iconCache = new Map();
        this.loadTablerIcons();
    }
    
    // @tabler/icons from node_modules読み込み
    loadTablerIcons() {
        try {
            // node_modules/から@tabler/icons-react読み込み
            const tablerPath = './node_modules/@tabler/icons-react/icons-sprite.svg';
            this.loadTablerSprite(tablerPath);
        } catch (error) {
            console.warn('⚠️ @tabler/icons未検出 - 内蔵アイコン使用');
            this.initializeFallbackIcons();
        }
    }
    
    // 指定アイコン取得（@tabler優先・フォールバック対応）
    getIcon(iconName, options = {}) {
        // 特定要求アイコン対応
        const iconMappings = {
            'bucket': 'bucket',           // 塗りつぶし用バケツ
            'selection': 'select',        // 選択範囲用破線□
            'pen': 'pencil',
            'eraser': 'eraser'
        };
        
        const tablerName = iconMappings[iconName] || iconName;
        
        // @tabler/icons優先
        if (this.hasTablerIcon(tablerName)) {
            return this.getTablerIcon(tablerName, options);
        }
        
        // フォールバック: 内蔵アイコン
        return this.getBuiltinIcon(iconName, options);
    }
    
    // バケツアイコン（@tabler/icons bucket使用）
    getBucketIcon(options = {}) {
        const bucketIcon = this.getIcon('bucket', options);
        return bucketIcon || this.getBuiltinBucketIcon();
    }
    
    // 選択範囲アイコン（破線□・@tabler/icons select使用）
    getSelectionIcon(options = {}) {
        let icon = this.getIcon('selection', options);
        
        // 破線効果追加
        if (icon) {
            icon = icon.replace('<svg', '<svg stroke-dasharray="4 2"');
        }
        
        return icon || this.getBuiltinSelectionIcon();
    }
}
```

#### 📏 リサイズ機能実装
```javascript
// managers/canvas-manager.js 修正・追加
/**
 * 🎯 AI_WORK_SCOPE: キャンバスリサイズ・中央寄せ・履歴管理
 * 🎯 DEPENDENCIES: js/main.js, js/app-core.js, managers/memory-manager.js
 * 🎯 NODE_MODULES: pixi.js（Application, Container）
 * 🎯 PIXI_EXTENSIONS: なし（基本PIXI機能のみ）
 */

class CanvasManager {
    constructor(appCore) {
        this.appCore = appCore;
        this.currentSize = { width: 400, height: 400 };
        this.history = [];
        this.setupResizeHandlers();
    }
    
    // リサイズ機能有効化
    setupResizeHandlers() {
        // リサイズボタン有効化
        const resizeTool = document.getElementById('resize-tool');
        resizeTool.classList.remove('disabled');
        
        // 適用ボタンイベント
        const applyResize = document.getElementById('apply-resize');
        const applyCenterResize = document.getElementById('apply-resize-center');
        
        applyResize?.addEventListener('click', () => this.applyResize(false));
        applyCenterResize?.addEventListener('click', () => this.applyResize(true));
        
        // プリセットボタン
        document.querySelectorAll('.resize-button[data-size]').forEach(button => {
            button.addEventListener('click', (e) => {
                const [width, height] = e.target.dataset.size.split(',').map(Number);
                this.setCanvasSize(width, height);
            });
        });
    }
    
    // キャンバスサイズ変更（非破壊）
    applyResize(centerContent = false) {
        const widthInput = document.getElementById('canvas-width');
        const heightInput = document.getElementById('canvas-height');
        
        const newWidth = parseInt(widthInput.value);
        const newHeight = parseInt(heightInput.value);
        
        if (!this.validateSize(newWidth, newHeight)) {
            this.showResizeError('無効なサイズです');
            return false;
        }
        
        // 履歴保存（アンドゥ対応）
        this.saveResizeHistory();
        
        // サイズ変更実行
        this.executeResize(newWidth, newHeight, centerContent);
        
        console.log(`✅ キャンバスサイズ変更: ${this.currentSize.width}×${this.currentSize.height} → ${newWidth}×${newHeight}`);
        return true;
    }
    
    // リサイズ実行（PIXI Application対応）
    executeResize(newWidth, newHeight, centerContent) {
        const app = this.appCore.app;
        
        // 現在の描画内容位置記録
        const contentBounds = this.getContentBounds();
        
        // PixiJS Application リサイズ
        app.renderer.resize(newWidth, newHeight);
        
        // キャンバス要素サイズ更新
        app.view.style.width = newWidth + 'px';
        app.view.style.height = newHeight + 'px';
        
        // 中央寄せ処理
        if (centerContent && contentBounds) {
            this.centerExistingContent(contentBounds, newWidth, newHeight);
        }
        
        // サイズ情報更新
        this.updateSizeInfo(newWidth, newHeight);
        
        // UI更新
        this.updateCanvasUI(newWidth, newHeight);
    }
    
    // 中央寄せ処理（非破壊）
    centerExistingContent(contentBounds, newWidth, newHeight) {
        const offsetX = (newWidth - this.currentSize.width) / 2;
        const offsetY = (newHeight - this.currentSize.height) / 2;
        
        // 全レイヤーを移動
        this.appCore.layers?.forEach(layer => {
            layer.container.x += offsetX;
            layer.container.y += offsetY;
        });
    }
    
    // サイズバリデーション
    validateSize(width, height) {
        return width >= 100 && width <= 4096 && 
               height >= 100 && height <= 4096 &&
               !isNaN(width) && !isNaN(height);
    }
    
    // リサイズ履歴保存（アンドゥ対応）
    saveResizeHistory() {
        const historyEntry = {
            type: 'canvas-resize',
            beforeSize: { ...this.currentSize },
            timestamp: Date.now()
        };
        
        // Memory Manager を活用
        if (window.MemoryManager) {
            window.MemoryManager.record('canvas-resize', historyEntry);
        }
    }
    
    // UI更新（ステータス表示等）
    updateCanvasUI(width, height) {
        this.currentSize = { width, height };
        
        // ステータス表示更新
        const canvasInfo = document.getElementById('canvas-info');
        if (canvasInfo) {
            canvasInfo.textContent = `${width}×${height}px`;
        }
        
        // 入力欄更新
        const widthInput = document.getElementById('canvas-width');
        const heightInput = document.getElementById('canvas-height');
        if (widthInput) widthInput.value = width;
        if (heightInput) heightInput.value = height;
    }
}
```

### 🎯 Phase1.3-STEP1: ベクター変形システム構築

**作業目標**: アニメ制作品質保証・完全復元可能変形・メモリ効率化

#### 🔄 非破壊変形実装
```javascript
// js/transform/vector-transform.js（新規作成）
/**
 * 🎯 AI_WORK_SCOPE: ベクター非破壊変形・履歴管理・品質保証
 * 🎯 DEPENDENCIES: managers/memory-manager.js, js/app-core.js
 * 🎯 NODE_MODULES: pixi.js（Matrix, Transform, Graphics）
 * 🎯 PIXI_EXTENSIONS: なし（基本PIXI Transform API）
 * 🎯 車輪の再発明回避: PIXI Transform API活用・数学ライブラリ活用
 */

class VectorTransform {
    constructor() {
        this.version = 'v1.0-Phase1.3';
        this.transformHistory = new Map();
        this.qualityThreshold = 0.001; // 品質劣化検出閾値
    }
    
    /**
     * 🔄 非破壊回転変形（完全復元可能）
     */
    rotate(target, angle, options = {}) {
        const transformId = this.generateTransformId(target);
        
        // 現在状態保存
        this.saveOriginalState(target, transformId);
        
        // 回転行列作成（PIXI Matrix活用）
        const rotationMatrix = new PIXI.Matrix();
        rotationMatrix.rotate(angle);
        
        // 基準点調整（中央基準・カスタム基準点対応）
        const pivot = options.pivot || this.getCenter(target);
        rotationMatrix.translate(-pivot.x, -pivot.y);
        rotationMatrix.rotate(angle);
        rotationMatrix.translate(pivot.x, pivot.y);
        
        // 非破壊適用
        this.applyTransformMatrix(target, rotationMatrix, 'rotate');
        
        // 品質検証
        this.validateTransformQuality(target, transformId);
        
        console.log(`🔄 非破壊回転適用: ${angle}rad (${(angle * 180 / Math.PI).toFixed(1)}°)`);
        
        return transformId;
    }
    
    /**
     * 📏 非破壊スケール変形（品質保持）
     */
    scale(target, scaleX, scaleY = scaleX, options = {}) {
        const transformId = this.generateTransformId(target);
        
        // 現在状態保存
        this.saveOriginalState(target, transformId);
        
        // スケール行列作成
        const scaleMatrix = new PIXI.Matrix();
        const pivot = options.pivot || this.getCenter(target);
        
        scaleMatrix.translate(-pivot.x, -pivot.y);
        scaleMatrix.scale(scaleX, scaleY);
        scaleMatrix.translate(pivot.x, pivot.y);
        
        // 非破壊適用
        this.applyTransformMatrix(target, scaleMatrix, 'scale');
        
        // ベクター品質保証
        if (target instanceof PIXI.Graphics) {
            this.preserveVectorQuality(target, scaleX, scaleY);
        }
        
        console.log(`📏 非破壊スケール適用: ${scaleX}x, ${scaleY}y`);
        
        return transformId;
    }
    
    /**
     * ↔️ 非破壊移動変形（座標精度保持）
     */
    translate(target, deltaX, deltaY, options = {}) {
        const transformId = this.generateTransformId(target);
        
        // 現在状態保存
        this.saveOriginalState(target, transformId);
        
        // 移動行列作成
        const translateMatrix = new PIXI.Matrix();
        translateMatrix.translate(deltaX, deltaY);
        
        // 非破壊適用
        this.applyTransformMatrix(target, translateMatrix, 'translate');
        
        console.log(`↔️ 非破壊移動適用: (${deltaX}, ${deltaY})`);
        
        return transformId;
    }
    
    /**
     * 🎯 変形行列適用（非破壊・履歴対応）
     */
    applyTransformMatrix(target, matrix, transformType) {
        // 現在の変形に追加適用（累積）
        const currentMatrix = target.transform.localTransform;
        const newMatrix = currentMatrix.clone();
        newMatrix.append(matrix);
        
        // 行列適用
        target.transform.setFromMatrix(newMatrix);
        
        // 履歴記録（Memory Manager活用）
        if (window.MemoryManager) {
            window.MemoryManager.record('transform', {
                targetId: target.id || target.name,
                type: transformType,
                matrix: newMatrix.clone(),
                timestamp: Date.now(),
                reversible: true
            });
        }
    }
    
    /**
     * 💾 原状態保存（完全復元用）
     */
    saveOriginalState(target, transformId) {
        if (this.transformHistory.has(transformId)) return;
        
        const originalState = {
            transform: target.transform.localTransform.clone(),
            position: { x: target.x, y: target.y },
            scale: { x: target.scale.x, y: target.scale.y },
            rotation: target.rotation,
            pivot: { x: target.pivot.x, y: target.pivot.y },
            anchor: target.anchor ? { x: target.anchor.x, y: target.anchor.y } : null,
            
            // ベクター品質保持用
            originalGeometry: target.geometry ? this.cloneGeometry(target.geometry) : null,
            
            timestamp: Date.now()
        };
        
        this.transformHistory.set(transformId, originalState);
        
        console.log(`💾 原状態保存完了: ${transformId}`);
    }
    
    /**
     * ↩️ 変形リセット（完全復元）
     */
    resetTransform(target, transformId) {
        const originalState = this.transformHistory.get(transformId);
        
        if (!originalState) {
            console.warn('⚠️ 復元データが見つかりません:', transformId);
            return false;
        }
        
        // 完全復元
        target.transform.setFromMatrix(originalState.transform);
        target.position.set(originalState.position.x, originalState.position.y);
        target.scale.set(originalState.scale.x, originalState.scale.y);
        target.rotation = originalState.rotation;
        target.pivot.set(originalState.pivot.x, originalState.pivot.y);
        
        if (originalState.anchor && target.anchor) {
            target.anchor.set(originalState.anchor.x, originalState.anchor.y);
        }
        
        // ベクター品質復元
        if (originalState.originalGeometry && target.geometry) {
            this.restoreGeometry(target, originalState.originalGeometry);
        }
        
        console.log(`↩️ 変形リセット完了: ${transformId}`);
        
        return true;
    }
    
    /**
     * 🔍 変形品質検証（品質劣化検出）
     */
    validateTransformQuality(target, transformId) {
        const originalState = this.transformHistory.get(transformId);
        if (!originalState) return true;
        
        let qualityScore = 1.0;
        
        // ベクター品質チェック
        if (target instanceof PIXI.Graphics && originalState.originalGeometry) {
            qualityScore = this.calculateVectorQuality(target.geometry, originalState.originalGeometry);
        }
        
        // 変形精度チェック
        const transformAccuracy = this.calculateTransformAccuracy(target);
        qualityScore *= transformAccuracy;
        
        // 品質警告
        if (qualityScore < this.qualityThreshold) {
            console.warn(`⚠️ 品質劣化検出: ${transformId} (${(qualityScore * 100).toFixed(2)}%)`);
        }
        
        return qualityScore;
    }
    
    /**
     * 📊 ベクター品質計算
     */
    calculateVectorQuality(currentGeometry, originalGeometry) {
        // 点数比較
        const currentPoints = currentGeometry.points?.length || 0;
        const originalPoints = originalGeometry.points?.length || 0;
        
        if (originalPoints === 0) return 1.0;
        
        // 点数維持率
        const pointRetention = Math.min(currentPoints / originalPoints, 1.0);
        
        // 形状精度（簡易）
        let shapeAccuracy = 1.0;
        if (currentGeometry.bounds && originalGeometry.bounds) {
            const widthRatio = currentGeometry.bounds.width / originalGeometry.bounds.width;
            const heightRatio = currentGeometry.bounds.height / originalGeometry.bounds.height;
            shapeAccuracy = Math.min(widthRatio, heightRatio);
        }
        
        return pointRetention * shapeAccuracy;
    }
    
    /**
     * 🎯 中央座標取得（ピボット計算）
     */
    getCenter(target) {
        const bounds = target.getBounds();
        return {
            x: bounds.x + bounds.width / 2,
            y: bounds.y + bounds.height / 2
        };
    }
    
    /**
     * 🆔 変形ID生成
     */
    generateTransformId(target) {
        const timestamp = Date.now();
        const targetId = target.id || target.name || 'unknown';
        return `transform_${targetId}_${timestamp}`;
    }
    
    /**
     * 📋 変形統計取得
     */
    getTransformStats() {
        return {
            totalTransforms: this.transformHistory.size,
            memoryUsage: this.calculateMemoryUsage(),
            averageQuality: this.calculateAverageQuality(),
            version: this.version
        };
    }
```

```javascript
     */
    recordTransform(transformData) {

        }

        if (this.getCurrentMemoryUsage() > this.maxMemoryMB) {

        }

            data: this.optimizeTransformData(transformData),
            timestamp: Date.now(),
            memorySize: this.calculateDataSize(transformData)
        };
        
        this.history.set(recordId, historyEntry);
        this.undoStack.push(recordId);
        
        // redo履歴クリア（新しい操作により分岐）
        this.redoStack = [];

            return false;
        }

        const historyEntry = this.history.get(recordId);
        
        if (!historyEntry) {
            console.error('❌ 履歴エントリが見つかりません:', recordId);
            return false;
        }
        
        // 変形リセット実行
        const success = this.executeUndo(historyEntry);
        
        if (success) {
            // redo履歴に追加

        const currentTime = Date.now();

        let deletedCount = 0;
        
        if (this.hasLodash) {
            // lodash活用で効率的な削除
            const oldEntries = _.filter(Array.from(this.history.entries()), 
                ([id, entry]) => currentTime - entry.timestamp > maxAge
            );
            
            oldEntries.forEach(([id]) => {
                this.history.delete(id);
                deletedCount++;
            });
        } else {
            // 基本実装
            for (const [id, entry] of this.history) {
                if (currentTime - entry.timestamp > maxAge) {
                    this.history.delete(id);
                    deletedCount++;
                }
            }
        }
        
        console.log(`🧹 履歴クリーンアップ完了 - ${deletedCount}個削除`);
    }

     */
    performEmergencyGC() {
        console.warn('🗑️ 緊急ガベージコレクション実行...');
        
        const beforeSize = this.history.size;
        const beforeMemory = this.getCurrentMemoryUsage();
        
        // 古い履歴の強制削除
        const cutoffTime = Date.now() - 300000; // 5分以上古い
        
        for (const [id, entry] of this.history) {
            if (entry.timestamp < cutoffTime) {
                this.history.delete(id);
            }
        }
        
        // undo/redo スタックサイズ制限
        if (this.undoStack.length > 50) {
            this.undoStack = this.undoStack.slice(-50);
        }
        if (this.redoStack.length > 50) {
            this.redoStack = this.redoStack.slice(-50);
        }
        
        // ブラウザGC要求
        if (window.gc) {
            window.gc();
        }
        
        const afterSize = this.history.size;
        const afterMemory = this.getCurrentMemoryUsage();
        
        this.memoryStats.gcCount++;
        this.memoryStats.lastGC = Date.now();
        
        console.log(`🗑️ 緊急GC完了 - 履歴: ${beforeSize}→${afterSize}, メモリ: ${beforeMemory}→${afterMemory}MB`);
    }
    
    /**
     * 📊 現在メモリ使用量取得
     */
    getCurrentMemoryUsage() {
        // performance.memory活用（Chrome系）
        if (performance.memory) {
            const usedMB = performance.memory.usedJSHeapSize / 1024 / 1024;
            this.memoryStats.currentUsage = usedMB;
            
            if (usedMB > this.memoryStats.peakUsage) {
                this.memoryStats.peakUsage = usedMB;
            }
            
            return usedMB;
        }
        
        // フォールバック: 履歴サイズベース推定
        let estimatedSize = 0;
        for (const [id, entry] of this.history) {
            estimatedSize += entry.memorySize || 1024; // 1KB基準
        }
        
        return estimatedSize / 1024 / 1024; // MB変換
    }
    
    /**
     * 📈 メモリ統計取得
     */
    getMemoryStats() {
        return {
            ...this.memoryStats,
            historySize: this.history.size,
            undoStackSize: this.undoStack.length,
            redoStackSize: this.redoStack.length,
            maxHistorySize: this.maxHistorySize,
            maxMemoryMB: this.maxMemoryMB,
            version: this.version
        };
    }
```

### 🛠️ トラブルシューティングガイド

#### ❌ よくある問題と解決法

**問題1**: @tabler/iconsが読み込まれない
```javascript
// 解決法: フォールバック確認
if (!window.TablerIcons) {
    console.warn('⚠️ @tabler/icons未検出 - 内蔵アイコン使用');
    this.useBuiltinIcons = true;
}
```

**問題2**: 変形後の品質劣化
```javascript
// 解決法: 変形前状態保存
const originalGeometry = target.geometry.clone();
// 変形実行
// 品質確認
if (qualityScore < threshold) {
    target.geometry = originalGeometry; // 復元
}
```

**問題3**: メモリリーク
```javascript
// 解決法: 適切なクリーンアップ
// WeakMapの活用
this.transformHistory = new WeakMap();
// 定期GC
setInterval(() => this.performGC(), 30000);
```

**問題4**: AI分業時の依存関係エラー
```javascript
// 解決法: 依存関係チェック
const checkDependencies = () => {
    const required = ['AppCore', 'MemoryManager', 'VectorTransform'];
    const missing = required.filter(dep => !window[dep]);
    if (missing.length > 0) {
        console.error('❌ 依存関係不足:', missing);
        return false;
    }
    return true;
};
```
