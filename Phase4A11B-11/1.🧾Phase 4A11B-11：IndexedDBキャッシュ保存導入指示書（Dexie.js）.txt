📥 Step 2. IndexedDB（db-indexed.js）
js
コピーする
編集する
const db = new Dexie("TegakiProjectDB");
db.version(1).stores({
  layers: "++id, name, imageData"
});

export async function saveLayerToIndexedDB(layerId, name, dataURL) {
  await db.layers.put({ id: layerId, name, imageData: dataURL });
}

export async function loadLayersFromIndexedDB() {
  return await db.layers.toArray();
}
🎨 Step 3. 復元処理（core-engine.js）

コピーする
編集する
const layers = await loadLayersFromIndexedDB();
for (const layer of layers) {
  layerManager.createLayer(layer.id, layer.name);
  const ctx = canvasManager.getLayerContext(layer.id);
  if (ctx) {
    const img = new Image();
    img.onload = () => {
      ctx.imageSmoothingEnabled = false;
      ctx.drawImage(img, 0, 0);
    };
    img.src = layer.imageData;
  }
}
layerManager.setCurrentLayer(layers.at(-1)?.id ?? 0);
toolManager.setTool("pen"); // 初期ツール指定

export function copyLayerWithoutSmoothing(sourceCtx, destCtx, width, height) {
  const imageData = sourceCtx.getImageData(0, 0, width, height);
  destCtx.putImageData(imageData, 0, 0);
}
この関数は core/utils/transfer-utils.js などに切り出しておくと安全。

✅ 安定化ポイント（必ず守る）
項目	やること
