2. core/db/db-indexed.js を新規作成

コピーする
編集する
const db = new Dexie("TegakiProjectDB");
db.version(1).stores({
  layers: "++id, name, imageData"
});

export async function saveLayerToIndexedDB(layerId, name, dataURL) {
  await db.layers.put({ id: layerId, name, imageData: dataURL });
}

export async function loadLayersFromIndexedDB() {
  return await db.layers.toArray();
}
3. core-engine.js を再構成して初期化を確実に行う
js
コピーする
編集する
import { CanvasManager } from './core/rendering/canvas-manager.js';
import { LayerManager } from './layer-manager/layer-manager.js';
import { ToolManager } from './tools/tool-manager.js';
import { LayerUIManager } from './ui/layer-ui-manager.js';
import { loadLayersFromIndexedDB, saveLayerToIndexedDB } from './core/db/db-indexed.js';

window.addEventListener('load', async () => {
  console.log("🛠️ アプリケーションの初期化を開始します...");

  const canvasManager = new CanvasManager();
  const layerManager = new LayerManager(canvasManager);
  const toolManager = new ToolManager({ canvasManager });
  const layerUIManager = new LayerUIManager(layerManager);

  // 初期ツールを必ず設定
  toolManager.setTool('pen');

  // IndexedDBから復元（描画エンジン初期化後に行うこと！）
  console.log("💾 IndexedDBからレイヤーデータを復元します。");
  const layers = await loadLayersFromIndexedDB();
  for (const layer of layers) {
    layerManager.createLayer(layer.id, layer.name);
    canvasManager.drawFromDataURL(layer.imageData, layer.id);
  }

  // UI反映（未定義だとクラッシュするので必ず定義すること！）
  if (typeof layerUIManager.renderLayerList === 'function') {
    layerUIManager.renderLayerList();
  } else {
    console.warn("⚠️ layerUIManager.renderLayerList() が未定義です！");
  }

  console.log("✅ アプリケーションの初期化が完了しました。");
});
