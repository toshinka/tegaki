    // ピクセルパーフェクト描画（PixiJS Graphics最適化）
    drawStroke(points, toolConfig) {
        const graphics = new PIXI.Graphics();
        
        graphics.lineStyle({
            width: toolConfig.size,
            color: toolConfig.color,
            alpha: toolConfig.opacity,
            cap: PIXI.LINE_CAP.ROUND,
            join: PIXI.LINE_JOIN.ROUND,
            native: this.app.renderer.type === PIXI.RENDERER_TYPE.WEBGPU
        });
        
        // スムーズ曲線描画（PixiJS最適化）
        if (points.length > 0) {
            graphics.moveTo(points[0].x, points[0].y);
            
            for (let i = 1; i < points.length - 1; i++) {
                const xc = (points[i].x + points[i + 1].x) / 2;
                const yc = (points[i].y + points[i + 1].y) / 2;
                graphics.quadraticCurveTo(points[i].x, points[i].y, xc, yc);
            }
        }
        
        return graphics;

class ColorProcessor {
    constructor() {
        this.chroma = chroma; // Chroma.js活用
        this.futabaColors = this.initializeFutabaColors();
        this.hsv = { h: 0, s: 100, v: 100 };
    }
    
    initializeFutabaColors() {
        return {
            maroon: '#800000',
            lightMaroon: '#aa5a56',
            medium: '#cf9c97',
            cream: '#f0e0d6',
            background: '#ffffee'
        };
    }
    
    // PixiJS Color統合
    pixiColorFromHSV(h, s, v) {
        const hex = this.chroma.hsv(h, s / 100, v / 100).hex();
        return parseInt(hex.replace('#', ''), 16);
    }
    
    // HSV円形ピッカー（PixiJS Graphics活用）
    createHSVPicker(pixiContainer) {
        const radius = 60;
        const wheel = new PIXI.Graphics();
        
        // 色相環描画
        for (let angle = 0; angle < 360; angle++) {
            const color = this.pixiColorFromHSV(angle, 100, 100);
            wheel.beginFill(color);
            wheel.arc(60, 60, radius, 
                (angle - 1) * Math.PI / 180, 
                angle * Math.PI / 180);
            wheel.endFill();
        }
        
        pixiContainer.addChild(wheel);
        return wheel;
    }
}

class PixiLayerProcessor {
    constructor(pixiApp) {
        this.app = pixiApp;
        this.coordinate = new PixiCoordinateUnifier(pixiApp);
        this.layers = new Map();
        this.layerContainer = new PIXI.Container();
        this.app.stage.addChild(this.layerContainer);
    }
    
    // 非破壊レイヤー作成（PixiJS Container活用）
    createLayer(layerId, parentId = null) {
        const layer = {
            id: layerId,
            container: new PIXI.Container(),
            vectorStrokes: [],
            metadata: {
                name: 'Layer',
                opacity: 1.0,
                blendMode: PIXI.BLEND_MODES.NORMAL,
                visible: true,
                created: Date.now()
            }
        };
        
        layer.container.name = layerId;
        
        const parent = parentId ? 
            this.layers.get(parentId).container : 
            this.layerContainer;
        
        parent.addChild(layer.container);
        this.layers.set(layerId, layer);
        
        return layer;
    }
    
    // レイヤー合成（PixiJS RenderTexture活用）
    compositeLayer(layerId) {
        const layer = this.layers.get(layerId);
        
        const renderTexture = PIXI.RenderTexture.create({
            width: this.app.view.width,
            height: this.app.view.height,
            resolution: this.app.renderer.resolution
        });
        
        this.app.renderer.render(layer.container, { renderTexture });
        return renderTexture;
    }
    
    // サムネイル生成（PixiJS Extract活用）
    async generateThumbnail(layerId) {
        const layer = this.layers.get(layerId);
        
        const renderTexture = PIXI.RenderTexture.create({
            width: 64, height: 64,
            resolution: 1
        });
        
        this.app.renderer.render(layer.container, { renderTexture });
        
        const canvas = this.app.renderer.extract.canvas(renderTexture);
        const blob = await new Promise(resolve => {
            canvas.toBlob(resolve, 'image/png');
        });
        
        renderTexture.destroy(true);
        return blob;
    }
}
```

class CanvasController {
    constructor(pixiApp) {
        this.app = pixiApp;
        this.coordinate = new PixiCoordinateUnifier(pixiApp);
        
        this.viewTransform = {
            x: 0, y: 0,
            scale: 1.0,
            rotation: 0
        };
        
        this.setupPixiCamera();
    }
    
    setupPixiCamera() {
        // PixiJS統一座標でのカメラ制御
        this.cameraContainer = new PIXI.Container();
        this.app.stage.addChild(this.cameraContainer);
        
        // パン・ズーム・回転対応
        this.setupViewControls();
    }
    
    // パン操作（PixiJS統一座標）
    panView(deltaX, deltaY) {
        this.viewTransform.x += deltaX;
        this.viewTransform.y += deltaY;
        this.updateCameraTransform();
    }
    
    // ズーム操作（PixiJS統一座標）
    zoomView(scaleFactor, centerX, centerY) {
        const newScale = Math.max(0.1, Math.min(10, this.viewTransform.scale * scaleFactor));
        
        // ズーム中心点計算（PixiJS統一座標）
        const pivotX = centerX - this.viewTransform.x;
        const pivotY = centerY - this.viewTransform.y;
        
        this.viewTransform.x -= pivotX * (newScale / this.viewTransform.scale - 1);
        this.viewTransform.y -= pivotY * (newScale / this.viewTransform.scale - 1);
        this.viewTransform.scale = newScale;
        
        this.updateCameraTransform();
    }
    
    updateCameraTransform() {
        this.cameraContainer.position.set(this.viewTransform.x, this.viewTransform.y);
        this.cameraContainer.scale.set(this.viewTransform.scale);
        this.cameraContainer.rotation = this.viewTransform.rotation;
    }
}
```
