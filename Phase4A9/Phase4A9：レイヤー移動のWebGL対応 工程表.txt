🎯 目的
レイヤー移動機能をWebGL描画と連携させつつ、描画点の入力には影響を与えない構造を作る。

🧱 1. 行列構成
plaintext
コピーする
編集する
- projectionMatrix: 正射影 + Y軸反転 (scaleY = -1)
- viewMatrix: カメラズーム/パン用
- modelMatrix: レイヤー固有の移動・回転・スケーリング
- 最終行列: MVP = projection × view × model
🧠 2. 実装要点
機能	実装方針
描画時	MVP行列で shader に描画。modelMatrix はレイヤーごとに使う。
ペンの当たり判定	convertCanvasToLayerCoords() は viewMatrix の逆行列のみ使用。modelMatrix は使わない。
Y軸方向のズレ補正	projectionMatrix に mat4.scale(..., [1, -1, 1]) を加える。
UI側のレイヤー移動	レイヤーの .position や .rotation を modelMatrix に反映する処理を書く。

🧪 3. テスト項目
 レイヤーを動かしても、描画座標（筆先）はズレない

 レイヤーの見た目は動く（表示としての移動は成功）

 上下移動が反転せず、自然に操作できる

参考テンプレ
// === core-engine.js ===

const camera = {
  x: 0,
  y: 0,
  scale: 1,
};

const viewMatrix = mat4.create();
const projectionMatrix = mat4.create();

function updateViewMatrix() {
  mat4.identity(viewMatrix);
  mat4.translate(viewMatrix, viewMatrix, [-camera.x, -camera.y, 0]);
  mat4.scale(viewMatrix, viewMatrix, [camera.scale, camera.scale, 1]);
}

function updateProjectionMatrix() {
  mat4.ortho(projectionMatrix, -1, 1, -1, 1, -1, 1);
  mat4.scale(projectionMatrix, projectionMatrix, [1, -1, 1]); // Y反転
}

function convertCanvasToLayerCoords(x, y) {
  const nx = (2 * x) / canvas.width - 1;
  const ny = 1 - (2 * y) / canvas.height;
  const invView = mat4.create();
  if (!mat4.invert(invView, viewMatrix)) return null;

  const clipPos = vec4.fromValues(nx, ny, 0, 1);
  vec4.transformMat4(clipPos, clipPos, invView);
  return { x: clipPos[0], y: clipPos[1] };
}

// === rendering-bridge.js ===

function renderAllLayers(layers) {
  updateViewMatrix();
  updateProjectionMatrix();
  layers.forEach(layer => {
    const mvpMatrix = mat4.create();
    mat4.multiply(mvpMatrix, projectionMatrix, viewMatrix);
    mat4.multiply(mvpMatrix, mvpMatrix, layer.modelMatrix);
    drawLayerToFramebuffer(layer, mvpMatrix);
  });
}

// === webgl-engine.js ===

function drawLayerToFramebuffer(layer, mvpMatrix) {
  gl.useProgram(shaderProgram);

  gl.bindBuffer(gl.ARRAY_BUFFER, layer.vertexBuffer);
  gl.enableVertexAttribArray(shaderProgram.attribs.a_position);
  gl.vertexAttribPointer(shaderProgram.attribs.a_position, 2, gl.FLOAT, false, 0, 0);

  gl.activeTexture(gl.TEXTURE0);
  gl.bindTexture(gl.TEXTURE_2D, layer.texture);
  gl.uniform1i(shaderProgram.uniforms.u_texture, 0);

  gl.uniformMatrix4fv(shaderProgram.uniforms.u_mvpMatrix, false, mvpMatrix);

  gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
}

// === vertex shader (GLSL) ===
/*
attribute vec2 a_position;
uniform mat4 u_mvpMatrix;

void main() {
  gl_Position = u_mvpMatrix * vec4(a_position, 0.0, 1.0);
}
*/

function moveLayer(layer, deltaX, deltaY) {
  // 単純な平行移動だけ行う
  mat4.translate(layer.modelMatrix, layer.modelMatrix, [deltaX, deltaY, 0]);
  layer.gpuDirty = true;  // 再描画フラグ
}

📌 前回のGEMINIの主な誤り（教訓）
誤り	教訓
modelMatrix をペン座標変換に使ってしまった	描画の見た目だけに使うこと。入力には使わない。
shaderのuniformやattribの管理が曖昧だった	GLSLとJSのバインドを明示し、名前を統一すること
MVP順序が不定だった	必ず projection × view × model の順に合成すること

✅ 実装要点（必ず満たすべき要件）
1. 座標系の責務分離
行列	用途	更新頻度
