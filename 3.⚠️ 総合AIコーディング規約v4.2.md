class PixiV8ChromeVectorLayer {
    constructor(pixiContainer = null) {
        this.container = pixiContainer || new Container();
        this.vectorStrokes = []; // 元データ（非破壊保持）
        this.graphicsObjects = []; // PixiJS v8.11.0 Graphics管理
        this.offscreenProcessor = null; // OffscreenCanvas並列処理
    }
    
    // Chrome API活用・非破壊変形
    async transformWithChromeAPI(transformMatrix) {
        const newLayer = new PixiV8ChromeVectorLayer();
        
        // OffscreenCanvas並列処理
        if (this.offscreenProcessor) {
            const transformedData = await this.offscreenProcessor.processTransform(
                this.vectorStrokes, transformMatrix
            );
            newLayer.vectorStrokes = transformedData;
        } else {
            // フォールバック・メインスレッド処理
            newLayer.vectorStrokes = this.vectorStrokes.map(stroke => ({
                ...structuredClone(stroke), // v8.11.0対応ディープコピー
                id: this.generateId(),
                points: this.transformPoints(stroke.points, transformMatrix)
            }));
        }
        
        // WebGPU最適化描画
        await newLayer.renderWithWebGPU();
        
        return newLayer; // 新レイヤー返却（元データ保持）
    }
    
    async renderWithWebGPU() {
        // WebGPU最適化描画・120FPS対応
        this.vectorStrokes.forEach(async (stroke) => {
            const graphics = new Graphics();
            
            // PixiJS v8.11.0新API・WebGPU最適化
            if (stroke.points.length > 0) {
                graphics
                    .moveTo(stroke.points[0].x, stroke.points[0].y)
                    .stroke({
                        width: stroke.size,
                        color: stroke.color,
                        alpha: stroke.opacity,
                        cap: 'round',
                        join: 'round'
                    });
                
                // WebGPU最適化スムーズ曲線
                for (let i = 1; i < stroke.points.length - 1; i++) {
                    const xc = (stroke.points[i].x + stroke.points[i + 1].x) / 2;
                    const yc = (stroke.points[i].y + stroke.points[i + 1].y) / 2;
                    graphics.quadraticCurveTo(stroke.points[i].x, stroke.points[i].y, xc, yc);
                }
            }
            
            this.container.addChild(graphics);
            this.graphicsObjects.push(graphics);
        });
    }
}
